11   export interface ReadPreferenceOption {
12     readPreference?: ReadPreferenceLike;
13   }
14   
15   export function getReadPreference(options?: ReadPreferenceOption): ReadPreference {
16     // Default to command version of the readPreference
17     let readPreference = options?.readPreference ?? ReadPreference.primary;
18     // If we have an option readPreference override the command one
19     if (options?.readPreference) {
20       readPreference = options.readPreference;
21     }
22   
23     if (typeof readPreference === 'string') {
24       readPreference = ReadPreference.fromString(readPreference);
25     }
26   
27     if (!(readPreference instanceof ReadPreference)) {
28       throw new MongoInvalidArgumentError(
29         'Option "readPreference" must be a ReadPreference instance'
30       );
31     }
32   
33     return readPreference;
34   }
35   
36   export function isSharded(topologyOrServer?: Topology | Server | Connection): boolean {
37     if (topologyOrServer == null) {
38       return false;
39     }
40   
41     if (topologyOrServer.description && topologyOrServer.description.type === ServerType.Mongos) {
42       return true;
43     }
44   
45     // NOTE: This is incredibly inefficient, and should be removed once command construction
46     //       happens based on `Server` not `Topology`.
47     if (topologyOrServer.description && topologyOrServer.description instanceof TopologyDescription) {
48       const servers: ServerDescription[] = Array.from(topologyOrServer.description.servers.values());
49       return servers.some((server: ServerDescription) => server.type === ServerType.Mongos);
50     }
51   
52     return false;
53   }
