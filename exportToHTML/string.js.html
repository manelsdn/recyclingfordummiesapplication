176   * Adds an enum validator
177   *
178   * #### Example:
179   *
180   *     const states = ['opening', 'open', 'closing', 'closed']
181   *     const s = new Schema({ state: { type: String, enum: states }})
182   *     const M = db.model('M', s)
183   *     const m = new M({ state: 'invalid' })
184   *     m.save(function (err) {
185   *       console.error(String(err)) // ValidationError: `invalid` is not a valid enum value for path `state`.
186   *       m.state = 'open'
187   *       m.save(callback) // success
188   *     })
189   *
190   *     // or with custom error messages
191   *     const enum = {
192   *       values: ['opening', 'open', 'closing', 'closed'],
193   *       message: 'enum validator failed for path `{PATH}` with value `{VALUE}`'
194   *     }
195   *     const s = new Schema({ state: { type: String, enum: enum })
196   *     const M = db.model('M', s)
197   *     const m = new M({ state: 'invalid' })
198   *     m.save(function (err) {
199   *       console.error(String(err)) // ValidationError: enum validator failed for path `state` with value `invalid`
200   *       m.state = 'open'
201   *       m.save(callback) // success
202   *     })
203   *
204   * @param {...String|Object} [args] enumeration values
205   * @return {SchemaType} this
206   * @see Customized Error Messages https://mongoosejs.com/docs/api/error.html#Error.messages
207   * @see Enums in JavaScript https://masteringjs.io/tutorials/fundamentals/enum
208   * @api public

211  SchemaString.prototype.enum = function() {
212    if (this.enumValidator) {
213      this.validators = this.validators.filter(function(v) {
214        return v.validator !== this.enumValidator;
215      }, this);
216      this.enumValidator = false;
217    }
218  
219    if (arguments[0] === void 0 || arguments[0] === false) {
220      return this;
221    }
222  
223    let values;
224    let errorMessage;
225  
226    if (utils.isObject(arguments[0])) {
227      if (Array.isArray(arguments[0].values)) {
228        values = arguments[0].values;
229        errorMessage = arguments[0].message;
230      } else {
231        values = utils.object.vals(arguments[0]);
232        errorMessage = MongooseError.messages.String.enum;
233      }
234    } else {
235      values = arguments;
236      errorMessage = MongooseError.messages.String.enum;
237    }
238  
239    for (const value of values) {
240      if (value !== undefined) {
241        this.enumValues.push(this.cast(value));
242      }
243    }
244  
245    const vals = this.enumValues;
246    this.enumValidator = function(v) {
247      return null == v || ~vals.indexOf(v);
248    };
249    this.validators.push({
250      validator: this.enumValidator,
251      message: errorMessage,
252      type: 'enum',
253      enumValues: vals
254    });
255  
256    return this;
257  };

412        return v.validator !== this.minlengthValidator;
413      }, this);
414    }
415  
416    if (value !== null && value !== undefined) {
417      let msg = message || MongooseError.messages.String.minlength;
418      msg = msg.replace(/{MINLENGTH}/, value);
419      this.validators.push({
420        validator: this.minlengthValidator = function(v) {
421          return v === null || v.length >= value;
422        },
423        message: msg,
424        type: 'minlength',

490  /**
491   * Sets a regexp validator.
492   *
493   * Any value that does not pass `regExp`.test(val) will fail validation.
494   *
495   * #### Example:
496   *
497   *     const s = new Schema({ name: { type: String, match: /^a/ }})
498   *     const M = db.model('M', s)
499   *     const m = new M({ name: 'I am invalid' })
500   *     m.validate(function (err) {
501   *       console.error(String(err)) // "ValidationError: Path `name` is invalid (I am invalid)."
502   *       m.name = 'apples'
503   *       m.validate(function (err) {
504   *         assert.ok(err) // success
505   *       })
506   *     })
507   *
508   *     // using a custom error message
509   *     const match = [ /\.html$/, "That file doesn't end in .html ({VALUE})" ];
510   *     const s = new Schema({ file: { type: String, match: match }})
511   *     const M = db.model('M', s);
512   *     const m = new M({ file: 'invalid' });
513   *     m.validate(function (err) {
514   *       console.log(String(err)) // "ValidationError: That file doesn't end in .html (invalid)"
515   *     })
516   *
517   * Empty strings, `undefined`, and `null` values always pass the match validator. If you require these values, enable the `required` validator also.
518   *
519   *     const s = new Schema({ name: { type: String, match: /^a/, required: true }})
520   *
521   * @param {RegExp} regExp regular expression to test against
522   * @param {String} [message] optional custom error message
523   * @return {SchemaType} this
524   * @see Customized Error Messages https://mongoosejs.com/docs/api/error.html#Error.messages
525   * @api public
526   */
527  
528  SchemaString.prototype.match = function match(regExp, message) {
529    // yes, we allow multiple match validators
530  
531    const msg = message || MongooseError.messages.String.match;
532  
533    const matchValidator = function(v) {
534      if (!regExp) {
535        return false;
536      }
537  
538      // In case RegExp happens to have `/g` flag set, we need to reset the
539      // `lastIndex`, otherwise `match` will intermittently fail.
540      regExp.lastIndex = 0;
541  
542      const ret = ((v != null && v !== '')
543        ? regExp.test(v)
544        : true);
545      return ret;
546    };
547  
548    this.validators.push({
549      validator: matchValidator,
550      message: msg,
551      type: 'regexp',
552      regexp: regExp
553    });
554    return this;
555  };

557  /**
558   * Check if the given value satisfies the `required` validator. The value is
559   * considered valid if it is a string (that is, not `null` or `undefined`) and
560   * has positive length. The `required` validator **will** fail for empty
561   * strings.
562   *
563   * @param {Any} value
564   * @param {Document} doc
565   * @return {Boolean}
566   * @api public
567   */
568  
569  SchemaString.prototype.checkRequired = function checkRequired(value, doc) {
570    if (typeof value === 'object' && SchemaType._isRef(this, value, doc, true)) {
571      return value != null;
572    }
573  
574    // `require('util').inherits()` does **not** copy static properties, and
575    // plugins like mongoose-float use `inherits()` for pre-ES6.
576    const _checkRequired = typeof this.constructor.checkRequired === 'function' ?
577      this.constructor.checkRequired() :
578      SchemaString.checkRequired();
579  
580    return _checkRequired(value);
581  };

584   * Casts to String
585   *
586   * @api private
587   */

589  SchemaString.prototype.cast = function(value, doc, init) {
590    if (typeof value !== 'string' && SchemaType._isRef(this, value, doc, init)) {
591      return this._castRef(value, doc, init);
592    }
593  
594    let castString;
595    if (typeof this._castFunction === 'function') {
596      castString = this._castFunction;
597    } else if (typeof this.constructor.cast === 'function') {
598      castString = this.constructor.cast();
599    } else {
600      castString = SchemaString.cast();
601    }
602  
603    try {
604      return castString(value);
605    } catch (error) {
606      throw new CastError('string', value, this.path, null, this);
607    }
608  };

669  /**
670   * Casts contents for queries.
671   *
672   * @param {String} $conditional
673   * @param {any} [val]
674   * @api private
675   */
676  
677  SchemaString.prototype.castForQuery = function($conditional, val, context) {
678    let handler;
679    if ($conditional != null) {
680      handler = this.$conditionalHandlers[$conditional];
681      if (!handler) {
682        throw new Error('Can\'t use ' + $conditional + ' with String.');
683      }
684      return handler.call(this, val, context);
685    }
686  
687    if (Object.prototype.toString.call(val) === '[object RegExp]' || isBsonType(val, 'BSONRegExp')) {
688      return val;
689    }
690  
691    try {
692      return this.applySetters(val, context);
693    } catch (err) {
694      if (err instanceof CastError && err.path === this.path && this.$fullPath != null) {
695        err.path = this.$fullPath;
696      }
697      throw err;
698    }
699  };
