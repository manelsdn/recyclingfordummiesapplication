192    const clone = typeof options.clone === 'boolean' ? options.clone : true;
193  
194    if (schema.instanceOfSchema && clone) {
195      schema = schema.clone();
196    }
197  
198    schema = discriminator(this.casterConstructor, name, schema, tiedValue);
199  
200    const EmbeddedDocument = _createConstructor(schema, null, this.casterConstructor);
201    EmbeddedDocument.baseCasterConstructor = this.casterConstructor;
202  
203    try {
204      Object.defineProperty(EmbeddedDocument, 'name', {
205        value: name

207    } catch (error) {
208      // Ignore error, only happens on old versions of node
209    }
210  
211    this.casterConstructor.discriminators[name] = EmbeddedDocument;

222  SchemaDocumentArray.prototype.doValidate = function(array, fn, scope, options) {
223    // lazy load
224    MongooseDocumentArray || (MongooseDocumentArray = require('../types/documentArray'));
225  
226    const _this = this;
227    try {
228      SchemaType.prototype.doValidate.call(this, array, cb, scope);
229    } catch (err) {
230      return fn(err);
231    }
232  
233    function cb(err) {
234      if (err) {
235        return fn(err);
236      }
237  
238      let count = array && array.length;
239      let error;
240  
241      if (!count) {
242        return fn();
243      }
244      if (options && options.updateValidator) {
245        return fn();
246      }
247      if (!utils.isMongooseDocumentArray(array)) {
248        array = new MongooseDocumentArray(array, _this.path, scope);
249      }
250  
251      // handle sparse arrays, do not use array.forEach which does not
252      // iterate over sparse elements yet reports array.length including
253      // them :(
254  
255      function callback(err) {
256        if (err != null) {
257          error = err;
258        }
259        --count || fn(error);
260      }
261  
262      for (let i = 0, len = count; i < len; ++i) {
263        // sidestep sparse entries
264        let doc = array[i];
265        if (doc == null) {
266          --count || fn(error);
267          continue;
268        }
269  
270        // If you set the array index directly, the doc might not yet be
271        // a full fledged mongoose subdoc, so make it into one.
272        if (!(doc instanceof Subdocument)) {
273          const Constructor = getConstructor(_this.casterConstructor, array[i]);
274          doc = array[i] = new Constructor(doc, array, undefined, undefined, i);
275        }
276  
277        if (options != null && options.validateModifiedOnly && !doc.$isModified()) {
278          --count || fn(error);
279          continue;
280        }
281  
282        doc.$__validate(callback);
283      }
284    }

298  SchemaDocumentArray.prototype.doValidateSync = function(array, scope, options) {
299    const schemaTypeError = SchemaType.prototype.doValidateSync.call(this, array, scope);
300    if (schemaTypeError != null) {
301      return schemaTypeError;
302    }
303  
304    const count = array && array.length;
305    let resultError = null;
306  
307    if (!count) {
308      return;
309    }
310  
311    // handle sparse arrays, do not use array.forEach which does not
312    // iterate over sparse elements yet reports array.length including
313    // them :(
314  
315    for (let i = 0, len = count; i < len; ++i) {
316      // sidestep sparse entries
317      let doc = array[i];
318      if (!doc) {
319        continue;
320      }
321  
322      // If you set the array index directly, the doc might not yet be
323      // a full fledged mongoose subdoc, so make it into one.
324      if (!(doc instanceof Subdocument)) {
325        const Constructor = getConstructor(this.casterConstructor, array[i]);
326        doc = array[i] = new Constructor(doc, array, undefined, undefined, i);
327      }
328  
329      if (options != null && options.validateModifiedOnly && !doc.$isModified()) {
330        continue;
331      }
332  
333      const subdocValidateError = doc.validateSync();
334  
335      if (subdocValidateError && resultError == null) {
336        resultError = subdocValidateError;
337      }
338    }
339  
340    return resultError;
341  };

362

398  SchemaDocumentArray.prototype.cast = function(value, doc, init, prev, options) {
399    // lazy load
400    MongooseDocumentArray || (MongooseDocumentArray = require('../types/documentArray'));
401  
402    // Skip casting if `value` is the same as the previous value, no need to cast. See gh-9266
403    if (value != null && value[arrayPathSymbol] != null && value === prev) {
404      return value;
405    }
406  
407    let selected;
408    let subdoc;
409  
410    options = options || {};
411  
412    const path = options.path || this.path;
413  
414    if (!Array.isArray(value)) {
415      if (!init && !SchemaDocumentArray.options.castNonArrays) {
416        throw new CastError('DocumentArray', value, this.path, null, this);
417      }
418      // gh-2442 mark whole array as modified if we're initializing a doc from
419      // the db and the path isn't an array in the document
420      if (!!doc && init) {
421        doc.markModified(path);
422      }
423      return this.cast([value], doc, init, prev, options);
424    }
425  
426    // We need to create a new array, otherwise change tracking will
427    // update the old doc (gh-4449)
428    if (!options.skipDocumentArrayCast || utils.isMongooseDocumentArray(value)) {
429      value = new MongooseDocumentArray(value, path, doc);
430    }
431  
432    if (prev != null) {
433      value[arrayAtomicsSymbol] = prev[arrayAtomicsSymbol] || {};
434    }
435  
436    if (options.arrayPathIndex != null) {
437      value[arrayPathSymbol] = path + '.' + options.arrayPathIndex;
438    }
439  
440    const rawArray = utils.isMongooseDocumentArray(value) ? value.__array : value;
441    const len = rawArray.length;
442  
443    for (let i = 0; i < len; ++i) {
444      if (!rawArray[i]) {
445        continue;
446      }
447  
448      const Constructor = getConstructor(this.casterConstructor, rawArray[i]);
449  
450      // Check if the document has a different schema (re gh-3701)
451      if (rawArray[i].$__ != null && !(rawArray[i] instanceof Constructor)) {
452        const spreadDoc = handleSpreadDoc(rawArray[i], true);
453        if (rawArray[i] !== spreadDoc) {
454          rawArray[i] = spreadDoc;
455        } else {
456          rawArray[i] = rawArray[i].toObject({
457            transform: false,
458            // Special case: if different model, but same schema, apply virtuals
459            // re: gh-7898
460            virtuals: rawArray[i].schema === Constructor.schema
461          });
462        }
463      }
464  
465      if (rawArray[i] instanceof Subdocument) {
466        if (rawArray[i][documentArrayParent] !== doc) {
467          if (init) {
468            const subdoc = new Constructor(null, value, initDocumentOptions, selected, i);
469            rawArray[i] = subdoc.$init(rawArray[i]);
470          } else {
471            const subdoc = new Constructor(rawArray[i], value, undefined, undefined, i);
472            rawArray[i] = subdoc;
473          }
474        }
475        // Might not have the correct index yet, so ensure it does.
476        if (rawArray[i].__index == null) {
477          rawArray[i].$setIndex(i);
478        }
479      } else if (rawArray[i] != null) {
480        if (init) {
481          if (doc) {
482            selected || (selected = scopePaths(this, doc.$__.selected, init));
483          } else {
484            selected = true;
485          }
486  
487          subdoc = new Constructor(null, value, initDocumentOptions, selected, i);
488          rawArray[i] = subdoc.$init(rawArray[i]);
489        } else {
490          if (prev && typeof prev.id === 'function') {
491            subdoc = prev.id(rawArray[i]._id);
492          }
493  
494          if (prev && subdoc && utils.deepEqual(subdoc.toObject(_toObjectOptions), rawArray[i])) {
495            // handle resetting doc with existing id and same data
496            subdoc.set(rawArray[i]);
497            // if set() is hooked it will have no return value
498            // see gh-746
499            rawArray[i] = subdoc;
500          } else {
501            try {
502              subdoc = new Constructor(rawArray[i], value, undefined,
503                undefined, i);
504              // if set() is hooked it will have no return value
505              // see gh-746
506              rawArray[i] = subdoc;
507            } catch (error) {
508              throw new CastError('embedded', rawArray[i],
509                value[arrayPathSymbol], error, this);
510            }
511          }
512        }
513      }
514    }
515  
516    return value;
517  };

543  /**
544   * Scopes paths selected in a query to this array.
545   * Necessary for proper default application of subdocument values.
546   *
547   * @param {DocumentArrayPath} array the array to scope `fields` paths
548   * @param {Object|undefined} fields the root fields selected in the query
549   * @param {Boolean|undefined} init if we are being created part of a query result
550   * @api private
551   */
552  
553  function scopePaths(array, fields, init) {
554    if (!(init && fields)) {
555      return undefined;
556    }
557  
558    const path = array.path + '.';
559    const keys = Object.keys(fields);
560    let i = keys.length;
561    const selected = {};
562    let hasKeys;
563    let key;
564    let sub;
565  
566    while (i--) {
567      key = keys[i];
568      if (key.startsWith(path)) {
569        sub = key.substring(path.length);
570        if (sub === '$') {
571          continue;
572        }
573        if (sub.startsWith('$.')) {
574          sub = sub.substring(2);
575        }
576        hasKeys || (hasKeys = true);
577        selected[sub] = fields[key];
578      }
579    }
580  
581    return hasKeys && selected || undefined;
582  }
