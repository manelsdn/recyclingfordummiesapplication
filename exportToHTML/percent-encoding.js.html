9    // https://url.spec.whatwg.org/#percent-encode
10   function percentEncode(c) {
11     let hex = c.toString(16).toUpperCase();
12     if (hex.length === 1) {
13       hex = `0${hex}`;
14     }
15   
16     return `%${hex}`;
17   }

19   // https://url.spec.whatwg.org/#percent-decode
20   function percentDecodeBytes(input) {
21     const output = new Uint8Array(input.byteLength);
22     let outputIndex = 0;
23     for (let i = 0; i < input.byteLength; ++i) {
24       const byte = input[i];
25       if (byte !== 0x25) {
26         output[outputIndex++] = byte;
27       } else if (byte === 0x25 && (!isASCIIHex(input[i + 1]) || !isASCIIHex(input[i + 2]))) {
28         output[outputIndex++] = byte;
29       } else {
30         const bytePoint = parseInt(String.fromCodePoint(input[i + 1], input[i + 2]), 16);
31         output[outputIndex++] = bytePoint;
32         i += 2;
33       }
34     }
35   
36     return output.slice(0, outputIndex);
37   }

39   // https://url.spec.whatwg.org/#string-percent-decode
40   function percentDecodeString(input) {
41     const bytes = utf8Encode(input);
42     return percentDecodeBytes(bytes);
43   }

45   // https://url.spec.whatwg.org/#c0-control-percent-encode-set
46   function isC0ControlPercentEncode(c) {
47     return c <= 0x1F || c > 0x7E;
48   }

50   // https://url.spec.whatwg.org/#fragment-percent-encode-set
51   const extraFragmentPercentEncodeSet = new Set([p(" "), p("\""), p("<"), p(">"), p("`")]);
52   function isFragmentPercentEncode(c) {
53     return isC0ControlPercentEncode(c) || extraFragmentPercentEncodeSet.has(c);
54   }

57   const extraQueryPercentEncodeSet = new Set([p(" "), p("\""), p("#"), p("<"), p(">")]);
58   function isQueryPercentEncode(c) {
59     return isC0ControlPercentEncode(c) || extraQueryPercentEncodeSet.has(c);
60   }

63   function isSpecialQueryPercentEncode(c) {
64     return isQueryPercentEncode(c) || c === p("'");
65   }

68   const extraPathPercentEncodeSet = new Set([p("?"), p("`"), p("{"), p("}")]);
69   function isPathPercentEncode(c) {

74   const extraUserinfoPercentEncodeSet =
75     new Set([p("/"), p(":"), p(";"), p("="), p("@"), p("["), p("\\"), p("]"), p("^"), p("|")]);
76   function isUserinfoPercentEncode(c) {
77     return isPathPercentEncode(c) || extraUserinfoPercentEncodeSet.has(c);
78   }

80   // https://url.spec.whatwg.org/#component-percent-encode-set
81   const extraComponentPercentEncodeSet = new Set([p("$"), p("%"), p("&"), p("+"), p(",")]);
82   function isComponentPercentEncode(c) {
83     return isUserinfoPercentEncode(c) || extraComponentPercentEncodeSet.has(c);

87   const extraURLEncodedPercentEncodeSet = new Set([p("!"), p("'"), p("("), p(")"), p("~")]);
88   function isURLEncodedPercentEncode(c) {
89     return isComponentPercentEncode(c) || extraURLEncodedPercentEncodeSet.has(c);

92   // https://url.spec.whatwg.org/#code-point-percent-encode-after-encoding
93   // https://url.spec.whatwg.org/#utf-8-percent-encode
94   // Assuming encoding is always utf-8 allows us to trim one of the logic branches. TODO: support encoding.
95   // The "-Internal" variant here has code points as JS strings. The external version used by other files has code points
96   // as JS numbers, like the rest of the codebase.
97   function utf8PercentEncodeCodePointInternal(codePoint, percentEncodePredicate) {
98     const bytes = utf8Encode(codePoint);
99     let output = "";
100    for (const byte of bytes) {
101      // Our percentEncodePredicate operates on bytes, not code points, so this is slightly different from the spec.
102      if (!percentEncodePredicate(byte)) {
103        output += String.fromCharCode(byte);
104      } else {
105        output += percentEncode(byte);
106      }
107    }
108  
109    return output;
110  }
111  
112  function utf8PercentEncodeCodePoint(codePoint, percentEncodePredicate) {
113    return utf8PercentEncodeCodePointInternal(String.fromCodePoint(codePoint), percentEncodePredicate);
114  }

116  // https://url.spec.whatwg.org/#string-percent-encode-after-encoding
117  // https://url.spec.whatwg.org/#string-utf-8-percent-encode
118  function utf8PercentEncodeString(input, percentEncodePredicate, spaceAsPlus = false) {
119    let output = "";
120    for (const codePoint of input) {
121      if (spaceAsPlus && codePoint === " ") {
122        output += "+";
123      } else {
124        output += utf8PercentEncodeCodePointInternal(codePoint, percentEncodePredicate);
125      }
126    }
127    return output;
128  }
