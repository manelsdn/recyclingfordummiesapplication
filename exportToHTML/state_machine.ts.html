22   let socks: SocksLib | null = null;
23   function loadSocks(): SocksLib {
24     if (socks == null) {
25       const socksImport = getSocks();
26       if ('kModuleError' in socksImport) {
27         throw socksImport.kModuleError;
28       }
29       socks = socksImport;
30     }
31     return socks;

154    /**
155     * Executes the state machine according to the specification
156     */
157    async execute<T extends Document>(
158      executor: StateMachineExecutable,
159      context: MongoCryptContext
160    ): Promise<T> {
161      const keyVaultNamespace = executor._keyVaultNamespace;
162      const keyVaultClient = executor._keyVaultClient;
163      const metaDataClient = executor._metaDataClient;
164      const mongocryptdClient = executor._mongocryptdClient;
165      const mongocryptdManager = executor._mongocryptdManager;
166      let result: T | null = null;
167  
168      while (context.state !== MONGOCRYPT_CTX_DONE && context.state !== MONGOCRYPT_CTX_ERROR) {
169        debug(`[context#${context.id}] ${stateToString.get(context.state) || context.state}`);
170  
171        switch (context.state) {
172          case MONGOCRYPT_CTX_NEED_MONGO_COLLINFO: {
173            const filter = deserialize(context.nextMongoOperation());
174            if (!metaDataClient) {
175              throw new MongoCryptError(
176                'unreachable state machine state: entered MONGOCRYPT_CTX_NEED_MONGO_COLLINFO but metadata client is undefined'
177              );
178            }
179            const collInfo = await this.fetchCollectionInfo(metaDataClient, context.ns, filter);
180  
181            if (collInfo) {
182              context.addMongoOperationResponse(collInfo);
183            }
184  
185            context.finishMongoOperation();
186            break;
187          }
188  
189          case MONGOCRYPT_CTX_NEED_MONGO_MARKINGS: {
190            const command = context.nextMongoOperation();
191            if (!mongocryptdClient) {
192              throw new MongoCryptError(
193                'unreachable state machine state: entered MONGOCRYPT_CTX_NEED_MONGO_MARKINGS but mongocryptdClient is undefined'
194              );
195            }
196  
197            // When we are using the shared library, we don't have a mongocryptd manager.
198            const markedCommand: Uint8Array = mongocryptdManager
199              ? await mongocryptdManager.withRespawn(
200                  this.markCommand.bind(this, mongocryptdClient, context.ns, command)
201                )
202              : await this.markCommand(mongocryptdClient, context.ns, command);
203  
204            context.addMongoOperationResponse(markedCommand);
205            context.finishMongoOperation();
206            break;
207          }
208  
209          case MONGOCRYPT_CTX_NEED_MONGO_KEYS: {
210            const filter = context.nextMongoOperation();
211            const keys = await this.fetchKeys(keyVaultClient, keyVaultNamespace, filter);
212  
213            if (keys.length === 0) {
214              // This is kind of a hack.  For `rewrapManyDataKey`, we have tests that
215              // guarantee that when there are no matching keys, `rewrapManyDataKey` returns
216              // nothing.  We also have tests for auto encryption that guarantee for `encrypt`
217              // we return an error when there are no matching keys.  This error is generated in
218              // subsequent iterations of the state machine.
219              // Some apis (`encrypt`) throw if there are no filter matches and others (`rewrapManyDataKey`)
220              // do not.  We set the result manually here, and let the state machine continue.  `libmongocrypt`
221              // will inform us if we need to error by setting the state to `MONGOCRYPT_CTX_ERROR` but
222              // otherwise we'll return `{ v: [] }`.
223              result = { v: [] } as any as T;
224            }
225            for await (const key of keys) {
226              context.addMongoOperationResponse(serialize(key));
227            }
228  
229            context.finishMongoOperation();
230  
231            break;
232          }
233  
234          case MONGOCRYPT_CTX_NEED_KMS_CREDENTIALS: {
235            const kmsProviders = await executor.askForKMSCredentials();
236            context.provideKMSProviders(serialize(kmsProviders));
237            break;
238          }
239  
240          case MONGOCRYPT_CTX_NEED_KMS: {
241            const requests = Array.from(this.requests(context));
242            await Promise.all(requests);
243  
244            context.finishKMSRequests();
245            break;
246          }
247  
248          case MONGOCRYPT_CTX_READY: {
249            const finalizedContext = context.finalize();
250            // @ts-expect-error finalize can change the state, check for error
251            if (context.state === MONGOCRYPT_CTX_ERROR) {
252              const message = context.status.message || 'Finalization error';
253              throw new MongoCryptError(message);
254            }
255            result = deserialize(finalizedContext, this.options) as T;
256            break;
257          }
258  
259          default:
260            throw new MongoCryptError(`Unknown state: ${context.state}`);
261        }
262      }
263  
264      if (context.state === MONGOCRYPT_CTX_ERROR || result == null) {
265        const message = context.status.message;
266        if (!message) {
267          debug(
268            `unidentifiable error in MongoCrypt - received an error status from \`libmongocrypt\` but received no error message.`
269          );
270        }
271        throw new MongoCryptError(
272          message ??
273            'unidentifiable error in MongoCrypt - received an error status from `libmongocrypt` but received no error message.'
274        );
275      }
276  
277      return result;
278    }

280    /**
281     * Handles the request to the KMS service. Exposed for testing purposes. Do not directly invoke.
282     * @param kmsContext - A C++ KMS context returned from the bindings
283     * @returns A promise that resolves when the KMS reply has be fully parsed
284     */
285    kmsRequest(request: MongoCryptKMSRequest): Promise<void> {
286      const parsedUrl = request.endpoint.split(':');
287      const port = parsedUrl[1] != null ? Number.parseInt(parsedUrl[1], 10) : HTTPS_PORT;
288      const options: tls.ConnectionOptions & { host: string; port: number } = {
289        host: parsedUrl[0],
290        servername: parsedUrl[0],
291        port
292      };
293      const message = request.message;
294  
295      // TODO(NODE-3959): We can adopt `for-await on(socket, 'data')` with logic to control abort
296      // eslint-disable-next-line @typescript-eslint/no-misused-promises, no-async-promise-executor
297      return new Promise(async (resolve, reject) => {
298        const buffer = new BufferPool();
299  
300        // eslint-disable-next-line prefer-const
301        let socket: net.Socket;
302        let rawSocket: net.Socket;
303  
304        function destroySockets() {
305          for (const sock of [socket, rawSocket]) {
306            if (sock) {
307              sock.removeAllListeners();
308              sock.destroy();
309            }
310          }
311        }
312  
313        function ontimeout() {
314          destroySockets();
315          reject(new MongoCryptError('KMS request timed out'));
316        }
317  
318        function onerror(err: Error) {
319          destroySockets();
320          const mcError = new MongoCryptError('KMS request failed', { cause: err });
321          reject(mcError);
322        }
323  
324        if (this.options.proxyOptions && this.options.proxyOptions.proxyHost) {
325          rawSocket = net.connect({
326            host: this.options.proxyOptions.proxyHost,
327            port: this.options.proxyOptions.proxyPort || 1080
328          });
329  
330          rawSocket.on('timeout', ontimeout);
331          rawSocket.on('error', onerror);
332          try {
333            // eslint-disable-next-line @typescript-eslint/no-var-requires
334            const events = require('events') as typeof import('events');
335            await events.once(rawSocket, 'connect');
336            socks ??= loadSocks();
337            options.socket = (
338              await socks.SocksClient.createConnection({
339                existing_socket: rawSocket,
340                command: 'connect',
341                destination: { host: options.host, port: options.port },
342                proxy: {
343                  // host and port are ignored because we pass existing_socket
344                  host: 'iLoveJavaScript',
345                  port: 0,
346                  type: 5,
347                  userId: this.options.proxyOptions.proxyUsername,
348                  password: this.options.proxyOptions.proxyPassword
349                }
350              })
351            ).socket;
352          } catch (err) {
353            return onerror(err);
354          }
355        }
356  
357        const tlsOptions = this.options.tlsOptions;
358        if (tlsOptions) {
359          const kmsProvider = request.kmsProvider as ClientEncryptionDataKeyProvider;
360          const providerTlsOptions = tlsOptions[kmsProvider];
361          if (providerTlsOptions) {
362            const error = this.validateTlsOptions(kmsProvider, providerTlsOptions);
363            if (error) reject(error);
364            try {
365              await this.setTlsOptions(providerTlsOptions, options);
366            } catch (error) {
367              return onerror(error);
368            }
369          }
370        }
371        socket = tls.connect(options, () => {
372          socket.write(message);
373        });
374  
375        socket.once('timeout', ontimeout);
376        socket.once('error', onerror);
377  
378        socket.on('data', data => {
379          buffer.append(data);
380          while (request.bytesNeeded > 0 && buffer.length) {
381            const bytesNeeded = Math.min(request.bytesNeeded, buffer.length);
382            request.addResponse(buffer.read(bytesNeeded));
383          }
384  
385          if (request.bytesNeeded <= 0) {
386            // There's no need for any more activity on this socket at this point.
387            destroySockets();
388            resolve();
389          }
390        });
391      });
392    }

394    *requests(context: MongoCryptContext) {
395      for (
396        let request = context.nextKMSRequest();
397        request != null;
398        request = context.nextKMSRequest()
399      ) {
400        yield this.kmsRequest(request);
401      }
402    }

404    /**
405     * Validates the provided TLS options are secure.
406     *
407     * @param kmsProvider - The KMS provider name.
408     * @param tlsOptions - The client TLS options for the provider.
409     *
410     * @returns An error if any option is invalid.
411     */
412    validateTlsOptions(
413      kmsProvider: string,
414      tlsOptions: ClientEncryptionTlsOptions
415    ): MongoCryptError | void {
416      const tlsOptionNames = Object.keys(tlsOptions);
417      for (const option of INSECURE_TLS_OPTIONS) {
418        if (tlsOptionNames.includes(option)) {
419          return new MongoCryptError(`Insecure TLS options prohibited for ${kmsProvider}: ${option}`);
420        }
421      }
422    }

424    /**
425     * Sets only the valid secure TLS options.
426     *
427     * @param tlsOptions - The client TLS options for the provider.
428     * @param options - The existing connection options.
429     */
430    async setTlsOptions(
431      tlsOptions: ClientEncryptionTlsOptions,
432      options: tls.ConnectionOptions
433    ): Promise<void> {
434      if (tlsOptions.tlsCertificateKeyFile) {
435        const cert = await fs.readFile(tlsOptions.tlsCertificateKeyFile);
436        options.cert = options.key = cert;
437      }
438      if (tlsOptions.tlsCAFile) {
439        options.ca = await fs.readFile(tlsOptions.tlsCAFile);
440      }
441      if (tlsOptions.tlsCertificateKeyFilePassword) {
442        options.passphrase = tlsOptions.tlsCertificateKeyFilePassword;
443      }
444    }

446    /**
447     * Fetches collection info for a provided namespace, when libmongocrypt
448     * enters the `MONGOCRYPT_CTX_NEED_MONGO_COLLINFO` state. The result is
449     * used to inform libmongocrypt of the schema associated with this
450     * namespace. Exposed for testing purposes. Do not directly invoke.
451     *
452     * @param client - A MongoClient connected to the topology
453     * @param ns - The namespace to list collections from
454     * @param filter - A filter for the listCollections command
455     * @param callback - Invoked with the info of the requested collection, or with an error
456     */
457    async fetchCollectionInfo(
458      client: MongoClient,
459      ns: string,
460      filter: Document
461    ): Promise<Uint8Array | null> {
462      const { db } = MongoDBCollectionNamespace.fromString(ns);
463  
464      const collections = await client
465        .db(db)
466        .listCollections(filter, {
467          promoteLongs: false,
468          promoteValues: false
469        })
470        .toArray();
471  
472      const info = collections.length > 0 ? serialize(collections[0]) : null;
473      return info;
474    }

476    /**
477     * Calls to the mongocryptd to provide markings for a command.
478     * Exposed for testing purposes. Do not directly invoke.
479     * @param client - A MongoClient connected to a mongocryptd
480     * @param ns - The namespace (database.collection) the command is being executed on
481     * @param command - The command to execute.
482     * @param callback - Invoked with the serialized and marked bson command, or with an error
483     */
484    async markCommand(client: MongoClient, ns: string, command: Uint8Array): Promise<Uint8Array> {
485      const options = { promoteLongs: false, promoteValues: false };
486      const { db } = MongoDBCollectionNamespace.fromString(ns);
487      const rawCommand = deserialize(command, options);
488  
489      const response = await client.db(db).command(rawCommand, options);
490  
491      return serialize(response, this.bsonOptions);
492    }

494    /**
495     * Requests keys from the keyVault collection on the topology.
496     * Exposed for testing purposes. Do not directly invoke.
497     * @param client - A MongoClient connected to the topology
498     * @param keyVaultNamespace - The namespace (database.collection) of the keyVault Collection
499     * @param filter - The filter for the find query against the keyVault Collection
500     * @param callback - Invoked with the found keys, or with an error
501     */
502    fetchKeys(
503      client: MongoClient,
504      keyVaultNamespace: string,
505      filter: Uint8Array
506    ): Promise<Array<DataKey>> {
507      const { db: dbName, collection: collectionName } =
508        MongoDBCollectionNamespace.fromString(keyVaultNamespace);
509  
510      return client
511        .db(dbName)
512        .collection<DataKey>(collectionName, { readConcern: { level: 'majority' } })
513        .find(deserialize(filter))
514        .toArray();
515    }
