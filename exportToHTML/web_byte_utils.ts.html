39     if (byteLength < 0) {
40       throw new RangeError(`The argument 'byteLength' is invalid. Received ${byteLength}`);
41     }
42     return webByteUtils.fromNumberArray(
43       Array.from({ length: byteLength }, () => Math.floor(Math.random() * 256))

48   const webRandomBytes: (byteLength: number) => Uint8Array = (() => {
49     const { crypto } = globalThis as {
50       crypto?: { getRandomValues?: (space: Uint8Array) => Uint8Array };
51     };
52     if (crypto != null && typeof crypto.getRandomValues === 'function') {
53       return (byteLength: number) => {
54         // @ts-expect-error: crypto.getRandomValues cannot actually be null here
55         // You cannot separate getRandomValues from crypto (need to have this === crypto)
56         return crypto.getRandomValues(webByteUtils.allocate(byteLength));
57       };
58     } else {
59       if (isReactNative()) {
60         const { console } = globalThis as { console?: { warn?: (message: string) => void } };
61         console?.warn?.(
62           'BSON: For React Native please polyfill crypto.getRandomValues, e.g. using: https://www.npmjs.com/package/react-native-get-random-values.'
63         );
64       }
65       return webMathRandomBytes;
66     }

72   export const webByteUtils = {
73     toLocalBufferType(
74       potentialUint8array: Uint8Array | ArrayBufferViewWithTag | ArrayBuffer
75     ): Uint8Array {
76       const stringTag =
77         potentialUint8array?.[Symbol.toStringTag] ??
78         Object.prototype.toString.call(potentialUint8array);
79   
80       if (stringTag === 'Uint8Array') {
81         return potentialUint8array as Uint8Array;
82       }
83   
84       if (ArrayBuffer.isView(potentialUint8array)) {
85         return new Uint8Array(
86           potentialUint8array.buffer.slice(
87             potentialUint8array.byteOffset,
88             potentialUint8array.byteOffset + potentialUint8array.byteLength
89           )
90         );
91       }
92   
93       if (
94         stringTag === 'ArrayBuffer' ||
95         stringTag === 'SharedArrayBuffer' ||
96         stringTag === '[object ArrayBuffer]' ||
97         stringTag === '[object SharedArrayBuffer]'
98       ) {
99         return new Uint8Array(potentialUint8array);
100      }
101  
102      throw new BSONError(`Cannot make a Uint8Array from ${String(potentialUint8array)}`);
103    },

116    equals(a: Uint8Array, b: Uint8Array): boolean {
117      if (a.byteLength !== b.byteLength) {
118        return false;
119      }
120      for (let i = 0; i < a.byteLength; i++) {
121        if (a[i] !== b[i]) {
122          return false;
123        }
124      }
125      return true;

150    fromHex(hex: string): Uint8Array {
151      const evenLengthHex = hex.length % 2 === 0 ? hex : hex.slice(0, hex.length - 1);
152      const buffer = [];
153  
154      for (let i = 0; i < evenLengthHex.length; i += 2) {
155        const firstDigit = evenLengthHex[i];
156        const secondDigit = evenLengthHex[i + 1];
157  
158        if (!HEX_DIGIT.test(firstDigit)) {
159          break;
160        }
161        if (!HEX_DIGIT.test(secondDigit)) {
162          break;
163        }
164  
165        const hexDigit = Number.parseInt(`${firstDigit}${secondDigit}`, 16);
166        buffer.push(hexDigit);
167      }
168  
169      return Uint8Array.from(buffer);
170    },

172    toHex(uint8array: Uint8Array): string {
173      return Array.from(uint8array, byte => byte.toString(16).padStart(2, '0')).join('');
174    },

176    toUTF8(uint8array: Uint8Array, start: number, end: number, fatal: boolean): string {
177      const basicLatin = end - start <= 20 ? tryReadBasicLatin(uint8array, start, end) : null;
178      if (basicLatin != null) {
179        return basicLatin;
180      }
181  
182      if (fatal) {
183        try {
184          return new TextDecoder('utf8', { fatal }).decode(uint8array.slice(start, end));
185        } catch (cause) {
186          throw new BSONError('Invalid UTF-8 string in BSON document', { cause });
187        }
188      }
189      return new TextDecoder('utf8', { fatal }).decode(uint8array.slice(start, end));
190    },
