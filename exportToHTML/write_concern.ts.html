51   /**
52    * A MongoDB WriteConcern, which describes the level of acknowledgement
53    * requested from MongoDB for write operations.
54    * @public
55    *
56    * @see https://www.mongodb.com/docs/manual/reference/write-concern/
57    */
58   export class WriteConcern {
59     /** Request acknowledgment that the write operation has propagated to a specified number of mongod instances or to mongod instances with specified tags. */
60     readonly w?: W;
61     /** Request acknowledgment that the write operation has been written to the on-disk journal */
62     readonly journal?: boolean;
63     /** Specify a time limit to prevent write operations from blocking indefinitely */
64     readonly wtimeoutMS?: number;
65     /**
66      * Specify a time limit to prevent write operations from blocking indefinitely.
67      * @deprecated Will be removed in the next major version. Please use wtimeoutMS.
68      */
69     wtimeout?: number;
70     /**
71      * Request acknowledgment that the write operation has been written to the on-disk journal.
72      * @deprecated Will be removed in the next major version. Please use journal.
73      */
74     j?: boolean;
75     /**
76      * Equivalent to the j option.
77      * @deprecated Will be removed in the next major version. Please use journal.
78      */
79     fsync?: boolean | 1;
80   
81     /**
82      * Constructs a WriteConcern from the write concern properties.
83      * @param w - request acknowledgment that the write operation has propagated to a specified number of mongod instances or to mongod instances with specified tags.
84      * @param wtimeoutMS - specify a time limit to prevent write operations from blocking indefinitely
85      * @param journal - request acknowledgment that the write operation has been written to the on-disk journal
86      * @param fsync - equivalent to the j option. Is deprecated and will be removed in the next major version.
87      */
88     constructor(w?: W, wtimeoutMS?: number, journal?: boolean, fsync?: boolean | 1) {
89       if (w != null) {
90         if (!Number.isNaN(Number(w))) {
91           this.w = Number(w);
92         } else {
93           this.w = w;
94         }
95       }
96       if (wtimeoutMS != null) {
97         this.wtimeoutMS = this.wtimeout = wtimeoutMS;
98       }
99       if (journal != null) {
100        this.journal = this.j = journal;
101      }
102      if (fsync != null) {
103        this.journal = this.j = fsync ? true : false;
104      }
105    }
106  
107    /**
108     * Apply a write concern to a command document. Will modify and return the command.
109     */
110    static apply(command: Document, writeConcern: WriteConcern): Document {
111      const wc: CommandWriteConcernOptions = {};
112      // The write concern document sent to the server has w/wtimeout/j fields.
113      if (writeConcern.w != null) wc.w = writeConcern.w;
114      if (writeConcern.wtimeoutMS != null) wc.wtimeout = writeConcern.wtimeoutMS;
115      if (writeConcern.journal != null) wc.j = writeConcern.j;
116      command.writeConcern = wc;
117      return command;
118    }
119  
120    /** Construct a WriteConcern given an options object. */
121    static fromOptions(
122      options?: WriteConcernOptions | WriteConcern | W,
123      inherit?: WriteConcernOptions | WriteConcern
124    ): WriteConcern | undefined {
125      if (options == null) return undefined;
126      inherit = inherit ?? {};
127      let opts: WriteConcernSettings | WriteConcern | undefined;
128      if (typeof options === 'string' || typeof options === 'number') {
129        opts = { w: options };
130      } else if (options instanceof WriteConcern) {
131        opts = options;
132      } else {
133        opts = options.writeConcern;
134      }
135      const parentOpts: WriteConcern | WriteConcernSettings | undefined =
136        inherit instanceof WriteConcern ? inherit : inherit.writeConcern;
137  
138      const {
139        w = undefined,
140        wtimeout = undefined,
141        j = undefined,
142        fsync = undefined,
143        journal = undefined,
144        wtimeoutMS = undefined
145      } = {
146        ...parentOpts,
147        ...opts
148      };
149      if (
150        w != null ||
151        wtimeout != null ||
152        wtimeoutMS != null ||
153        j != null ||
154        journal != null ||
155        fsync != null
156      ) {
157        return new WriteConcern(w, wtimeout ?? wtimeoutMS, j ?? journal, fsync);
158      }
159      return undefined;
160    }
