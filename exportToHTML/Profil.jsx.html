1    import React, { useState, useEffect } from "react";
2    import { FontAwesomeIcon } from "@fortawesome/react-fontawesome";
3    import { faStar, faEnvelope, faQuoteLeft } from "@fortawesome/free-solid-svg-icons";
4    
5    const ProfilePage = ({ userData }) => {
6      // State pour l'URL de l'image du profil, les points, le score du quiz et l'adresse e-mail de l'utilisateur
7      const [imageURL, setImageURL] = useState("");
8      const [points, setPoints] = useState(0);
9      const [quizScore, setQuizScore] = useState(0);
10     const [userEmail, setUserEmail] = useState("");
11   
12     // Effet de montage pour récupérer les données du profil depuis le stockage local
13     useEffect(() => {
14       const storedImageURL = localStorage.getItem("profileImageURL");
15       if (storedImageURL) {
16         setImageURL(storedImageURL);
17       }
18   
19       const userPoints = localStorage.getItem("userPoints");
20       if (userPoints) {
21         setPoints(parseInt(userPoints));
22       }
23   
24       const storedQuizScore = localStorage.getItem("quizScore");
25       if (storedQuizScore) {
26         setQuizScore(parseInt(storedQuizScore));
27       }
28   
29       const userEmail = localStorage.getItem("userEmail");
30       if (userEmail) {
31         setUserEmail(userEmail);
32       }
33     }, []);
34   
35     // Fonction pour gérer le téléchargement d'une image de profil
36     const handleImageUpload = (event) => {
37       const file = event.target.files[0];
38       const reader = new FileReader();
39   
40       reader.onloadend = () => {
41         if (file) {
42           setImageURL(reader.result);
43           localStorage.setItem("profileImageURL", reader.result);
44         }
45       };
46   
47       if (file) {
48         reader.readAsDataURL(file);
49       }
50     };
51   
52     // Fonction pour afficher le champ de téléchargement de fichier
53     const showFileInput = () => {
54       const input = document.getElementById("fileInput");
55       if (input) {
56         input.click();
57       }
58     };
59   
60     return (
61         <div className="profile-container">
62           {/* Titre de la page */}
63           <h2>Mon profil</h2>
64           {/* En-tête du profil avec avatar et adresse e-mail */}
65           <div className="profile-header">
66             <label htmlFor="fileInput">
67               <div className="profile-avatar" onClick={showFileInput}>
68                 {/* Affichage de l'avatar */}
69                 {!imageURL ? (
70                     <img
71                         src="https://via.placeholder.com/150"
72                         alt="Avatar"
73                         className="avatar-image"
74                     />
75                 ) : (
76                     <img src={imageURL} alt="Avatar" className="avatar-image" />
77                 )}
78               </div>
79             </label>
80             {/* Champ de téléchargement de fichier caché */}
81             <input
82                 id="fileInput"
83                 type="file"
84                 accept="image/*"
85                 onChange={handleImageUpload}
86                 style={{ display: "none" }}
87             />
88             {/* Container pour afficher l'adresse e-mail */}
89             <div className="info-container">
90               <p><FontAwesomeIcon icon={faEnvelope}/> {userEmail}</p>
91             </div>
92           </div>
93           {/* Citation */}
94           <div className="citation-container">
95             <div className="citation-card">
96               <div className="quote-container">
97                 {/* Affichage de la citation */}
98                 <p className="citation">
99                   <FontAwesomeIcon icon={faQuoteLeft}/>
100                  <span className="bold">Le recyclage : un petit geste pour vous, un grand pas pour la planète.</span>{" "}
101                  - <i>Emily Brown </i>  <FontAwesomeIcon icon={faQuoteLeft}/>
102                </p>
103              </div>
104            </div>
105          </div>
106          {/* Statistiques du profil */}
107          <div className="profile-stats">
108            <div className="stat">
109              <p>
110                <FontAwesomeIcon icon={faStar} /> <strong>Points du quiz:</strong>{" "}
111                {quizScore}
112              </p>
113            </div>
114          </div>
115        </div>
116    );
117  };
118  
119  export default ProfilePage;
