66    * The core Mongoose document constructor. You should not call this directly,
67    * the Mongoose [Model constructor](./api/model.html#Model) calls this for you.
68    *
69    * @param {Object} obj the values to set
70    * @param {Object} [fields] optional object containing the fields which were selected in the query returning this document and any populated paths data
71    * @param {Object} [options] various configuration options for the document
72    * @param {Boolean} [options.defaults=true] if `false`, skip applying default values to this document.
73    * @inherits NodeJS EventEmitter https://nodejs.org/api/events.html#class-eventemitter
74    * @event `init`: Emitted on a document after it has been retrieved from the db and fully hydrated by Mongoose.
75    * @event `save`: Emitted when the document is successfully saved
76    * @api private
77    */

79   function Document(obj, fields, skipId, options) {
80     if (typeof skipId === 'object' && skipId != null) {
81       options = skipId;
82       skipId = options.skipId;
83     }
84     options = Object.assign({}, options);
85   
86     // Support `browserDocument.js` syntax
87     if (this.$__schema == null) {
88       const _schema = utils.isObject(fields) && !fields.instanceOfSchema ?
89         new Schema(fields) :
90         fields;
91       this.$__setSchema(_schema);
92       fields = skipId;
93       skipId = options;
94       options = arguments[4] || {};
95     }
96   
97     this.$__ = new InternalCache();
98   
99     // Avoid setting `isNew` to `true`, because it is `true` by default
100    if (options.isNew != null && options.isNew !== true) {
101      this.$isNew = options.isNew;
102    }
103  
104    if (options.priorDoc != null) {
105      this.$__.priorDoc = options.priorDoc;
106    }
107  
108    if (skipId) {
109      this.$__.skipId = skipId;
110    }
111  
112    if (obj != null && typeof obj !== 'object') {
113      throw new ObjectParameterError(obj, 'obj', 'Document');
114    }
115  
116    let defaults = true;
117    if (options.defaults !== undefined) {
118      this.$__.defaults = options.defaults;
119      defaults = options.defaults;
120    }
121  
122    const schema = this.$__schema;
123  
124    if (typeof fields === 'boolean' || fields === 'throw') {
125      if (fields !== true) {
126        this.$__.strictMode = fields;
127      }
128      fields = undefined;
129    } else if (schema.options.strict !== true) {
130      this.$__.strictMode = schema.options.strict;
131    }
132  
133    const requiredPaths = schema.requiredPaths(true);
134    for (const path of requiredPaths) {
135      this.$__.activePaths.require(path);
136    }
137  
138    let exclude = null;
139  
140    // determine if this doc is a result of a query with
141    // excluded fields
142    if (utils.isPOJO(fields) && Object.keys(fields).length > 0) {
143      exclude = isExclusive(fields);
144      this.$__.selected = fields;
145      this.$__.exclude = exclude;
146    }
147  
148    const hasIncludedChildren = exclude === false && fields ?
149      $__hasIncludedChildren(fields) :
150      null;
151  
152    if (this._doc == null) {
153      this.$__buildDoc(obj, fields, skipId, exclude, hasIncludedChildren, false);
154  
155      // By default, defaults get applied **before** setting initial values
156      // Re: gh-6155
157      if (defaults) {
158        applyDefaults(this, fields, exclude, hasIncludedChildren, true, null);
159      }
160    }
161    if (obj) {
162      // Skip set hooks
163      if (this.$__original_set) {
164        this.$__original_set(obj, undefined, true, options);
165      } else {
166        this.$set(obj, undefined, true, options);
167      }
168  
169      if (obj instanceof Document) {
170        this.$isNew = obj.$isNew;
171      }
172    }
173  
174    // Function defaults get applied **after** setting initial values so they
175    // see the full doc rather than an empty one, unless they opt out.
176    // Re: gh-3781, gh-6155
177    if (options.willInit && defaults) {
178      if (options.skipDefaults) {
179        this.$__.skipDefaults = options.skipDefaults;
180      }
181    } else if (defaults) {
182      applyDefaults(this, fields, exclude, hasIncludedChildren, false, options.skipDefaults);
183    }
184  
185    if (!this.$__.strictMode && obj) {
186      const _this = this;
187      const keys = Object.keys(this._doc);
188  
189      keys.forEach(function(key) {
190        // Avoid methods, virtuals, existing fields, and `$` keys. The latter is to avoid overwriting
191        // Mongoose internals.
192        if (!(key in schema.tree) && !(key in schema.methods) && !(key in schema.virtuals) && !key.startsWith('$')) {
193          defineKey({ prop: key, subprops: null, prototype: _this });
194        }
195      });
196    }
197  
198    applyQueue(this);
199  }

203  /**
204   * Boolean flag specifying if the document is new. If you create a document
205   * using `new`, this document will be considered "new". `$isNew` is how
206   * Mongoose determines whether `save()` should use `insertOne()` to create
207   * a new document or `updateOne()` to update an existing document.
208   *
209   * #### Example:
210   *
211   *     const user = new User({ name: 'John Smith' });
212   *     user.$isNew; // true
213   *
214   *     await user.save(); // Sends an `insertOne` to MongoDB
215   *
216   * On the other hand, if you load an existing document from the database
217   * using `findOne()` or another [query operation](https://mongoosejs.com/docs/queries.html),
218   * `$isNew` will be false.
219   *
220   * #### Example:
221   *
222   *     const user = await User.findOne({ name: 'John Smith' });
223   *     user.$isNew; // false
224   *
225   * Mongoose sets `$isNew` to `false` immediately after `save()` succeeds.
226   * That means Mongoose sets `$isNew` to false **before** `post('save')` hooks run.
227   * In `post('save')` hooks, `$isNew` will be `false` if `save()` succeeded.
228   *
229   * #### Example:
230   *
231   *     userSchema.post('save', function() {
232   *       this.$isNew; // false
233   *     });
234   *     await User.create({ name: 'John Smith' });
235   *
236   * For subdocuments, `$isNew` is true if either the parent has `$isNew` set,
237   * or if you create a new subdocument.
238   *
239   * #### Example:
240   *
241   *     // Assume `Group` has a document array `users`
242   *     const group = await Group.findOne();
243   *     group.users[0].$isNew; // false
244   *
245   *     group.users.push({ name: 'John Smith' });
246   *     group.users[1].$isNew; // true
247   *
248   * @api public
249   * @property $isNew
250   * @memberOf Document
251   * @instance
252   */
253  
254  Object.defineProperty(Document.prototype, 'isNew', {
255    get: function() {
256      return this.$isNew;
257    },
258    set: function(value) {
259      this.$isNew = value;
260    }
261  });

472  function $applyDefaultsToNested(val, path, doc) {
473    if (val == null) {
474      return;
475    }
476  
477    const paths = Object.keys(doc.$__schema.paths);
478    const plen = paths.length;
479  
480    const pathPieces = path.indexOf('.') === -1 ? [path] : path.split('.');
481  
482    for (let i = 0; i < plen; ++i) {
483      let curPath = '';
484      const p = paths[i];
485  
486      if (!p.startsWith(path + '.')) {
487        continue;
488      }
489  
490      const type = doc.$__schema.paths[p];
491      const pieces = type.splitPath().slice(pathPieces.length);
492      const len = pieces.length;
493  
494      if (type.defaultValue === void 0) {
495        continue;
496      }
497  
498      let cur = val;
499  
500      for (let j = 0; j < len; ++j) {
501        if (cur == null) {
502          break;
503        }
504  
505        const piece = pieces[j];
506  
507        if (j === len - 1) {
508          if (cur[piece] !== void 0) {
509            break;
510          }
511  
512          try {
513            const def = type.getDefault(doc, false);
514            if (def !== void 0) {
515              cur[piece] = def;
516            }
517          } catch (err) {
518            doc.invalidate(path + '.' + curPath, err);
519            break;
520          }
521  
522          break;
523        }
524  
525        curPath += (!curPath.length ? '' : '.') + piece;
526  
527        cur[piece] = cur[piece] || {};
528        cur = cur[piece];
529      }
530    }
531  }

533  /**
534   * Builds the default doc structure
535   *
536   * @param {Object} obj
537   * @param {Object} [fields]
538   * @param {Boolean} [skipId]
539   * @param {Boolean} [exclude]
540   * @param {Object} [hasIncludedChildren]
541   * @api private
542   * @method $__buildDoc
543   * @memberOf Document
544   * @instance
545   */
546  
547  Document.prototype.$__buildDoc = function(obj, fields, skipId, exclude, hasIncludedChildren) {
548    const doc = {};
549  
550    const paths = Object.keys(this.$__schema.paths).
551      // Don't build up any paths that are underneath a map, we don't know
552      // what the keys will be
553      filter(p => !p.includes('$*'));
554    const plen = paths.length;
555    let ii = 0;
556  
557    for (; ii < plen; ++ii) {
558      const p = paths[ii];
559  
560      if (p === '_id') {
561        if (skipId) {
562          continue;
563        }
564        if (obj && '_id' in obj) {
565          continue;
566        }
567      }
568  
569      const path = this.$__schema.paths[p].splitPath();
570      const len = path.length;
571      const last = len - 1;
572      let curPath = '';
573      let doc_ = doc;
574      let included = false;
575  
576      for (let i = 0; i < len; ++i) {
577        const piece = path[i];
578  
579        if (!curPath.length) {
580          curPath = piece;
581        } else {
582          curPath += '.' + piece;
583        }
584  
585        // support excluding intermediary levels
586        if (exclude === true) {
587          if (curPath in fields) {
588            break;
589          }
590        } else if (exclude === false && fields && !included) {
591          if (curPath in fields) {
592            included = true;
593          } else if (!hasIncludedChildren[curPath]) {
594            break;
595          }
596        }
597  
598        if (i < last) {
599          doc_ = doc_[piece] || (doc_[piece] = {});
600        }
601      }
602    }
603  
604    this._doc = doc;
605  };

666  Document.prototype.$__init = function(doc, opts) {
667    this.$isNew = false;
668    opts = opts || {};
669  
670    // handle docs with populated paths
671    // If doc._id is not null or undefined
672    if (doc._id != null && opts.populated && opts.populated.length) {
673      const id = String(doc._id);
674      for (const item of opts.populated) {
675        if (item.isVirtual) {
676          this.$populated(item.path, utils.getValue(item.path, doc), item);
677        } else {
678          this.$populated(item.path, item._docs[id], item);
679        }
680  
681        if (item._childDocs == null) {
682          continue;
683        }
684        for (const child of item._childDocs) {
685          if (child == null || child.$__ == null) {
686            continue;
687          }
688          child.$__.parent = this;
689        }
690        item._childDocs = [];
691      }
692    }
693  
694    init(this, doc, this._doc, opts);
695  
696    markArraySubdocsPopulated(this, opts.populated);
697    this.$emit('init', this);
698    this.constructor.emit('init', this);
699  
700    const hasIncludedChildren = this.$__.exclude === false && this.$__.selected ?
701      $__hasIncludedChildren(this.$__.selected) :
702      null;
703  
704    applyDefaults(this, this.$__.selected, this.$__.exclude, hasIncludedChildren, false, this.$__.skipDefaults);
705    return this;

708  /**
709   * Init helper.
710   *
711   * @param {Object} self document instance
712   * @param {Object} obj raw mongodb doc
713   * @param {Object} doc object we are initializing
714   * @param {Object} [opts] Optional Options
715   * @param {Boolean} [opts.setters] Call `applySetters` instead of `cast`
716   * @param {String} [prefix] Prefix to add to each path
717   * @api private
718   */
719  
720  function init(self, obj, doc, opts, prefix) {
721    prefix = prefix || '';
722  
723    if (obj.$__ != null) {
724      obj = obj._doc;
725    }
726    const keys = Object.keys(obj);
727    const len = keys.length;
728    let schemaType;
729    let path;
730    let i;
731    let index = 0;
732    const strict = self.$__.strictMode;
733    const docSchema = self.$__schema;
734  
735    while (index < len) {
736      _init(index++);
737    }
738  
739    function _init(index) {
740      i = keys[index];
741      // avoid prototype pollution
742      if (i === '__proto__' || i === 'constructor') {
743        return;
744      }
745      path = prefix ? prefix + i : i;
746      schemaType = docSchema.path(path);
747      // Should still work if not a model-level discriminator, but should not be
748      // necessary. This is *only* to catch the case where we queried using the
749      // base model and the discriminated model has a projection
750      if (docSchema.$isRootDiscriminator && !self.$__isSelected(path)) {
751        return;
752      }
753  
754      const value = obj[i];
755      if (!schemaType && utils.isPOJO(value)) {

3837 /**
3838  * Converts this document into a plain-old JavaScript object ([POJO](https://masteringjs.io/tutorials/fundamentals/pojo)).
3839  *
3840  * Buffers are converted to instances of [mongodb.Binary](https://mongodb.github.io/node-mongodb-native/4.9/classes/Binary.html) for proper storage.
3841  *
3842  * #### Getters/Virtuals
3843  *
3844  * Example of only applying path getters
3845  *
3846  *     doc.toObject({ getters: true, virtuals: false })
3847  *
3848  * Example of only applying virtual getters
3849  *
3850  *     doc.toObject({ virtuals: true })
3851  *
3852  * Example of applying both path and virtual getters
3853  *
3854  *     doc.toObject({ getters: true })
3855  *
3856  * To apply these options to every document of your schema by default, set your [schemas](https://mongoosejs.com/docs/api/schema.html#Schema()) `toObject` option to the same argument.
3857  *
3858  *     schema.set('toObject', { virtuals: true })
3859  *
3860  * #### Transform:
3861  *
3862  * We may need to perform a transformation of the resulting object based on some criteria, say to remove some sensitive information or return a custom object. In this case we set the optional `transform` function.
3863  *
3864  * Transform functions receive three arguments
3865  *
3866  *     function (doc, ret, options) {}
3867  *
3868  * - `doc` The mongoose document which is being converted
3869  * - `ret` The plain object representation which has been converted
3870  * - `options` The options in use (either schema options or the options passed inline)
3871  *
3872  * #### Example:
3873  *
3874  *     // specify the transform schema option
3875  *     if (!schema.options.toObject) schema.options.toObject = {};
3876  *     schema.options.toObject.transform = function (doc, ret, options) {
3877  *       // remove the _id of every document before returning the result
3878  *       delete ret._id;
3879  *       return ret;
3880  *     }
3881  *
3882  *     // without the transformation in the schema
3883  *     doc.toObject(); // { _id: 'anId', name: 'Wreck-it Ralph' }
3884  *
3885  *     // with the transformation
3886  *     doc.toObject(); // { name: 'Wreck-it Ralph' }
3887  *
3888  * With transformations we can do a lot more than remove properties. We can even return completely new customized objects:
3889  *
3890  *     if (!schema.options.toObject) schema.options.toObject = {};
3891  *     schema.options.toObject.transform = function (doc, ret, options) {
3892  *       return { movie: ret.name }
3893  *     }
3894  *
3895  *     // without the transformation in the schema
3896  *     doc.toObject(); // { _id: 'anId', name: 'Wreck-it Ralph' }
3897  *
3898  *     // with the transformation
3899  *     doc.toObject(); // { movie: 'Wreck-it Ralph' }
3900  *
3901  * _Note: if a transform function returns `undefined`, the return value will be ignored._
3902  *
3903  * Transformations may also be applied inline, overridding any transform set in the schema options.
3904  * Any transform function specified in `toObject` options also propagates to any subdocuments.
3905  *
3906  *     function deleteId(doc, ret, options) {
3907  *       delete ret._id;
3908  *       return ret;
3909  *     }
3910  *
3911  *     const schema = mongoose.Schema({ name: String, docArr: [{ name: String }] });
3912  *     const TestModel = mongoose.model('Test', schema);
3913  *
3914  *     const doc = new TestModel({ name: 'test', docArr: [{ name: 'test' }] });
3915  *
3916  *     // pass the transform as an inline option. Deletes `_id` property
3917  *     // from both the top-level document and the subdocument.
3918  *     const obj = doc.toObject({ transform: deleteId });
3919  *     obj._id; // undefined
3920  *     obj.docArr[0]._id; // undefined

3922  * If you want to skip transformations, use `transform: false`:
3923  *
3924  *     schema.options.toObject.hide = '_id';
3925  *     schema.options.toObject.transform = function (doc, ret, options) {
3926  *       if (options.hide) {
3927  *         options.hide.split(' ').forEach(function (prop) {

3940  * to [subdocuments](https://mongoosejs.com/docs/subdocs.html) in addition to the top-level document.
3941  * Similarly, `transform: false` skips transforms for all subdocuments.
3942  * Note that this behavior is different for transforms defined in the schema:
3943  * if you define a transform in `schema.options.toObject.transform`, that transform
3944  * will **not** apply to subdocuments.

3946  *     const memberSchema = new Schema({ name: String, email: String });
