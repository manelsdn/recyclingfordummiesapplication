7    const MongooseError = require('./mongooseError');
8    const getConstructorName = require('../helpers/getConstructorName');
9    const util = require('util');
10   const combinePathErrors = require('../helpers/error/combinePathErrors');
11   
12   class ValidationError extends MongooseError {
13     /**
14      * Document Validation Error
15      *
16      * @api private
17      * @param {Document} [instance]
18      * @inherits MongooseError
19      */
20     constructor(instance) {
21       let _message;
22       if (getConstructorName(instance) === 'model') {
23         _message = instance.constructor.modelName + ' validation failed';
24       } else {
25         _message = 'Validation failed';
26       }
27   
28       super(_message);
29   
30       this.errors = {};
31       this._message = _message;
32   
33       if (instance) {
34         instance.$errors = this.errors;
35       }
36     }
37   
38     /**
39      * Console.log helper
40      */
41     toString() {
42       return this.name + ': ' + combinePathErrors(this);
43     }
44   
45     /**
46      * inspect helper
47      * @api private
48      */
49     inspect() {
50       return Object.assign(new Error(this.message), this);
51     }
52   
53     /**
54     * add message
55     * @param {String} path
56     * @param {String|Error} error
57     * @api private
58     */
59     addError(path, error) {
60       if (error instanceof ValidationError) {
61         const { errors } = error;
62         for (const errorPath of Object.keys(errors)) {
63           this.addError(`${path}.${errorPath}`, errors[errorPath]);
64         }
65   
66         return;
67       }
68   
69       this.errors[path] = error;
70       this.message = this._message + ': ' + combinePathErrors(this);
71     }
72   }
73   
74   
75   if (util.inspect.custom) {
76     // Avoid Node deprecation warning DEP0079
77     ValidationError.prototype[util.inspect.custom] = ValidationError.prototype.inspect;
78   }
79   
80   /**
81    * Helper for JSON.stringify
82    * Ensure `name` and `message` show up in toJSON output re: gh-9847
83    * @api private
84    */
85   Object.defineProperty(ValidationError.prototype, 'toJSON', {
86     enumerable: false,
87     writable: false,
88     configurable: true,
89     value: function() {
90       return Object.assign({}, this, { name: this.name, message: this.message });
91     }
92   });
93   
94   
95   Object.defineProperty(ValidationError.prototype, 'name', {
96     value: 'ValidationError'
