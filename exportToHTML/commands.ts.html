144    // Uses a single allocated buffer for the process, avoiding multiple memory allocations
145    toBin(): Uint8Array[] {
146      const buffers = [];
147      let projection = null;
148  
149      // Set up the flags
150      let flags = 0;
151      if (this.tailable) {
152        flags |= OPTS_TAILABLE_CURSOR;
153      }
154  
155      if (this.secondaryOk) {
156        flags |= OPTS_SECONDARY;
157      }
158  
159      if (this.oplogReplay) {
160        flags |= OPTS_OPLOG_REPLAY;
161      }
162  
163      if (this.noCursorTimeout) {
164        flags |= OPTS_NO_CURSOR_TIMEOUT;
165      }
166  
167      if (this.awaitData) {
168        flags |= OPTS_AWAIT_DATA;
169      }
170  
171      if (this.exhaust) {
172        flags |= OPTS_EXHAUST;
173      }
174  
175      if (this.partial) {
176        flags |= OPTS_PARTIAL;
177      }
178  
179      // If batchSize is different to this.numberToReturn
180      if (this.batchSize !== this.numberToReturn) this.numberToReturn = this.batchSize;
181  
182      // Allocate write protocol header buffer
183      const header = Buffer.alloc(
184        4 * 4 + // Header
185          4 + // Flags
186          Buffer.byteLength(this.ns) +
187          1 + // namespace
188          4 + // numberToSkip
189          4 // numberToReturn
190      );
191  
192      // Add header to buffers
193      buffers.push(header);
194  
195      // Serialize the query
196      const query = BSON.serialize(this.query, {
197        checkKeys: this.checkKeys,
198        serializeFunctions: this.serializeFunctions,
199        ignoreUndefined: this.ignoreUndefined
200      });
201  
202      // Add query document
203      buffers.push(query);
204  
205      if (this.returnFieldSelector && Object.keys(this.returnFieldSelector).length > 0) {
206        // Serialize the projection document
207        projection = BSON.serialize(this.returnFieldSelector, {
208          checkKeys: this.checkKeys,
209          serializeFunctions: this.serializeFunctions,
210          ignoreUndefined: this.ignoreUndefined
211        });
212        // Add projection document
213        buffers.push(projection);
214      }
215  
216      // Total message size
217      const totalLength = header.length + query.length + (projection ? projection.length : 0);
218  
219      // Set up the index
220      let index = 4;
221  
222      // Write total document length
223      header[3] = (totalLength >> 24) & 0xff;
224      header[2] = (totalLength >> 16) & 0xff;
225      header[1] = (totalLength >> 8) & 0xff;
226      header[0] = totalLength & 0xff;
227  
228      // Write header information requestId
229      header[index + 3] = (this.requestId >> 24) & 0xff;
230      header[index + 2] = (this.requestId >> 16) & 0xff;
231      header[index + 1] = (this.requestId >> 8) & 0xff;
232      header[index] = this.requestId & 0xff;
233      index = index + 4;
234  
235      // Write header information responseTo
236      header[index + 3] = (0 >> 24) & 0xff;
237      header[index + 2] = (0 >> 16) & 0xff;
238      header[index + 1] = (0 >> 8) & 0xff;
239      header[index] = 0 & 0xff;
240      index = index + 4;
241  
242      // Write header information OP_QUERY
243      header[index + 3] = (OP_QUERY >> 24) & 0xff;
244      header[index + 2] = (OP_QUERY >> 16) & 0xff;
245      header[index + 1] = (OP_QUERY >> 8) & 0xff;
246      header[index] = OP_QUERY & 0xff;
247      index = index + 4;
248  
249      // Write header information flags
250      header[index + 3] = (flags >> 24) & 0xff;
251      header[index + 2] = (flags >> 16) & 0xff;
252      header[index + 1] = (flags >> 8) & 0xff;
253      header[index] = flags & 0xff;
254      index = index + 4;
255  
256      // Write collection name
257      index = index + header.write(this.ns, index, 'utf8') + 1;
258      header[index - 1] = 0;
259  
260      // Write header information flags numberToSkip
261      header[index + 3] = (this.numberToSkip >> 24) & 0xff;
262      header[index + 2] = (this.numberToSkip >> 16) & 0xff;
263      header[index + 1] = (this.numberToSkip >> 8) & 0xff;
264      header[index] = this.numberToSkip & 0xff;
265      index = index + 4;
266  
267      // Write header information flags numberToReturn
268      header[index + 3] = (this.numberToReturn >> 24) & 0xff;
269      header[index + 2] = (this.numberToReturn >> 16) & 0xff;
270      header[index + 1] = (this.numberToReturn >> 8) & 0xff;
271      header[index] = this.numberToReturn & 0xff;
272      index = index + 4;
273  
274      // Return the buffers
275      return buffers;
276    }

358    parse(options: OpResponseOptions): void {
359      // Don't parse again if not needed
360      if (this.parsed) return;
361      options = options ?? {};
362  
363      // Allow the return of raw documents instead of parsing
364      const raw = options.raw || false;
365      const documentsReturnedIn = options.documentsReturnedIn || null;
366      const useBigInt64 = options.useBigInt64 ?? this.opts.useBigInt64;
367      const promoteLongs = options.promoteLongs ?? this.opts.promoteLongs;
368      const promoteValues = options.promoteValues ?? this.opts.promoteValues;
369      const promoteBuffers = options.promoteBuffers ?? this.opts.promoteBuffers;
370      const bsonRegExp = options.bsonRegExp ?? this.opts.bsonRegExp;
371      let bsonSize;
372  
373      // Set up the options
374      const _options: BSONSerializeOptions = {
375        useBigInt64,
376        promoteLongs,
377        promoteValues,
378        promoteBuffers,
379        bsonRegExp
380      };
381  
382      // Position within OP_REPLY at which documents start
383      // (See https://www.mongodb.com/docs/manual/reference/mongodb-wire-protocol/#wire-op-reply)
384      this.index = 20;
385  
386      // Read the message body
387      this.responseFlags = this.data.readInt32LE(0);
388      this.cursorId = new BSON.Long(this.data.readInt32LE(4), this.data.readInt32LE(8));
389      this.startingFrom = this.data.readInt32LE(12);
390      this.numberReturned = this.data.readInt32LE(16);
391  
392      // Preallocate document array
393      this.documents = new Array(this.numberReturned);
394  
395      this.cursorNotFound = (this.responseFlags & CURSOR_NOT_FOUND) !== 0;
396      this.queryFailure = (this.responseFlags & QUERY_FAILURE) !== 0;
397      this.shardConfigStale = (this.responseFlags & SHARD_CONFIG_STALE) !== 0;
398      this.awaitCapable = (this.responseFlags & AWAIT_CAPABLE) !== 0;
399  
400      // Parse Body
401      for (let i = 0; i < this.numberReturned; i++) {
402        bsonSize =
403          this.data[this.index] |
404          (this.data[this.index + 1] << 8) |
405          (this.data[this.index + 2] << 16) |
406          (this.data[this.index + 3] << 24);
407  
408        // If we have raw results specified slice the return document
409        if (raw) {
410          this.documents[i] = this.data.slice(this.index, this.index + bsonSize);
411        } else {
412          this.documents[i] = BSON.deserialize(
413            this.data.slice(this.index, this.index + bsonSize),
414            _options
415          );
416        }
417  
418        // Adjust the index
419        this.index = this.index + bsonSize;
420      }
421  
422      if (this.documents.length === 1 && documentsReturnedIn != null && raw) {
423        const fieldsAsRaw: Document = {};
424        fieldsAsRaw[documentsReturnedIn] = true;
425        _options.fieldsAsRaw = fieldsAsRaw;
426  
427        const doc = BSON.deserialize(this.documents[0] as Buffer, _options);
428        this.documents = [doc];
429      }
430  
431      // Set parsed
432      this.parsed = true;
433    }

436  // Implementation of OP_MSG spec:
437  // https://github.com/mongodb/specifications/blob/master/source/message/OP_MSG.rst
438  //
439  // struct Section {
440  //   uint8 payloadType;
441  //   union payload {
442  //       document  document; // payloadType == 0
443  //       struct sequence { // payloadType == 1
444  //           int32      size;
445  //           cstring    identifier;
446  //           document*  documents;
447  //       };
448  //   };
449  // };
450  
451  // struct OP_MSG {
452  //   struct MsgHeader {
453  //       int32  messageLength;
454  //       int32  requestID;
455  //       int32  responseTo;
456  //       int32  opCode = 2013;
457  //   };
458  //   uint32      flagBits;
459  //   Section+    sections;
460  //   [uint32     checksum;]
461  // };
462  
463  // Msg Flags
464  const OPTS_CHECKSUM_PRESENT = 1;
465  const OPTS_MORE_TO_COME = 2;
466  const OPTS_EXHAUST_ALLOWED = 1 << 16;

482    requestId: number;
483    serializeFunctions: boolean;
484    ignoreUndefined: boolean;
485    checkKeys: boolean;
486    maxBsonSize: number;
487    checksumPresent: boolean;
488    moreToCome: boolean;

491    constructor(
492      public databaseName: string,
493      public command: Document,
494      public options: OpQueryOptions
495    ) {
496      // Basic options needed to be passed in
497      if (command == null)
498        throw new MongoInvalidArgumentError('Query document must be specified for query');
499  
500      // Basic options
501      this.command.$db = databaseName;
502  
503      if (options.readPreference && options.readPreference.mode !== ReadPreference.PRIMARY) {
504        this.command.$readPreference = options.readPreference.toJSON();
505      }
506  
507      // Ensure empty options
508      this.options = options ?? {};
509  
510      // Additional options
511      this.requestId = options.requestId ? options.requestId : OpMsgRequest.getRequestId();
512  
513      // Serialization option
514      this.serializeFunctions =
515        typeof options.serializeFunctions === 'boolean' ? options.serializeFunctions : false;
516      this.ignoreUndefined =
517        typeof options.ignoreUndefined === 'boolean' ? options.ignoreUndefined : false;
518      this.checkKeys = typeof options.checkKeys === 'boolean' ? options.checkKeys : false;
519      this.maxBsonSize = options.maxBsonSize || 1024 * 1024 * 16;
520  
521      // flags
522      this.checksumPresent = false;
523      this.moreToCome = options.moreToCome || false;
524      this.exhaustAllowed =
525        typeof options.exhaustAllowed === 'boolean' ? options.exhaustAllowed : false;
526    }
527  
528    toBin(): Buffer[] {
529      const buffers: Buffer[] = [];
530      let flags = 0;
531  
532      if (this.checksumPresent) {
533        flags |= OPTS_CHECKSUM_PRESENT;
534      }
535  
536      if (this.moreToCome) {
537        flags |= OPTS_MORE_TO_COME;
538      }
539  
540      if (this.exhaustAllowed) {
541        flags |= OPTS_EXHAUST_ALLOWED;
542      }
543  
544      const header = Buffer.alloc(
545        4 * 4 + // Header
546          4 // Flags
547      );
548  
549      buffers.push(header);
550  
551      let totalLength = header.length;
552      const command = this.command;
553      totalLength += this.makeDocumentSegment(buffers, command);
554  
555      header.writeInt32LE(totalLength, 0); // messageLength
556      header.writeInt32LE(this.requestId, 4); // requestID
557      header.writeInt32LE(0, 8); // responseTo
558      header.writeInt32LE(OP_MSG, 12); // opCode
559      header.writeUInt32LE(flags, 16); // flags
560      return buffers;
561    }
562  
563    makeDocumentSegment(buffers: Uint8Array[], document: Document): number {
564      const payloadTypeBuffer = Buffer.alloc(1);
565      payloadTypeBuffer[0] = 0;
566  
567      const documentBuffer = this.serializeBson(document);
568      buffers.push(payloadTypeBuffer);
569      buffers.push(documentBuffer);
570  
571      return payloadTypeBuffer.length + documentBuffer.length;
572    }
573  
574    serializeBson(document: Document): Uint8Array {
575      return BSON.serialize(document, {
576        checkKeys: this.checkKeys,
577        serializeFunctions: this.serializeFunctions,
578        ignoreUndefined: this.ignoreUndefined
579      });
580    }
581  
582    static getRequestId(): number {
583      _requestId = (_requestId + 1) & 0x7fffffff;
584      return _requestId;
585    }

611    constructor(
612      message: Buffer,
613      msgHeader: MessageHeader,
614      msgBody: Buffer,
615      opts?: OpResponseOptions
616    ) {
617      this.parsed = false;
618      this.raw = message;
619      this.data = msgBody;
620      this.opts = opts ?? {
621        useBigInt64: false,
622        promoteLongs: true,
623        promoteValues: true,
624        promoteBuffers: false,
625        bsonRegExp: false
626      };
627  
628      // Read the message header
629      this.length = msgHeader.length;
630      this.requestId = msgHeader.requestId;
631      this.responseTo = msgHeader.responseTo;
632      this.opCode = msgHeader.opCode;
633      this.fromCompressed = msgHeader.fromCompressed;
634  
635      // Read response flags
636      this.responseFlags = msgBody.readInt32LE(0);
637      this.checksumPresent = (this.responseFlags & OPTS_CHECKSUM_PRESENT) !== 0;
638      this.moreToCome = (this.responseFlags & OPTS_MORE_TO_COME) !== 0;
639      this.exhaustAllowed = (this.responseFlags & OPTS_EXHAUST_ALLOWED) !== 0;
640      this.useBigInt64 = typeof this.opts.useBigInt64 === 'boolean' ? this.opts.useBigInt64 : false;
641      this.promoteLongs = typeof this.opts.promoteLongs === 'boolean' ? this.opts.promoteLongs : true;
642      this.promoteValues =
643        typeof this.opts.promoteValues === 'boolean' ? this.opts.promoteValues : true;
644      this.promoteBuffers =
645        typeof this.opts.promoteBuffers === 'boolean' ? this.opts.promoteBuffers : false;
646      this.bsonRegExp = typeof this.opts.bsonRegExp === 'boolean' ? this.opts.bsonRegExp : false;
647  
648      this.documents = [];
649    }
650  
651    isParsed(): boolean {
652      return this.parsed;
653    }
654  
655    parse(options: OpResponseOptions): void {
656      // Don't parse again if not needed
657      if (this.parsed) return;
658      options = options ?? {};
659  
660      this.index = 4;
661      // Allow the return of raw documents instead of parsing
662      const raw = options.raw || false;
663      const documentsReturnedIn = options.documentsReturnedIn || null;
664      const useBigInt64 = options.useBigInt64 ?? this.opts.useBigInt64;
665      const promoteLongs = options.promoteLongs ?? this.opts.promoteLongs;
666      const promoteValues = options.promoteValues ?? this.opts.promoteValues;
667      const promoteBuffers = options.promoteBuffers ?? this.opts.promoteBuffers;
668      const bsonRegExp = options.bsonRegExp ?? this.opts.bsonRegExp;
669      const validation = this.parseBsonSerializationOptions(options);
670  
671      // Set up the options
672      const bsonOptions: BSONSerializeOptions = {
673        useBigInt64,
674        promoteLongs,
675        promoteValues,
676        promoteBuffers,
677        bsonRegExp,
678        validation
679        // Due to the strictness of the BSON libraries validation option we need this cast
680      } as BSONSerializeOptions & { validation: { utf8: { writeErrors: boolean } } };
681  
682      while (this.index < this.data.length) {
683        const payloadType = this.data.readUInt8(this.index++);
684        if (payloadType === 0) {
685          const bsonSize = this.data.readUInt32LE(this.index);
686          const bin = this.data.slice(this.index, this.index + bsonSize);
687          this.documents.push(raw ? bin : BSON.deserialize(bin, bsonOptions));
688          this.index += bsonSize;
689        } else if (payloadType === 1) {
690          // It was decided that no driver makes use of payload type 1
691  
692          // TODO(NODE-3483): Replace with MongoDeprecationError
693          throw new MongoRuntimeError('OP_MSG Payload Type 1 detected unsupported protocol');
694        }
695      }
696  
697      if (this.documents.length === 1 && documentsReturnedIn != null && raw) {
698        const fieldsAsRaw: Document = {};
699        fieldsAsRaw[documentsReturnedIn] = true;
700        bsonOptions.fieldsAsRaw = fieldsAsRaw;
701        const doc = BSON.deserialize(this.documents[0] as Buffer, bsonOptions);
702        this.documents = [doc];
703      }
704  
705      this.parsed = true;
706    }
707  
708    parseBsonSerializationOptions({ enableUtf8Validation }: BSONSerializeOptions): {
709      utf8: { writeErrors: false } | false;
710    } {
711      if (enableUtf8Validation === false) {
712        return { utf8: false };
713      }
714  
715      return { utf8: { writeErrors: false } };
716    }
717  }

719  const MESSAGE_HEADER_SIZE = 16;
720  const COMPRESSION_DETAILS_SIZE = 9; // originalOpcode + uncompressedSize, compressorID
721  
722  /**
723   * @internal
724   *
725   * An OP_COMPRESSED request wraps either an OP_QUERY or OP_MSG message.
726   */
727  export class OpCompressedRequest {
728    constructor(
729      private command: WriteProtocolMessageType,
730      private options: { zlibCompressionLevel: number; agreedCompressor: CompressorName }
731    ) {}
732  
733    // Return whether a command contains an uncompressible command term
734    // Will return true if command contains no uncompressible command terms
735    static canCompress(command: WriteProtocolMessageType) {
736      const commandDoc = command instanceof OpMsgRequest ? command.command : command.query;
737      const commandName = Object.keys(commandDoc)[0];
738      return !uncompressibleCommands.has(commandName);
739    }
740  
741    async toBin(): Promise<Buffer[]> {
742      const concatenatedOriginalCommandBuffer = Buffer.concat(this.command.toBin());
743      // otherwise, compress the message
744      const messageToBeCompressed = concatenatedOriginalCommandBuffer.slice(MESSAGE_HEADER_SIZE);
745  
746      // Extract information needed for OP_COMPRESSED from the uncompressed message
747      const originalCommandOpCode = concatenatedOriginalCommandBuffer.readInt32LE(12);
748  
749      // Compress the message body
750      const compressedMessage = await compress(this.options, messageToBeCompressed);
751      // Create the msgHeader of OP_COMPRESSED
752      const msgHeader = Buffer.alloc(MESSAGE_HEADER_SIZE);
753      msgHeader.writeInt32LE(
754        MESSAGE_HEADER_SIZE + COMPRESSION_DETAILS_SIZE + compressedMessage.length,
755        0
756      ); // messageLength
757      msgHeader.writeInt32LE(this.command.requestId, 4); // requestID
758      msgHeader.writeInt32LE(0, 8); // responseTo (zero)
759      msgHeader.writeInt32LE(OP_COMPRESSED, 12); // opCode
760  
761      // Create the compression details of OP_COMPRESSED
762      const compressionDetails = Buffer.alloc(COMPRESSION_DETAILS_SIZE);
763      compressionDetails.writeInt32LE(originalCommandOpCode, 0); // originalOpcode
764      compressionDetails.writeInt32LE(messageToBeCompressed.length, 4); // Size of the uncompressed compressedMessage, excluding the MsgHeader
765      compressionDetails.writeUInt8(Compressor[this.options.agreedCompressor], 8); // compressorID
766      return [msgHeader, compressionDetails, compressedMessage];
767    }
768  }
