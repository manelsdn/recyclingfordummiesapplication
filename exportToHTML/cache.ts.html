2    const EXPIRATION_BUFFER_MS = 300000;

4    /**
5     * An entry in a cache that can expire in a certain amount of time.
6     */
7    export abstract class ExpiringCacheEntry {
8      expiration: number;
9    
10     /**
11      * Create a new expiring token entry.
12      */
13     constructor(expiration: number) {
14       this.expiration = this.expirationTime(expiration);
15     }
16     /**
17      * The entry is still valid if the expiration is more than
18      * 5 minutes from the expiration time.
19      */
20     isValid() {
21       return this.expiration - Date.now() > EXPIRATION_BUFFER_MS;
22     }
23   
24     /**
25      * Get an expiration time in milliseconds past epoch.
26      */
27     private expirationTime(expiresInSeconds: number): number {
28       return Date.now() + expiresInSeconds * 1000;
29     }
30   }
