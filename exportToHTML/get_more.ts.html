1    import type { Document, Long } from '../bson';
2    import { MongoRuntimeError } from '../error';
3    import type { Server } from '../sdam/server';
4    import type { ClientSession } from '../sessions';
5    import { maxWireVersion, type MongoDBNamespace } from '../utils';
6    import { AbstractOperation, Aspect, defineAspects, type OperationOptions } from './operation';
7    
8    /** @internal */
9    export interface GetMoreOptions extends OperationOptions {
10     /** Set the batchSize for the getMoreCommand when iterating over the query results. */
11     batchSize?: number;
12     /**
13      * Comment to apply to the operation.
14      *
15      * getMore only supports 'comment' in server versions 4.4 and above.
16      */
17     comment?: unknown;
18     /** Number of milliseconds to wait before aborting the query. */
19     maxTimeMS?: number;
20     /** TODO(NODE-4413): Address bug with maxAwaitTimeMS not being passed in from the cursor correctly */
21     maxAwaitTimeMS?: number;
22   }
23   
24   /**
25    * GetMore command: https://www.mongodb.com/docs/manual/reference/command/getMore/
26    * @internal
27    */
28   export interface GetMoreCommand {
29     getMore: Long;
30     collection: string;
31     batchSize?: number;
32     maxTimeMS?: number;
33     /** Only supported on wire versions 10 or greater */
34     comment?: unknown;
35   }
36   
37   /** @internal */
38   export class GetMoreOperation extends AbstractOperation {
39     cursorId: Long;
40     override options: GetMoreOptions;
41   
42     constructor(ns: MongoDBNamespace, cursorId: Long, server: Server, options: GetMoreOptions) {
43       super(options);
44   
45       this.options = options;
46       this.ns = ns;
47       this.cursorId = cursorId;
48       this.server = server;
49     }
50   
51     /**
52      * Although there is a server already associated with the get more operation, the signature
53      * for execute passes a server so we will just use that one.
54      */
55     override async execute(server: Server, _session: ClientSession | undefined): Promise<Document> {
56       if (server !== this.server) {
57         throw new MongoRuntimeError('Getmore must run on the same server operation began on');
58       }
59   
60       if (this.cursorId == null || this.cursorId.isZero()) {
61         throw new MongoRuntimeError('Unable to iterate cursor with no id');
62       }
63   
64       const collection = this.ns.collection;
65       if (collection == null) {
66         // Cursors should have adopted the namespace returned by MongoDB
67         // which should always defined a collection name (even a pseudo one, ex. db.aggregate())
68         throw new MongoRuntimeError('A collection name must be determined before getMore');
69       }
70   
71       const getMoreCmd: GetMoreCommand = {
72         getMore: this.cursorId,
73         collection
74       };
75   
76       if (typeof this.options.batchSize === 'number') {
77         getMoreCmd.batchSize = Math.abs(this.options.batchSize);
78       }
79   
80       if (typeof this.options.maxAwaitTimeMS === 'number') {
81         getMoreCmd.maxTimeMS = this.options.maxAwaitTimeMS;
82       }
83   
84       // we check for undefined specifically here to allow falsy values
85       // eslint-disable-next-line no-restricted-syntax
86       if (this.options.comment !== undefined && maxWireVersion(server) >= 9) {
87         getMoreCmd.comment = this.options.comment;
88       }
89   
90       const commandOptions = {
91         returnFieldSelector: null,
92         documentsReturnedIn: 'nextBatch',
93         ...this.options
94       };
95   
96       return server.commandAsync(this.ns, getMoreCmd, commandOptions);
97     }
98   }
99   
100  defineAspects(GetMoreOperation, [Aspect.READ_OPERATION, Aspect.MUST_SELECT_SAME_SERVER]);
