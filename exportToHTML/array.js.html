30   /**
31    * Array SchemaType constructor
32    *
33    * @param {String} key
34    * @param {SchemaType} cast
35    * @param {Object} options
36    * @param {Object} schemaOptions
37    * @inherits SchemaType
38    * @api public
39    */
40   
41   function SchemaArray(key, cast, options, schemaOptions) {
42     // lazy load
43     EmbeddedDoc || (EmbeddedDoc = require('../types').Embedded);
44   
45     let typeKey = 'type';
46     if (schemaOptions && schemaOptions.typeKey) {
47       typeKey = schemaOptions.typeKey;
48     }
49     this.schemaOptions = schemaOptions;
50   
51     if (cast) {
52       let castOptions = {};
53   
54       if (utils.isPOJO(cast)) {
55         if (cast[typeKey]) {
56           // support { type: Woot }
57           castOptions = clone(cast); // do not alter user arguments
58           delete castOptions[typeKey];
59           cast = cast[typeKey];
60         } else {
61           cast = Mixed;
62         }
63       }
64   
65       if (options != null && options.ref != null && castOptions.ref == null) {
66         castOptions.ref = options.ref;
67       }
68   
69       if (cast === Object) {
70         cast = Mixed;
71       }
72   
73       // support { type: 'String' }
74       const name = typeof cast === 'string'
75         ? cast
76         : utils.getFunctionName(cast);
77   
78       const Types = require('./index.js');
79       const caster = Types.hasOwnProperty(name) ? Types[name] : cast;
80   
81       this.casterConstructor = caster;
82   
83       if (this.casterConstructor instanceof SchemaArray) {
84         this.casterConstructor[isNestedArraySymbol] = true;
85       }
86   
87       if (typeof caster === 'function' &&
88           !caster.$isArraySubdocument &&
89           !caster.$isSchemaMap) {
90         const path = this.caster instanceof EmbeddedDoc ? null : key;
91         this.caster = new caster(path, castOptions);
92       } else {
93         this.caster = caster;
94         if (!(this.caster instanceof EmbeddedDoc)) {
95           this.caster.path = key;
96         }
97       }
98   
99       this.$embeddedSchemaType = this.caster;
100    }
101  
102    this.$isMongooseArray = true;
103  
104    SchemaType.call(this, key, options, 'Array');
105  
106    let defaultArr;
107    let fn;
108  
109    if (this.defaultValue != null) {
110      defaultArr = this.defaultValue;
111      fn = typeof defaultArr === 'function';
112    }
113  
114    if (!('defaultValue' in this) || this.defaultValue !== void 0) {
115      const defaultFn = function() {
116        // Leave it up to `cast()` to convert the array
117        return fn
118          ? defaultArr.call(this)
119          : defaultArr != null
120            ? [].concat(defaultArr)
121            : [];
122      };
123      defaultFn.$runBeforeSetters = !fn;
124      this.default(defaultFn);
125    }
126  }

243  /**
244   * Adds an enum validator if this is an array of strings or numbers. Equivalent to
245   * `SchemaString.prototype.enum()` or `SchemaNumber.prototype.enum()`
246   *
247   * @param {...String|Object} [args] enumeration values
248   * @return {SchemaArray} this
249   */
250  
251  SchemaArray.prototype.enum = function() {
252    let arr = this;
253    while (true) {
254      const instance = arr &&
255      arr.caster &&
256      arr.caster.instance;
257      if (instance === 'Array') {
258        arr = arr.caster;
259        continue;
260      }
261      if (instance !== 'String' && instance !== 'Number') {
262        throw new Error('`enum` can only be set on an array of strings or numbers ' +
263          ', not ' + instance);
264      }
265      break;
266    }
267  
268    let enumArray = arguments;
269    if (!Array.isArray(arguments) && utils.isObject(arguments)) {
270      enumArray = utils.object.vals(enumArray);
271    }
272  
273    arr.caster.enum.apply(arr.caster, enumArray);
274    return this;
275  };

295  SchemaArray.prototype._applySetters = function(value, scope, init, priorVal) {
296    if (this.casterConstructor.$isMongooseArray &&
297        SchemaArray.options.castNonArrays &&
298        !this[isNestedArraySymbol]) {
299      // Check nesting levels and wrap in array if necessary
300      let depth = 0;
301      let arr = this;
302      while (arr != null &&
303        arr.$isMongooseArray &&
304        !arr.$isMongooseDocumentArray) {
305        ++depth;
306        arr = arr.casterConstructor;
307      }
308  
309      // No need to wrap empty arrays
310      if (value != null && value.length !== 0) {
311        const valueDepth = arrayDepth(value);
312        if (valueDepth.min === valueDepth.max && valueDepth.max < depth && valueDepth.containsNonArrayItem) {
313          for (let i = valueDepth.max; i < depth; ++i) {
314            value = [value];
315          }
316        }
317      }
318    }
319  
320    return SchemaType.prototype._applySetters.call(this, value, scope, init, priorVal);
321  };

323  /**
324   * Casts values for set().
325   *
326   * @param {Object} value
327   * @param {Document} doc document that triggers the casting
328   * @param {Boolean} init whether this is an initialization cast
329   * @api private
330   */
331  
332  SchemaArray.prototype.cast = function(value, doc, init, prev, options) {
333    // lazy load
334    MongooseArray || (MongooseArray = require('../types').Array);
335  
336    let i;
337    let l;
338  
339    if (Array.isArray(value)) {
340      const len = value.length;
341      if (!len && doc) {
342        const indexes = doc.schema.indexedPaths();
343  
344        const arrayPath = this.path;
345        for (i = 0, l = indexes.length; i < l; ++i) {
346          const pathIndex = indexes[i][0][arrayPath];
347          if (pathIndex === '2dsphere' || pathIndex === '2d') {
348            return;
349          }
350        }
351  
352        // Special case: if this index is on the parent of what looks like
353        // GeoJSON, skip setting the default to empty array re: #1668, #3233
354        const arrayGeojsonPath = this.path.endsWith('.coordinates') ?
355          this.path.substring(0, this.path.lastIndexOf('.')) : null;
356        if (arrayGeojsonPath != null) {
357          for (i = 0, l = indexes.length; i < l; ++i) {
358            const pathIndex = indexes[i][0][arrayGeojsonPath];
359            if (pathIndex === '2dsphere') {
360              return;
361            }
362          }
363        }
364      }
365  
366      options = options || emptyOpts;
367  
368      let rawValue = utils.isMongooseArray(value) ? value.__array : value;
369      let path = options.path || this.path;
370      if (options.arrayPathIndex != null) {
371        path += '.' + options.arrayPathIndex;
372      }
373      value = MongooseArray(rawValue, path, doc, this);
374      rawValue = value.__array;
375  
376      if (init && doc != null && doc.$__ != null && doc.$populated(this.path)) {
377        return value;
378      }
379  
380      const caster = this.caster;
381      const isMongooseArray = caster.$isMongooseArray;
382      if (caster && this.casterConstructor !== Mixed) {
383        try {
384          const len = rawValue.length;
385          for (i = 0; i < len; i++) {
386            const opts = {};
387            // Perf: creating `arrayPath` is expensive for large arrays.
388            // We only need `arrayPath` if this is a nested array, so
389            // skip if possible.
390            if (isMongooseArray) {
391              if (options.arrayPath != null) {
392                opts.arrayPathIndex = i;
393              } else if (caster._arrayParentPath != null) {
394                opts.arrayPathIndex = i;
395              }
396            }
397            rawValue[i] = caster.applySetters(rawValue[i], doc, init, void 0, opts);
398          }
399        } catch (e) {
400          // rethrow
401          throw new CastError('[' + e.kind + ']', util.inspect(value), this.path + '.' + i, e, this);
402        }
403      }
404  
405      return value;
406    }
407  
408    const castNonArraysOption = this.options.castNonArrays != null ? this.options.castNonArrays : SchemaArray.options.castNonArrays;
409    if (init || castNonArraysOption) {
410      // gh-2442: if we're loading this from the db and its not an array, mark
411      // the whole array as modified.
412      if (!!doc && !!init) {
413        doc.markModified(this.path);
414      }
415      return this.cast([value], doc, init);
416    }
417  
418    throw new CastError('Array', util.inspect(value), this.path, null, this);
419  };

492  SchemaArray.prototype._castForQuery = function(val, context) {
493    let Constructor = this.casterConstructor;
494  
495    if (val &&
496        Constructor.discriminators &&
497        Constructor.schema &&
498        Constructor.schema.options &&
499        Constructor.schema.options.discriminatorKey) {
500      if (typeof val[Constructor.schema.options.discriminatorKey] === 'string' &&
501          Constructor.discriminators[val[Constructor.schema.options.discriminatorKey]]) {
502        Constructor = Constructor.discriminators[val[Constructor.schema.options.discriminatorKey]];
503      } else {
504        const constructorByValue = getDiscriminatorByValue(Constructor.discriminators, val[Constructor.schema.options.discriminatorKey]);
505        if (constructorByValue) {
506          Constructor = constructorByValue;
507        }
508      }
509    }
510  
511    const proto = this.casterConstructor.prototype;
512    const protoCastForQuery = proto && proto.castForQuery;
513    const protoCast = proto && proto.cast;
514    const constructorCastForQuery = Constructor.castForQuery;
515    const caster = this.caster;
516  
517    if (Array.isArray(val)) {
518      this.setters.reverse().forEach(setter => {
519        val = setter.call(this, val, this);
520      });
521      val = val.map(function(v) {
522        if (utils.isObject(v) && v.$elemMatch) {
523          return v;
524        }
525        if (protoCastForQuery) {
526          v = protoCastForQuery.call(caster, null, v, context);
527          return v;
528        } else if (protoCast) {
529          v = protoCast.call(caster, v);
530          return v;
531        } else if (constructorCastForQuery) {
532          v = constructorCastForQuery.call(caster, null, v, context);
533          return v;
534        }
535        if (v != null) {
536          v = new Constructor(v);
537          return v;
538        }
539        return v;
540      });
541    } else if (protoCastForQuery) {
542      val = protoCastForQuery.call(caster, null, val, context);
543    } else if (protoCast) {
544      val = protoCast.call(caster, val);
545    } else if (constructorCastForQuery) {
546      val = constructorCastForQuery.call(caster, null, val, context);
547    } else if (val != null) {
548      val = new Constructor(val);
549    }
550  
551    return val;
552  };

562  SchemaArray.prototype.castForQuery = function($conditional, val, context) {
563    let handler;
564  
565    if ($conditional != null) {
566      handler = this.$conditionalHandlers[$conditional];
567  
568      if (!handler) {
569        throw new Error('Can\'t use ' + $conditional + ' with Array.');

571  
572      return handler.call(this, val, context);
573    } else {
574      return this._castForQuery(val, context);
575    }

578  function cast$all(val, context) {
579    if (!Array.isArray(val)) {
580      val = [val];
581    }
582  
583    val = val.map((v) => {
584      if (!utils.isObject(v)) {
585        return v;
586      }
587      if (v.$elemMatch != null) {
588        return { $elemMatch: cast(this.casterConstructor.schema, v.$elemMatch, null, this && this.$$context) };
589      }
590  
591      const o = {};
592      o[this.path] = v;
593      return cast(this.casterConstructor.schema, o, null, this && this.$$context)[this.path];
594    }, this);
595  
596    return this.castForQuery(null, val, context);
597  }

599  function cast$elemMatch(val, context) {
600    const keys = Object.keys(val);
601    const numKeys = keys.length;
602    for (let i = 0; i < numKeys; ++i) {
603      const key = keys[i];
604      const value = val[key];
605      if (isOperator(key) && value != null) {
606        val[key] = this.castForQuery(key, value, context);
607      }
608    }
609  
610    // Is this an embedded discriminator and is the discriminator key set?
611    // If so, use the discriminator schema. See gh-7449
612    const discriminatorKey = this &&
613      this.casterConstructor &&
614      this.casterConstructor.schema &&
615      this.casterConstructor.schema.options &&
616      this.casterConstructor.schema.options.discriminatorKey;
617    const discriminators = this &&
618    this.casterConstructor &&
619    this.casterConstructor.schema &&
620    this.casterConstructor.schema.discriminators || {};
621    if (discriminatorKey != null &&
622        val[discriminatorKey] != null &&
623        discriminators[val[discriminatorKey]] != null) {
624      return cast(discriminators[val[discriminatorKey]], val, null, this && this.$$context);
625    }
626    const schema = this.casterConstructor.schema ?? context.schema;
627    return cast(schema, val, null, this && this.$$context);
628  }

640  function createLogicalQueryOperatorHandler(op) {
641    return function logicalQueryOperatorHandler(val, context) {
642      if (!Array.isArray(val)) {
643        throw new TypeError('conditional ' + op + ' requires an array');
644      }
645  
646      const ret = [];
647      for (const obj of val) {
648        ret.push(cast(this.casterConstructor.schema ?? context.schema, obj, null, this && this.$$context));
649      }
650  
651      return ret;
