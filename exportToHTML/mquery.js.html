12   /**
13    * Query constructor used for building queries.
14    *
15    * #### Example:
16    *
17    *     var query = new Query({ name: 'mquery' });
18    *     query.setOptions({ collection: moduleCollection })
19    *     await query.where('age').gte(21).exec();
20    *
21    * @param {Object} [criteria] criteria for the query OR the collection instance to use
22    * @param {Object} [options]
23    * @api public
24    */
25   
26   function Query(criteria, options) {
27     if (!(this instanceof Query))
28       return new Query(criteria, options);
29   
30     const proto = this.constructor.prototype;
31   
32     this.op = proto.op || undefined;
33   
34     this.options = Object.assign({}, proto.options);
35   
36     this._conditions = proto._conditions
37       ? utils.clone(proto._conditions)
38       : {};
39   
40     this._fields = proto._fields
41       ? utils.clone(proto._fields)
42       : undefined;
43   
44     this._updateDoc = proto._updateDoc
45       ? utils.clone(proto._updateDoc)
46       : undefined;
47   
48     this._path = proto._path || undefined;
49     this._distinctDoc = proto._distinctDoc || undefined;
50     this._collection = proto._collection || undefined;
51     this._traceFunction = proto._traceFunction || undefined;
52   
53     if (options) {
54       this.setOptions(options);
55     }
56   
57     if (criteria) {
58       this.find(criteria);
59     }
60   }

62   /**
63    * This is a parameter that the user can set which determines if mquery
64    * uses $within or $geoWithin for queries. It defaults to true which
65    * means $geoWithin will be used. If using MongoDB < 2.4 you should
66    * set this to false.
67    *
68    * @api public
69    * @property use$geoWithin
70    */
71   
72   let $withinCmd = '$geoWithin';
73   Object.defineProperty(Query, 'use$geoWithin', {
74     get: function() { return $withinCmd == '$geoWithin'; },
75     set: function(v) {
76       if (true === v) {
77         // mongodb >= 2.4
78         $withinCmd = '$geoWithin';
79       } else {
80         $withinCmd = '$within';
81       }
82     }

85   /**
86    * Converts this query to a constructor function with all arguments and options retained.
87    *
88    * #### Example:
89    *
90    *     // Create a query that will read documents with a "video" category from
91    *     // `aCollection` on the primary node in the replica-set unless it is down,
92    *     // in which case we'll read from a secondary node.
93    *     var query = mquery({ category: 'video' })
94    *     query.setOptions({ collection: aCollection, read: 'primaryPreferred' });
95    *
96    *     // create a constructor based off these settings
97    *     var Video = query.toConstructor();
98    *
99    *     // Video is now a subclass of mquery() and works the same way but with the
100   *     // default query parameters and options set.
101   *
102   *     // run a query with the previous settings but filter for movies with names
103   *     // that start with "Life".
104   *     Video().where({ name: /^Life/ }).exec(cb);
105   *
106   * @return {Query} new Query
107   * @api public
108   */
109  
110  Query.prototype.toConstructor = function toConstructor() {
111    function CustomQuery(criteria, options) {
112      if (!(this instanceof CustomQuery))
113        return new CustomQuery(criteria, options);
114      Query.call(this, criteria, options);
115    }
116  
117    utils.inherits(CustomQuery, Query);
118  
119    // set inherited defaults
120    const p = CustomQuery.prototype;
121  
122    p.options = {};
123    p.setOptions(this.options);
124  
125    p.op = this.op;
126    p._conditions = utils.clone(this._conditions);
127    p._fields = utils.clone(this._fields);
128    p._updateDoc = utils.clone(this._updateDoc);
129    p._path = this._path;
130    p._distinctDoc = this._distinctDoc;
131    p._collection = this._collection;
132    p._traceFunction = this._traceFunction;
133  
134    return CustomQuery;
135  };

160  Query.prototype.setOptions = function(options) {
161    if (!(options && utils.isObject(options)))
162      return this;
163  
164    // set arbitrary options
165    const methods = utils.keys(options);
166    let method;
167  
168    for (let i = 0; i < methods.length; ++i) {
169      method = methods[i];
170  
171      // use methods if exist (safer option manipulation)
172      if ('function' == typeof this[method]) {
173        const args = Array.isArray(options[method])
174          ? options[method]
175          : [options[method]];
176        this[method].apply(this, args);
177      } else {
178        this.options[method] = options[method];
179      }
180    }
181  
182    return this;
183  };

241  /**
242   * Specifies a `path` for use with chaining.
243   *
244   * #### Example:
245   *
246   *     // instead of writing:
247   *     await User.find({age: {$gte: 21, $lte: 65}});
248   *
249   *     // we can instead write:
250   *     User.where('age').gte(21).lte(65);
251   *
252   *     // passing query conditions is permitted
253   *     User.find().where({ name: 'vonderful' })
254   *
255   *     // chaining
256   *     await User
257   *       .where('age').gte(21).lte(65)
258   *       .where('name', /^vonderful/i)
259   *       .where('friends').slice(10)
260   *       .exec()
261   *
262   * @param {String} [path]
263   * @param {Object} [val]
264   * @return {Query} this
265   * @api public
266   */
267  
268  Query.prototype.where = function() {
269    if (!arguments.length) return this;
270    if (!this.op) this.op = 'find';
271  
272    const type = typeof arguments[0];
273  
274    if ('string' == type) {
275      this._path = arguments[0];
276  
277      if (2 === arguments.length) {
278        this._conditions[this._path] = arguments[1];
279      }
280  
281      return this;
282    }
283  
284    if ('object' == type && !Array.isArray(arguments[0])) {
285      return this.merge(arguments[0]);
286    }
287  
288    throw new TypeError('path must be a string or object');
289  };

673

675  
676    if ('function' === typeof arguments[0]) {
677      this._ensurePath('elemMatch');
678      path = this._path;
679      fn = arguments[0];
680    } else if (utils.isObject(arguments[0])) {

682      path = this._path;

690    } else {
691      throw new TypeError('Invalid argument');
692    }
693  
694    if (fn) {
695      criteria = new Query;
696      fn(criteria);
697      criteria = criteria._conditions;
698    }
699

707  /**
708   * Sugar for geo-spatial queries.
709   *
710   * #### Example:
711   *
712   *     query.within().box()
713   *     query.within().circle()
714   *     query.within().geometry()
715   *
716   *     query.where('loc').within({ center: [50,50], radius: 10, unique: true, spherical: true });
717   *     query.where('loc').within({ box: [[40.73, -73.9], [40.7, -73.988]] });
718   *     query.where('loc').within({ polygon: [[],[],[],[]] });
719   *
720   *     query.where('loc').within([], [], []) // polygon
721   *     query.where('loc').within([], []) // box
722   *     query.where('loc').within({ type: 'LineString', coordinates: [...] }); // geometry
723   *
724   * #### Note:
725   *
726   * Must be used after `where()`.
727   *
728   * @memberOf Query
729   * @return {Query} this
730   * @api public
731   */
732  
733  Query.prototype.within = function within() {
734    // opinionated, must be used after where
735    this._ensurePath('within');
736    this._geoComparison = $withinCmd;
737  
738    if (0 === arguments.length) {
739      return this;
740    }
741  
742    if (2 === arguments.length) {
743      return this.box.apply(this, arguments);
744    } else if (2 < arguments.length) {
745      return this.polygon.apply(this, arguments);
746    }
747  
748    const area = arguments[0];
749  
750    if (!area)
751      throw new TypeError('Invalid argument');
752  
753    if (area.center)
754      return this.circle(area);
755  
756    if (area.box)
757      return this.box.apply(this, area.box);
758  
759    if (area.polygon)
760      return this.polygon.apply(this, area.polygon);
761  
762    if (area.type && area.coordinates)
763      return this.geometry(area);
764  
765    throw new TypeError('Invalid argument');
766  };

775   *

911   * @param {Object} val

914   * @api public

918    let path, val;
919  
920    this._geoComparison = '$near';
921  
922    if (0 === arguments.length) {
923      return this;
924    } else if (1 === arguments.length) {
925      this._ensurePath('near');
926      path = this._path;
927      val = arguments[0];
928    } else if (2 === arguments.length) {
929      path = arguments[0];
930      val = arguments[1];
931    } else {
932      throw new TypeError('Invalid argument');

936      throw new Error('center is required');
937    }
938  
939    const conds = this._conditions[path] || (this._conditions[path] = {});
940

942      ? '$nearSphere'
943      : '$near';

1208 /**
1209  * Sets the sort order
1210  *
1211  * If an object is passed, values allowed are 'asc', 'desc', 'ascending', 'descending', 1, and -1.
1212  *
1213  * If a string is passed, it must be a space delimited list of path names. The sort order of each path is ascending unless the path name is prefixed with `-` which will be treated as descending.
1214  *
1215  * #### Example:
1216  *
1217  *     // these are equivalent
1218  *     query.sort({ field: 'asc', test: -1 });
1219  *     query.sort('field -test');
1220  *     query.sort([['field', 1], ['test', -1]]);
1221  *
1222  * #### Note:
1223  *
1224  *  - The array syntax `.sort([['field', 1], ['test', -1]])` can only be used with [mongodb driver >= 2.0.46](https://github.com/mongodb/node-mongodb-native/blob/2.1/HISTORY.md#2046-2015-10-15).
1225  *  - Cannot be used with `distinct()`
1226  *
1227  * @param {Object|String|Array} arg
1228  * @return {Query} this
1229  * @api public
1230  */
1231 
1232 Query.prototype.sort = function(arg) {
1233   if (!arg) return this;
1234   let i, len, field;
1235 
1236   this._validate('sort');
1237 
1238   const type = typeof arg;
1239 
1240   // .sort([['field', 1], ['test', -1]])
1241   if (Array.isArray(arg)) {
1242     len = arg.length;
1243     for (i = 0; i < arg.length; ++i) {
1244       if (!Array.isArray(arg[i])) {
1245         throw new Error('Invalid sort() argument, must be array of arrays');
1246       }
1247       _pushArr(this.options, arg[i][0], arg[i][1]);
1248     }
1249     return this;
1250   }
1251 
1252   // .sort('field -test')
1253   if (1 === arguments.length && 'string' == type) {
1254     arg = arg.split(/\s+/);
1255     len = arg.length;
1256     for (i = 0; i < len; ++i) {
1257       field = arg[i];
1258       if (!field) continue;
1259       const ascend = '-' == field[0] ? -1 : 1;
1260       if (ascend === -1) field = field.substring(1);
1261       push(this.options, field, ascend);
1262     }
1263 
1264     return this;
1265   }
1266 
1267   // .sort({ field: 1, test: -1 })
1268   if (utils.isObject(arg)) {
1269     const keys = utils.keys(arg);
1270     for (i = 0; i < keys.length; ++i) {
1271       field = keys[i];
1272       push(this.options, field, arg[field]);
1273     }
1274 
1275     return this;
1276   }
1277 
1278   if (typeof Map !== 'undefined' && arg instanceof Map) {
1279     _pushMap(this.options, arg);
1280     return this;
1281   }
1282   throw new TypeError('Invalid sort() argument. Must be a string, object, or array.');
1283 };

1298 function push(opts, field, value) {

1300     throw new TypeError('Can\'t mix sort syntaxes. Use either array or object:' +

1307     s = opts.sort || (opts.sort = {});
1308     s[field] = { $meta: value.$meta };
1309     return;
1310   }

1312   s = opts.sort || (opts.sort = {});
1313   let val = String(value || 1).toLowerCase();
1314   val = _validSortValue[val];
1315   if (!val) throw new TypeError('Invalid sort value: { ' + field + ': ' + value + ' }');
1316
