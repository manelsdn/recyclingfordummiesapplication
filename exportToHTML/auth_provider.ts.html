1    import type { Document } from '../../bson';
2    import { MongoRuntimeError } from '../../error';
3    import type { HandshakeDocument } from '../connect';
4    import type { Connection, ConnectionOptions } from '../connection';
5    import type { MongoCredentials } from './mongo_credentials';
6    
7    /**
8     * Context used during authentication
9     * @internal
10    */
11   export class AuthContext {
12     /** The connection to authenticate */
13     connection: Connection;
14     /** The credentials to use for authentication */
15     credentials?: MongoCredentials;
16     /** If the context is for reauthentication. */
17     reauthenticating = false;
18     /** The options passed to the `connect` method */
19     options: ConnectionOptions;
20   
21     /** A response from an initial auth attempt, only some mechanisms use this (e.g, SCRAM) */
22     response?: Document;
23     /** A random nonce generated for use in an authentication conversation */
24     nonce?: Buffer;
25   
26     constructor(
27       connection: Connection,
28       credentials: MongoCredentials | undefined,
29       options: ConnectionOptions
30     ) {
31       this.connection = connection;
32       this.credentials = credentials;
33       this.options = options;
34     }
35   }
36   
37   export abstract class AuthProvider {
38     /**
39      * Prepare the handshake document before the initial handshake.
40      *
41      * @param handshakeDoc - The document used for the initial handshake on a connection
42      * @param authContext - Context for authentication flow
43      */
44     async prepare(
45       handshakeDoc: HandshakeDocument,
46       _authContext: AuthContext
47     ): Promise<HandshakeDocument> {
48       return handshakeDoc;
49     }
50   
51     /**
52      * Authenticate
53      *
54      * @param context - A shared context for authentication flow
55      */
56     abstract auth(context: AuthContext): Promise<void>;
57   
58     /**
59      * Reauthenticate.
60      * @param context - The shared auth context.
61      */
62     async reauth(context: AuthContext): Promise<void> {
63       if (context.reauthenticating) {
64         throw new MongoRuntimeError('Reauthentication already in progress.');
65       }
66       try {
67         context.reauthenticating = true;
68         await this.auth(context);
69       } finally {
70         context.reauthenticating = false;
71       }
72     }
73   }
