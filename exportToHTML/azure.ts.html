6    const MINIMUM_TOKEN_REFRESH_IN_MILLISECONDS = 6000;
7    
8    /**
9     * The access token that libmongocrypt expects for Azure kms.
10    */
11   interface AccessToken {
12     accessToken: string;
13   }
14   
15   /**
16    * The response from the azure idms endpoint, including the `expiresOnTimestamp`.
17    * `expiresOnTimestamp` is needed for caching.
18    */
19   interface AzureTokenCacheEntry extends AccessToken {
20     accessToken: string;
21     expiresOnTimestamp: number;
22   }
23   
24   /**
25    * @internal
26    */
27   export class AzureCredentialCache {
28     cachedToken: AzureTokenCacheEntry | null = null;
29   
30     async getToken(): Promise<AccessToken> {
31       if (this.cachedToken == null || this.needsRefresh(this.cachedToken)) {
32         this.cachedToken = await this._getToken();
33       }
34   
35       return { accessToken: this.cachedToken.accessToken };
36     }
37   
38     needsRefresh(token: AzureTokenCacheEntry): boolean {
39       const timeUntilExpirationMS = token.expiresOnTimestamp - Date.now();
40       return timeUntilExpirationMS <= MINIMUM_TOKEN_REFRESH_IN_MILLISECONDS;
41     }
42   
43     /**
44      * exposed for testing
45      */
46     resetCache() {
47       this.cachedToken = null;
48     }
49   
50     /**
51      * exposed for testing
52      */
53     _getToken(): Promise<AzureTokenCacheEntry> {
54       return fetchAzureKMSToken();
55     }
56   }
57   
58   /** @internal */
59   export const tokenCache = new AzureCredentialCache();
60   
61   /** @internal */
62   async function parseResponse(response: {
63     body: string;
64     status?: number;
65   }): Promise<AzureTokenCacheEntry> {
66     const { status, body: rawBody } = response;
67   
68     const body: { expires_in?: number; access_token?: string } = (() => {
69       try {
70         return JSON.parse(rawBody);
71       } catch {
72         throw new MongoCryptAzureKMSRequestError('Malformed JSON body in GET request.');
73       }
74     })();
75   
76     if (status !== 200) {
77       throw new MongoCryptAzureKMSRequestError('Unable to complete request.', body);
78     }
79   
80     if (!body.access_token) {
81       throw new MongoCryptAzureKMSRequestError(
82         'Malformed response body - missing field `access_token`.'
83       );
84     }
85   
86     if (!body.expires_in) {
87       throw new MongoCryptAzureKMSRequestError(
88         'Malformed response body - missing field `expires_in`.'
89       );
90     }
91   
92     const expiresInMS = Number(body.expires_in) * 1000;
93     if (Number.isNaN(expiresInMS)) {
94       throw new MongoCryptAzureKMSRequestError(
95         'Malformed response body - unable to parse int from `expires_in` field.'
96       );
97     }
98   
99     return {
100      accessToken: body.access_token,
101      expiresOnTimestamp: Date.now() + expiresInMS
102    };
103  }

105  /**
106   * @internal
107   *
108   * exposed for CSFLE
109   * [prose test 18](https://github.com/mongodb/specifications/tree/master/source/client-side-encryption/tests#azure-imds-credentials)
110   */
111  export interface AzureKMSRequestOptions {
112    headers?: Document;
113    url?: URL | string;
114  }
115  
116  /**
117   * @internal
118   *
119   * parses any options provided by prose tests to `fetchAzureKMSToken` and merges them with
120   * the default values for headers and the request url.
121   */
122  export function prepareRequest(options: AzureKMSRequestOptions): {
123    headers: Document;
124    url: URL;
125  } {
126    const url = new URL(
127      options.url?.toString() ?? 'http://169.254.169.254/metadata/identity/oauth2/token'
128    );
129  
130    url.searchParams.append('api-version', '2018-02-01');
131    url.searchParams.append('resource', 'https://vault.azure.net');
132  
133    const headers = { ...options.headers, 'Content-Type': 'application/json', Metadata: true };
134    return { headers, url };
135  }
136  
137  /**
138   * @internal
139   *
140   * `AzureKMSRequestOptions` allows prose tests to modify the http request sent to the idms
141   * servers.  This is required to simulate different server conditions.  No options are expected to
142   * be set outside of tests.
143   *
144   * exposed for CSFLE
145   * [prose test 18](https://github.com/mongodb/specifications/tree/master/source/client-side-encryption/tests#azure-imds-credentials)
146   */
147  export async function fetchAzureKMSToken(
148    options: AzureKMSRequestOptions = {}
149  ): Promise<AzureTokenCacheEntry> {
150    const { headers, url } = prepareRequest(options);
151    const response = await get(url, { headers }).catch(error => {
152      if (error instanceof MongoCryptKMSRequestNetworkTimeoutError) {
153        throw new MongoCryptAzureKMSRequestError(`[Azure KMS] ${error.message}`);
154      }
155      throw error;
156    });
157    return parseResponse(response);
158  }

160  /**
161   * @internal
162   *
163   * @throws Will reject with a `MongoCryptError` if the http request fails or the http response is malformed.
164   */
165  export async function loadAzureCredentials(kmsProviders: KMSProviders): Promise<KMSProviders> {
166    const azure = await tokenCache.getToken();
167    return { ...kmsProviders, azure };
168  }
