51   function loadSnappy() {
52     if (Snappy == null) {
53       const snappyImport = getSnappy();
54       if ('kModuleError' in snappyImport) {
55         throw snappyImport.kModuleError;
56       }
57       Snappy = snappyImport;
58     }
59     return Snappy;

62   // Facilitate compressing a message using an agreed compressor
63   export async function compress(
64     options: { zlibCompressionLevel: number; agreedCompressor: CompressorName },
65     dataToBeCompressed: Buffer
66   ): Promise<Buffer> {
67     const zlibOptions = {} as zlib.ZlibOptions;
68     switch (options.agreedCompressor) {
69       case 'snappy': {
70         Snappy ??= loadSnappy();
71         return Snappy.compress(dataToBeCompressed);
72       }
73       case 'zstd': {
74         loadZstd();
75         if ('kModuleError' in zstd) {
76           throw zstd['kModuleError'];
77         }
78         return zstd.compress(dataToBeCompressed, ZSTD_COMPRESSION_LEVEL);
79       }
80       case 'zlib': {
81         if (options.zlibCompressionLevel) {
82           zlibOptions.level = options.zlibCompressionLevel;
83         }
84         return zlibDeflate(dataToBeCompressed, zlibOptions);
85       }
86       default: {
87         throw new MongoInvalidArgumentError(
88           `Unknown compressor ${options.agreedCompressor} failed to compress`
89         );
90       }
91     }
92   }

94   // Decompress a message using the given compressor
95   export async function decompress(compressorID: number, compressedData: Buffer): Promise<Buffer> {
96     if (
97       compressorID !== Compressor.snappy &&
98       compressorID !== Compressor.zstd &&
99       compressorID !== Compressor.zlib &&
100      compressorID !== Compressor.none
101    ) {
102      throw new MongoDecompressionError(
103        `Server sent message compressed using an unsupported compressor. (Received compressor ID ${compressorID})`
104      );
105    }
106  
107    switch (compressorID) {
108      case Compressor.snappy: {
109        Snappy ??= loadSnappy();
110        return Snappy.uncompress(compressedData, { asBuffer: true });
111      }
112      case Compressor.zstd: {
113        loadZstd();
114        if ('kModuleError' in zstd) {
115          throw zstd['kModuleError'];
116        }
117        return zstd.decompress(compressedData);
118      }
119      case Compressor.zlib: {
120        return zlibInflate(compressedData);
121      }
122      default: {
123        return compressedData;
124      }
125    }
126  }

139  /**
140   * @internal
141   *
142   * Compresses an OP_MSG or OP_QUERY message, if compression is configured.  This method
143   * also serializes the command to BSON.
144   */
145  export async function compressCommand(
146    command: WriteProtocolMessageType,
147    description: OperationDescription
148  ): Promise<Buffer> {
149    const finalCommand =
150      description.agreedCompressor === 'none' || !OpCompressedRequest.canCompress(command)
151        ? command
152        : new OpCompressedRequest(command, {
153            agreedCompressor: description.agreedCompressor ?? 'none',
154            zlibCompressionLevel: description.zlibCompressionLevel ?? 0
155          });
156    const data = await finalCommand.toBin();
157    return Buffer.concat(data);
158  }

160  /**
161   * @internal
162   *
163   * Decompresses an OP_MSG or OP_QUERY response from the server, if compression is configured.
164   *
165   * This method does not parse the response's BSON.
166   */
167  export async function decompressResponse(
168    message: Buffer
169  ): Promise<OpMsgResponse | OpQueryResponse> {
170    const messageHeader: MessageHeader = {
171      length: message.readInt32LE(0),
172      requestId: message.readInt32LE(4),
173      responseTo: message.readInt32LE(8),
174      opCode: message.readInt32LE(12)
175    };
176  
177    if (messageHeader.opCode !== OP_COMPRESSED) {
178      const ResponseType = messageHeader.opCode === OP_MSG ? OpMsgResponse : OpQueryResponse;
179      const messageBody = message.subarray(MESSAGE_HEADER_SIZE);
180      return new ResponseType(message, messageHeader, messageBody);
181    }
182  
183    const header: MessageHeader = {
184      ...messageHeader,
185      fromCompressed: true,
186      opCode: message.readInt32LE(MESSAGE_HEADER_SIZE),
187      length: message.readInt32LE(MESSAGE_HEADER_SIZE + 4)
188    };
189    const compressorID = message[MESSAGE_HEADER_SIZE + 8];
190    const compressedBuffer = message.slice(MESSAGE_HEADER_SIZE + 9);
191  
192    // recalculate based on wrapped opcode
193    const ResponseType = header.opCode === OP_MSG ? OpMsgResponse : OpQueryResponse;
194    const messageBody = await decompress(compressorID, compressedBuffer);
195    if (messageBody.length !== header.length) {
196      throw new MongoDecompressionError('Message body and message header must be the same length');
197    }
198    return new ResponseType(message, header, messageBody);
199  }
