1    import * as dns from 'dns';
2    import { clearTimeout, setTimeout } from 'timers';
3    
4    import { MongoRuntimeError } from '../error';
5    import { TypedEventEmitter } from '../mongo_types';
6    import { HostAddress, matchesParentDomain } from '../utils';

8    /**
9     * @internal
10    * @category Event
11    */
12   export class SrvPollingEvent {
13     srvRecords: dns.SrvRecord[];
14     constructor(srvRecords: dns.SrvRecord[]) {
15       this.srvRecords = srvRecords;
16     }
17   
18     hostnames(): Set<string> {
19       return new Set(this.srvRecords.map(r => HostAddress.fromSrvRecord(r).toString()));
20     }
21   }
22   
23   /** @internal */
24   export interface SrvPollerOptions {
25     srvServiceName: string;
26     srvMaxHosts: number;
27     srvHost: string;
28     heartbeatFrequencyMS: number;
29   }
30   
31   /** @internal */
32   export type SrvPollerEvents = {
33     srvRecordDiscovery(event: SrvPollingEvent): void;
34   };
35   
36   /** @internal */
37   export class SrvPoller extends TypedEventEmitter<SrvPollerEvents> {
38     srvHost: string;
39     rescanSrvIntervalMS: number;
40     heartbeatFrequencyMS: number;
41     haMode: boolean;
42     generation: number;
43     srvMaxHosts: number;
44     srvServiceName: string;
45     _timeout?: NodeJS.Timeout;
46   
47     /** @event */
48     static readonly SRV_RECORD_DISCOVERY = 'srvRecordDiscovery' as const;
49   
50     constructor(options: SrvPollerOptions) {
51       super();
52   
53       if (!options || !options.srvHost) {
54         throw new MongoRuntimeError('Options for SrvPoller must exist and include srvHost');
55       }
56   
57       this.srvHost = options.srvHost;
58       this.srvMaxHosts = options.srvMaxHosts ?? 0;
59       this.srvServiceName = options.srvServiceName ?? 'mongodb';
60       this.rescanSrvIntervalMS = 60000;
61       this.heartbeatFrequencyMS = options.heartbeatFrequencyMS ?? 10000;
62   
63       this.haMode = false;
64       this.generation = 0;
65   
66       this._timeout = undefined;
67     }
68   
69     get srvAddress(): string {
70       return `_${this.srvServiceName}._tcp.${this.srvHost}`;
71     }
72   
73     get intervalMS(): number {
74       return this.haMode ? this.heartbeatFrequencyMS : this.rescanSrvIntervalMS;
75     }
76   
77     start(): void {
78       if (!this._timeout) {
79         this.schedule();
80       }
81     }
82   
83     stop(): void {
84       if (this._timeout) {
85         clearTimeout(this._timeout);
86         this.generation += 1;
87         this._timeout = undefined;
88       }
89     }
90   
91     // TODO(NODE-4994): implement new logging logic for SrvPoller failures
92     schedule(): void {
93       if (this._timeout) {
94         clearTimeout(this._timeout);
95       }
96   
97       this._timeout = setTimeout(() => {
98         this._poll().catch(() => null);
99       }, this.intervalMS);
100    }
101  
102    success(srvRecords: dns.SrvRecord[]): void {
103      this.haMode = false;
104      this.schedule();
105      this.emit(SrvPoller.SRV_RECORD_DISCOVERY, new SrvPollingEvent(srvRecords));
106    }
107  
108    failure(): void {
109      this.haMode = true;
110      this.schedule();
111    }
112  
113    async _poll(): Promise<void> {
114      const generation = this.generation;
115      let srvRecords;
116  
117      try {
118        srvRecords = await dns.promises.resolveSrv(this.srvAddress);
119      } catch (dnsError) {
120        this.failure();
121        return;
122      }
123  
124      if (generation !== this.generation) {
125        return;
126      }
127  
128      const finalAddresses: dns.SrvRecord[] = [];
129      for (const record of srvRecords) {
130        if (matchesParentDomain(record.name, this.srvHost)) {
131          finalAddresses.push(record);
132        }
133      }
134  
135      if (!finalAddresses.length) {
136        this.failure();
137        return;
138      }
139  
140      this.success(finalAddresses);
141    }
142  }
