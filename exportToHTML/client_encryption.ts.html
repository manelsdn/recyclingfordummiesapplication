130    /**
131     * Creates a data key used for explicit encryption and inserts it into the key vault namespace
132     *
133     * @example
134     * ```ts
135     * // Using async/await to create a local key
136     * const dataKeyId = await clientEncryption.createDataKey('local');
137     * ```
138     *
139     * @example
140     * ```ts
141     * // Using async/await to create an aws key
142     * const dataKeyId = await clientEncryption.createDataKey('aws', {
143     *   masterKey: {
144     *     region: 'us-east-1',
145     *     key: 'xxxxxxxxxxxxxx' // CMK ARN here
146     *   }
147     * });
148     * ```
149     *
150     * @example
151     * ```ts
152     * // Using async/await to create an aws key with a keyAltName
153     * const dataKeyId = await clientEncryption.createDataKey('aws', {
154     *   masterKey: {
155     *     region: 'us-east-1',
156     *     key: 'xxxxxxxxxxxxxx' // CMK ARN here
157     *   },
158     *   keyAltNames: [ 'mySpecialKey' ]
159     * });
160     * ```
161     */
162    async createDataKey(
163      provider: ClientEncryptionDataKeyProvider,
164      options: ClientEncryptionCreateDataKeyProviderOptions = {}
165    ): Promise<UUID> {
166      if (options.keyAltNames && !Array.isArray(options.keyAltNames)) {
167        throw new MongoCryptInvalidArgumentError(
168          `Option "keyAltNames" must be an array of strings, but was of type ${typeof options.keyAltNames}.`
169        );
170      }
171  
172      let keyAltNames = undefined;
173      if (options.keyAltNames && options.keyAltNames.length > 0) {
174        keyAltNames = options.keyAltNames.map((keyAltName, i) => {
175          if (typeof keyAltName !== 'string') {
176            throw new MongoCryptInvalidArgumentError(
177              `Option "keyAltNames" must be an array of strings, but item at index ${i} was of type ${typeof keyAltName}`
178            );
179          }
180  
181          return serialize({ keyAltName });
182        });
183      }
184  
185      let keyMaterial = undefined;
186      if (options.keyMaterial) {
187        keyMaterial = serialize({ keyMaterial: options.keyMaterial });
188      }
189  
190      const dataKeyBson = serialize({
191        provider,
192        ...options.masterKey
193      });
194  
195      const context = this._mongoCrypt.makeDataKeyContext(dataKeyBson, {
196        keyAltNames,
197        keyMaterial
198      });
199  
200      const stateMachine = new StateMachine({
201        proxyOptions: this._proxyOptions,
202        tlsOptions: this._tlsOptions
203      });
204  
205      const dataKey = await stateMachine.execute<DataKey>(this, context);
206  
207      const { db: dbName, collection: collectionName } = MongoDBCollectionNamespace.fromString(
208        this._keyVaultNamespace
209      );
210  
211      const { insertedId } = await this._keyVaultClient
212        .db(dbName)
213        .collection<DataKey>(collectionName)
214        .insertOne(dataKey, { writeConcern: { w: 'majority' } });
215  
216      return insertedId;
217    }

219    /**
220     * Searches the keyvault for any data keys matching the provided filter.  If there are matches, rewrapManyDataKey then attempts to re-wrap the data keys using the provided options.
221     *
222     * If no matches are found, then no bulk write is performed.
223     *
224     * @example
225     * ```ts
226     * // rewrapping all data data keys (using a filter that matches all documents)
227     * const filter = {};
228     *
229     * const result = await clientEncryption.rewrapManyDataKey(filter);
230     * if (result.bulkWriteResult != null) {
231     *  // keys were re-wrapped, results will be available in the bulkWrite object.
232     * }
233     * ```
234     *
235     * @example
236     * ```ts
237     * // attempting to rewrap all data keys with no matches
238     * const filter = { _id: new Binary() } // assume _id matches no documents in the database
239     * const result = await clientEncryption.rewrapManyDataKey(filter);
240     *
241     * if (result.bulkWriteResult == null) {
242     *  // no keys matched, `bulkWriteResult` does not exist on the result object
243     * }
244     * ```
245     */
246    async rewrapManyDataKey(
247      filter: Filter<DataKey>,
248      options: ClientEncryptionRewrapManyDataKeyProviderOptions
249    ): Promise<{ bulkWriteResult?: BulkWriteResult }> {
250      let keyEncryptionKeyBson = undefined;
251      if (options) {
252        const keyEncryptionKey = Object.assign({ provider: options.provider }, options.masterKey);
253        keyEncryptionKeyBson = serialize(keyEncryptionKey);
254      }
255      const filterBson = serialize(filter);
256      const context = this._mongoCrypt.makeRewrapManyDataKeyContext(filterBson, keyEncryptionKeyBson);
257      const stateMachine = new StateMachine({
258        proxyOptions: this._proxyOptions,
259        tlsOptions: this._tlsOptions
260      });
261  
262      const { v: dataKeys } = await stateMachine.execute<{ v: DataKey[] }>(this, context);
263      if (dataKeys.length === 0) {
264        return {};
265      }
266  
267      const { db: dbName, collection: collectionName } = MongoDBCollectionNamespace.fromString(
268        this._keyVaultNamespace
269      );
270  
271      const replacements = dataKeys.map(
272        (key: DataKey): AnyBulkWriteOperation<DataKey> => ({
273          updateOne: {
274            filter: { _id: key._id },
275            update: {
276              $set: {
277                masterKey: key.masterKey,
278                keyMaterial: key.keyMaterial
279              },
280              $currentDate: {
281                updateDate: true
282              }
283            }
284          }
285        })
286      );
287  
288      const result = await this._keyVaultClient
289        .db(dbName)
290        .collection<DataKey>(collectionName)
291        .bulkWrite(replacements, {
292          writeConcern: { w: 'majority' }
293        });
294  
295      return { bulkWriteResult: result };
296    }

437    /**
438     * Adds a keyAltName to a key identified by the provided _id.
439     *
440     * This method resolves to/returns the *old* key value (prior to removing the new altKeyName).
441     *
442     * If the removed keyAltName is the last keyAltName for that key, the `altKeyNames` property is unset from the document.
443     *
444     * @param _id - The id of the document to update.
445     * @param keyAltName - a keyAltName to search for a key
446     * @returns Returns a promise that either resolves to a {@link DataKey} if a document matches the key or null if no documents
447     * match the id.  The promise rejects with an error if an error is thrown.
448     * @example
449     * ```ts
450     * // removing a key alt name from a data key
451     * const id = new Binary();  // id is a bson binary subtype 4 object
452     * const keyAltName = 'keyAltName';
453     * const oldKey = await clientEncryption.removeKeyAltName(id, keyAltName);
454     *
455     * if (!oldKey) {
456     *  // null is returned if there is no matching document with an id matching the supplied id
457     * }
458     * ```
459     */
460    async removeKeyAltName(_id: Binary, keyAltName: string): Promise<WithId<DataKey> | null> {
461      const { db: dbName, collection: collectionName } = MongoDBCollectionNamespace.fromString(
462        this._keyVaultNamespace
463      );
464  
465      const pipeline = [
466        {
467          $set: {
468            keyAltNames: {
469              $cond: [
470                {
471                  $eq: ['$keyAltNames', [keyAltName]]
472                },
473                '$$REMOVE',
474                {
475                  $filter: {
476                    input: '$keyAltNames',
477                    cond: {
478                      $ne: ['$$this', keyAltName]
479                    }
480                  }
481                }
482              ]
483            }
484          }
485        }
486      ];
487      const value = await this._keyVaultClient
488        .db(dbName)
489        .collection<DataKey>(collectionName)
490        .findOneAndUpdate({ _id }, pipeline, {
491          writeConcern: { w: 'majority' },
492          returnDocument: 'before'
493        });
494  
495      return value;
496    }

498    /**
499     * A convenience method for creating an encrypted collection.
500     * This method will create data keys for any encryptedFields that do not have a `keyId` defined
501     * and then create a new collection with the full set of encryptedFields.
502     *
503     * @param db - A Node.js driver Db object with which to create the collection
504     * @param name - The name of the collection to be created
505     * @param options - Options for createDataKey and for createCollection
506     * @returns created collection and generated encryptedFields
507     * @throws MongoCryptCreateDataKeyError - If part way through the process a createDataKey invocation fails, an error will be rejected that has the partial `encryptedFields` that were created.
508     * @throws MongoCryptCreateEncryptedCollectionError - If creating the collection fails, an error will be rejected that has the entire `encryptedFields` that were created.
509     */
510    async createEncryptedCollection<TSchema extends Document = Document>(
511      db: Db,
512      name: string,
513      options: {
514        provider: ClientEncryptionDataKeyProvider;
515        createCollectionOptions: Omit<CreateCollectionOptions, 'encryptedFields'> & {
516          encryptedFields: Document;
517        };
518        masterKey?: AWSEncryptionKeyOptions | AzureEncryptionKeyOptions | GCPEncryptionKeyOptions;
519      }
520    ): Promise<{ collection: Collection<TSchema>; encryptedFields: Document }> {
521      const {
522        provider,
523        masterKey,
524        createCollectionOptions: {
525          encryptedFields: { ...encryptedFields },
526          ...createCollectionOptions
527        }
528      } = options;
529  
530      if (Array.isArray(encryptedFields.fields)) {
531        const createDataKeyPromises = encryptedFields.fields.map(async field =>
532          field == null || typeof field !== 'object' || field.keyId != null
533            ? field
534            : {
535                ...field,
536                keyId: await this.createDataKey(provider, { masterKey })
537              }
538        );
539  
540        const createDataKeyResolutions = await Promise.allSettled(createDataKeyPromises);
541  
542        encryptedFields.fields = createDataKeyResolutions.map((resolution, index) =>
543          resolution.status === 'fulfilled' ? resolution.value : encryptedFields.fields[index]
544        );
545  
546        const rejection = createDataKeyResolutions.find(
547          (result): result is PromiseRejectedResult => result.status === 'rejected'
548        );
549        if (rejection != null) {
550          throw new MongoCryptCreateDataKeyError(encryptedFields, { cause: rejection.reason });
551        }
552      }
553  
554      try {
555        const collection = await db.createCollection<TSchema>(name, {
556          ...createCollectionOptions,
557          encryptedFields
558        });
559        return { collection, encryptedFields };
560      } catch (cause) {
561        throw new MongoCryptCreateEncryptedCollectionError(encryptedFields, { cause });
562      }
563    }

622     *
623     * @param value - An encrypted value
624     * @returns a Promise that either resolves with the decrypted value, or rejects with an error
625     *
626     * @example
627     * ```ts

634    async decrypt<T = any>(value: Binary): Promise<T> {
635      const valueBuffer = serialize({ v: value });
636      const context = this._mongoCrypt.makeExplicitDecryptionContext(valueBuffer);
637  
638      const stateMachine = new StateMachine({
639        proxyOptions: this._proxyOptions,
640        tlsOptions: this._tlsOptions
641      });
642  
643      const { v } = await stateMachine.execute<{ v: T }>(this, context);
644  
645      return v;
646    }

664    /**
665     * @internal
666     * A helper that perform explicit encryption of values and expressions.
667     * Explicitly encrypt a provided value. Note that either `options.keyId` or `options.keyAltName` must
668     * be specified. Specifying both `options.keyId` and `options.keyAltName` is considered an error.
669     *
670     * @param value - The value that you wish to encrypt. Must be of a type that can be serialized into BSON
671     * @param expressionMode - a boolean that indicates whether or not to encrypt the value as an expression
672     * @param options - options to pass to encrypt
673     * @returns the raw result of the call to stateMachine.execute().  When expressionMode is set to true, the return
674     *          value will be a bson document.  When false, the value will be a BSON Binary.
675     *
676     */
677    private async _encrypt(
678      value: unknown,
679      expressionMode: boolean,
680      options: ClientEncryptionEncryptOptions
681    ): Promise<Binary> {
682      const { algorithm, keyId, keyAltName, contentionFactor, queryType, rangeOptions } = options;
683      const contextOptions: ExplicitEncryptionContextOptions = {
684        expressionMode,
685        algorithm
686      };
687      if (keyId) {
688        contextOptions.keyId = keyId.buffer;
689      }
690      if (keyAltName) {
691        if (keyId) {
692          throw new MongoCryptInvalidArgumentError(
693            `"options" cannot contain both "keyId" and "keyAltName"`
694          );
695        }
696        if (typeof keyAltName !== 'string') {
697          throw new MongoCryptInvalidArgumentError(
698            `"options.keyAltName" must be of type string, but was of type ${typeof keyAltName}`
699          );
700        }
701  
702        contextOptions.keyAltName = serialize({ keyAltName });
703      }
704      if (typeof contentionFactor === 'number' || typeof contentionFactor === 'bigint') {
705        contextOptions.contentionFactor = contentionFactor;
706      }
707      if (typeof queryType === 'string') {
708        contextOptions.queryType = queryType;
709      }
710  
711      if (typeof rangeOptions === 'object') {
712        contextOptions.rangeOptions = serialize(rangeOptions);
713      }
714  
715      const valueBuffer = serialize({ v: value });
716      const stateMachine = new StateMachine({
717        proxyOptions: this._proxyOptions,
718        tlsOptions: this._tlsOptions
719      });
720      const context = this._mongoCrypt.makeExplicitEncryptionContext(valueBuffer, contextOptions);
721  
722      const result = await stateMachine.execute<{ v: Binary }>(this, context);
723      return result.v;
724    }
725  }
