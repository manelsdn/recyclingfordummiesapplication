16   module.exports = function assignVals(o) {
17     // Options that aren't explicitly listed in `populateOptions`
18     const userOptions = Object.assign({}, get(o, 'allOptions.options.options'), get(o, 'allOptions.options'));
19     // `o.options` contains options explicitly listed in `populateOptions`, like
20     // `match` and `limit`.
21     const populateOptions = Object.assign({}, o.options, userOptions, {
22       justOne: o.justOne,
23       isVirtual: o.isVirtual
24     });
25     populateOptions.$nullIfNotFound = o.isVirtual;
26     const populatedModel = o.populatedModel;
27   
28     const originalIds = [].concat(o.rawIds);
29   
30     // replace the original ids in our intermediate _ids structure
31     // with the documents found by query
32     o.allIds = [].concat(o.allIds);
33     assignRawDocsToIdStructure(o.rawIds, o.rawDocs, o.rawOrder, populateOptions);
34   
35     // now update the original documents being populated using the
36     // result structure that contains real documents.
37     const docs = o.docs;
38     const rawIds = o.rawIds;
39     const options = o.options;
40     const count = o.count && o.isVirtual;
41     let i;
42     let setValueIndex = 0;
43   
44     function setValue(val) {
45       ++setValueIndex;
46       if (count) {
47         return val;
48       }
49       if (val instanceof SkipPopulateValue) {
50         return val.val;
51       }
52       if (val === void 0) {
53         return val;
54       }
55   
56       const _allIds = o.allIds[i];
57   
58       if (o.path.endsWith('.$*')) {
59         // Skip maps re: gh-12494
60         return valueFilter(val, options, populateOptions, _allIds);
61       }
62   
63       if (o.justOne === true && Array.isArray(val)) {
64         // Might be an embedded discriminator (re: gh-9244) with multiple models, so make sure to pick the right
65         // model before assigning.
66         const ret = [];
67         for (const doc of val) {
68           const _docPopulatedModel = leanPopulateMap.get(doc);
69           if (_docPopulatedModel == null || _docPopulatedModel === populatedModel) {
70             ret.push(doc);
71           }
72         }
73         // Since we don't want to have to create a new mongoosearray, make sure to
74         // modify the array in place
75         while (val.length > ret.length) {
76           Array.prototype.pop.apply(val, []);
77         }
78         for (let i = 0; i < ret.length; ++i) {
79           val[i] = ret[i];
80         }
81   
82         return valueFilter(val[0], options, populateOptions, _allIds);
83       } else if (o.justOne === false && !Array.isArray(val)) {
84         return valueFilter([val], options, populateOptions, _allIds);
85       } else if (o.justOne === true && !Array.isArray(val) && Array.isArray(_allIds)) {
86         return valueFilter(val, options, populateOptions, val == null ? val : _allIds[setValueIndex - 1]);
87       }
88       return valueFilter(val, options, populateOptions, _allIds);
89     }
90   
91     for (i = 0; i < docs.length; ++i) {
92       setValueIndex = 0;
93       const _path = o.path.endsWith('.$*') ? o.path.slice(0, -3) : o.path;
94       const existingVal = mpath.get(_path, docs[i], lookupLocalFields);
95       if (existingVal == null && !getVirtual(o.originalModel.schema, _path)) {
96         continue;
97       }
98   
99       let valueToSet;
100      if (count) {
101        valueToSet = numDocs(rawIds[i]);
102      } else if (Array.isArray(o.match)) {
103        valueToSet = Array.isArray(rawIds[i]) ?
104          rawIds[i].filter(sift(o.match[i])) :
105          [rawIds[i]].filter(sift(o.match[i]))[0];
106      } else {
107        valueToSet = rawIds[i];
108      }
109  
110      // If we're populating a map, the existing value will be an object, so
111      // we need to transform again
112      const originalSchema = o.originalModel.schema;
113      const isDoc = get(docs[i], '$__', null) != null;
114      let isMap = isDoc ?
115        existingVal instanceof Map :
116        utils.isPOJO(existingVal);
117      // If we pass the first check, also make sure the local field's schematype
118      // is map (re: gh-6460)
119      isMap = isMap && get(originalSchema._getSchema(_path), '$isSchemaMap');
120      if (!o.isVirtual && isMap) {
121        const _keys = existingVal instanceof Map ?
122          Array.from(existingVal.keys()) :
123          Object.keys(existingVal);
124        valueToSet = valueToSet.reduce((cur, v, i) => {
125          cur.set(_keys[i], v);
126          return cur;
127        }, new Map());
128      }
129  
130      if (isDoc && Array.isArray(valueToSet)) {
131        for (const val of valueToSet) {
132          if (val != null && val.$__ != null) {
133            val.$__.parent = docs[i];
134          }
135        }
136      } else if (isDoc && valueToSet != null && valueToSet.$__ != null) {
137        valueToSet.$__.parent = docs[i];
138      }
139  
140      if (o.isVirtual && isDoc) {
141        docs[i].$populated(_path, o.justOne ? originalIds[0] : originalIds, o.allOptions);
142        // If virtual populate and doc is already init-ed, need to walk through
143        // the actual doc to set rather than setting `_doc` directly
144        if (Array.isArray(valueToSet)) {
145          valueToSet = valueToSet.map(v => v == null ? void 0 : v);
146        }
147        mpath.set(_path, valueToSet, docs[i], void 0, setValue, false);
148        continue;
149      }
150  
151      const parts = _path.split('.');
152      let cur = docs[i];
153      for (let j = 0; j < parts.length - 1; ++j) {
154        // If we get to an array with a dotted path, like `arr.foo`, don't set
155        // `foo` on the array.
156        if (Array.isArray(cur) && !utils.isArrayIndex(parts[j])) {
157          break;
158        }
159  
160        if (parts[j] === '$*') {
161          break;
162        }
163  
164        if (cur[parts[j]] == null) {
165          // If nothing to set, avoid creating an unnecessary array. Otherwise
166          // we'll end up with a single doc in the array with only defaults.
167          // See gh-8342, gh-8455
168          const curPath = parts.slice(0, j + 1).join('.');
169          const schematype = originalSchema._getSchema(curPath);
170          if (valueToSet == null && schematype != null && schematype.$isMongooseArray) {
171            break;
172          }
173          cur[parts[j]] = {};
174        }
175        cur = cur[parts[j]];
176        // If the property in MongoDB is a primitive, we won't be able to populate
177        // the nested path, so skip it. See gh-7545
178        if (typeof cur !== 'object') {
179          break;
180        }
181      }
182      if (docs[i].$__) {
183        o.allOptions.options[populateModelSymbol] = o.allOptions.model;
184        docs[i].$populated(_path, o.unpopulatedValues[i], o.allOptions.options);
185  
186        if (valueToSet != null && valueToSet.$__ != null) {
187          valueToSet.$__.wasPopulated = { value: o.unpopulatedValues[i] };
188        }
189  
190        if (valueToSet instanceof Map && !valueToSet.$isMongooseMap) {
191          valueToSet = new MongooseMap(valueToSet, _path, docs[i], docs[i].schema.path(_path).$__schemaType);
192        }
193      }
194  
195      // If lean, need to check that each individual virtual respects
196      // `justOne`, because you may have a populated virtual with `justOne`
197      // underneath an array. See gh-6867
198      mpath.set(_path, valueToSet, docs[i], lookupLocalFields, setValue, false);
199  
200      if (docs[i].$__) {
201        markArraySubdocsPopulated(docs[i], [o.allOptions.options]);
202      }
203    }
204  };

206  function numDocs(v) {
207    if (Array.isArray(v)) {
208      // If setting underneath an array of populated subdocs, we may have an
209      // array of arrays. See gh-7573
210      if (v.some(el => Array.isArray(el) || el === null)) {
211        return v.map(el => {
212          if (el == null) {
213            return 0;
214          }
215          if (Array.isArray(el)) {
216            return el.filter(el => el != null).length;
217          }
218          return 1;
219        });
220      }
221      return v.filter(el => el != null).length;
222    }
223    return v == null ? 0 : 1;
224  }

226  /**
227   * 1) Apply backwards compatible find/findOne behavior to sub documents
228   *
229   *    find logic:
230   *      a) filter out non-documents
231   *      b) remove _id from sub docs when user specified
232   *
233   *    findOne
234   *      a) if no doc found, set to null
235   *      b) remove _id from sub docs when user specified
236   *
237   * 2) Remove _ids when specified by users query.
238   *
239   * background:
240   * _ids are left in the query even when user excludes them so
241   * that population mapping can occur.
242   * @param {Any} val
243   * @param {Object} assignmentOpts
244   * @param {Object} populateOptions
245   * @param {Function} [populateOptions.transform]
246   * @param {Boolean} allIds
247   * @api private
248   */
249  
250  function valueFilter(val, assignmentOpts, populateOptions, allIds) {
251    const userSpecifiedTransform = typeof populateOptions.transform === 'function';
252    const transform = userSpecifiedTransform ? populateOptions.transform : noop;
253    if (Array.isArray(val)) {
254      // find logic
255      const ret = [];
256      const numValues = val.length;
257      for (let i = 0; i < numValues; ++i) {
258        let subdoc = val[i];
259        const _allIds = Array.isArray(allIds) ? allIds[i] : allIds;
260        if (!isPopulatedObject(subdoc) && (!populateOptions.retainNullValues || subdoc != null) && !userSpecifiedTransform) {
261          continue;
262        } else if (!populateOptions.retainNullValues && subdoc == null) {
263          continue;
264        } else if (userSpecifiedTransform) {
265          subdoc = transform(isPopulatedObject(subdoc) ? subdoc : null, _allIds);
266        }
267        maybeRemoveId(subdoc, assignmentOpts);
268        ret.push(subdoc);
269        if (assignmentOpts.originalLimit &&
270            ret.length >= assignmentOpts.originalLimit) {
271          break;
272        }
273      }
274  
275      const rLen = ret.length;
276      // Since we don't want to have to create a new mongoosearray, make sure to
277      // modify the array in place
278      while (val.length > rLen) {
279        Array.prototype.pop.apply(val, []);
280      }
281      let i = 0;
282      if (utils.isMongooseArray(val)) {
283        for (i = 0; i < rLen; ++i) {
284          val.set(i, ret[i], true);
285        }
286      } else {
287        for (i = 0; i < rLen; ++i) {
288          val[i] = ret[i];
289        }
290      }
291      return val;
292    }
293  
294    // findOne
295    if (isPopulatedObject(val) || utils.isPOJO(val)) {
296      maybeRemoveId(val, assignmentOpts);
297      return transform(val, allIds);
298    }
299    if (val instanceof Map) {
300      return val;
301    }
302  
303    if (populateOptions.justOne === false) {
304      return [];
305    }
306  
307    return val == null ? transform(val, allIds) : transform(null, allIds);
308  }

310  /**
311   * Remove _id from `subdoc` if user specified "lean" query option
312   * @param {Document} subdoc
313   * @param {Object} assignmentOpts
314   * @api private
315   */
316  
317  function maybeRemoveId(subdoc, assignmentOpts) {
318    if (subdoc != null && assignmentOpts.excludeId) {
319      if (typeof subdoc.$__setValue === 'function') {
320        delete subdoc._doc._id;
321      } else {
322        delete subdoc._id;
323      }
324    }
325  }

327  /**
328   * Determine if `obj` is something we can set a populated path to. Can be a
329   * document, a lean document, or an array/map that contains docs.
330   * @param {Any} obj
331   * @api private
332   */
333  
334  function isPopulatedObject(obj) {
335    if (obj == null) {
336      return false;
337    }
338  
339    return Array.isArray(obj) ||
340      obj.$isMongooseMap ||
341      obj.$__ != null ||
342      leanPopulateMap.has(obj);
343  }
