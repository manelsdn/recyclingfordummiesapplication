52   /**
53    * Lookup a `mongodb+srv` connection string, combine the parts and reparse it as a normal
54    * connection string.
55    *
56    * @param uri - The connection string to parse
57    * @param options - Optional user provided connection string options
58    */
59   export async function resolveSRVRecord(options: MongoOptions): Promise<HostAddress[]> {
60     if (typeof options.srvHost !== 'string') {
61       throw new MongoAPIError('Option "srvHost" must not be empty');
62     }
63   
64     if (options.srvHost.split('.').length < 3) {
65       // TODO(NODE-3484): Replace with MongoConnectionStringError
66       throw new MongoAPIError('URI must include hostname, domain name, and tld');
67     }
68   
69     // Resolve the SRV record and use the result as the list of hosts to connect to.
70     const lookupAddress = options.srvHost;
71     const addresses = await dns.promises.resolveSrv(
72       `_${options.srvServiceName}._tcp.${lookupAddress}`
73     );
74   
75     if (addresses.length === 0) {
76       throw new MongoAPIError('No addresses found at host');
77     }
78   
79     for (const { name } of addresses) {
80       if (!matchesParentDomain(name, lookupAddress)) {
81         throw new MongoAPIError('Server record does not share hostname with parent URI');
82       }
83     }
84   
85     const hostAddresses = addresses.map(r => HostAddress.fromString(`${r.name}:${r.port ?? 27017}`));
86   
87     validateLoadBalancedOptions(hostAddresses, options, true);
88   
89     // Resolve TXT record and add options from there if they exist.
90     let record;
91     try {
92       record = await dns.promises.resolveTxt(lookupAddress);
93     } catch (error) {
94       if (error.code !== 'ENODATA' && error.code !== 'ENOTFOUND') {
95         throw error;
96       }
97       return hostAddresses;
98     }
99   
100    if (record.length > 1) {
101      throw new MongoParseError('Multiple text records not allowed');
102    }
103  
104    const txtRecordOptions = new URLSearchParams(record[0].join(''));
105    const txtRecordOptionKeys = [...txtRecordOptions.keys()];
106    if (txtRecordOptionKeys.some(key => !VALID_TXT_RECORDS.includes(key))) {
107      throw new MongoParseError(`Text record may only set any of: ${VALID_TXT_RECORDS.join(', ')}`);
108    }
109  
110    if (VALID_TXT_RECORDS.some(option => txtRecordOptions.get(option) === '')) {
111      throw new MongoParseError('Cannot have empty URI params in DNS TXT Record');
112    }
113  
114    const source = txtRecordOptions.get('authSource') ?? undefined;
115    const replicaSet = txtRecordOptions.get('replicaSet') ?? undefined;
116    const loadBalanced = txtRecordOptions.get('loadBalanced') ?? undefined;
117  
118    if (
119      !options.userSpecifiedAuthSource &&
120      source &&
121      options.credentials &&
122      !AUTH_MECHS_AUTH_SRC_EXTERNAL.has(options.credentials.mechanism)
123    ) {
124      options.credentials = MongoCredentials.merge(options.credentials, { source });
125    }
126  
127    if (!options.userSpecifiedReplicaSet && replicaSet) {
128      options.replicaSet = replicaSet;
129    }
130  
131    if (loadBalanced === 'true') {
132      options.loadBalanced = true;
133    }
134  
135    if (options.replicaSet && options.srvMaxHosts > 0) {
136      throw new MongoParseError('Cannot combine replicaSet option with srvMaxHosts');
137    }
138  
139    validateLoadBalancedOptions(hostAddresses, options, true);
140  
141    return hostAddresses;
142  }

150  function checkTLSOptions(allOptions: CaseInsensitiveMap): void {
151    if (!allOptions) return;
152    const check = (a: string, b: string) => {
153      if (allOptions.has(a) && allOptions.has(b)) {
154        throw new MongoAPIError(`The '${a}' option cannot be used with the '${b}' option`);
155      }
156    };
157    check('tlsInsecure', 'tlsAllowInvalidCertificates');
158    check('tlsInsecure', 'tlsAllowInvalidHostnames');
159    check('tlsInsecure', 'tlsDisableCertificateRevocationCheck');
160    check('tlsInsecure', 'tlsDisableOCSPEndpointCheck');
161    check('tlsAllowInvalidCertificates', 'tlsDisableCertificateRevocationCheck');
162    check('tlsAllowInvalidCertificates', 'tlsDisableOCSPEndpointCheck');
163    check('tlsDisableCertificateRevocationCheck', 'tlsDisableOCSPEndpointCheck');
164  }
165  function getBoolean(name: string, value: unknown): boolean {
166    if (typeof value === 'boolean') return value;
167    switch (value) {
168      case 'true':
169        return true;
170      case 'false':
171        return false;
172      default:
173        throw new MongoParseError(`${name} must be either "true" or "false"`);
174    }
175  }

177  function getIntFromOptions(name: string, value: unknown): number {
178    const parsedInt = parseInteger(value);
179    if (parsedInt != null) {
180      return parsedInt;
181    }
182    throw new MongoParseError(`Expected ${name} to be stringified int value, got: ${value}`);
183  }

185  function getUIntFromOptions(name: string, value: unknown): number {
186    const parsedValue = getIntFromOptions(name, value);
187    if (parsedValue < 0) {
188      throw new MongoParseError(`${name} can only be a positive int value, got: ${value}`);
189    }
190    return parsedValue;

193  function* entriesFromString(value: string): Generator<[string, string]> {
194    if (value === '') {
195      return;
196    }
197    const keyValuePairs = value.split(',');
198    for (const keyValue of keyValuePairs) {
199      const [key, value] = keyValue.split(/:(.*)/);
200      if (value == null) {
201        throw new MongoParseError('Cannot have undefined values in key value pairs');
202      }
203  
204      yield [key, value];
205    }

208  class CaseInsensitiveMap<Value = any> extends Map<string, Value> {
209    constructor(entries: Array<[string, any]> = []) {
210      super(entries.map(([k, v]) => [k.toLowerCase(), v]));
211    }
212    override has(k: string) {
213      return super.has(k.toLowerCase());
214    }
215    override get(k: string) {
216      return super.get(k.toLowerCase());
217    }
218    override set(k: string, v: any) {
219      return super.set(k.toLowerCase(), v);
220    }
221    override delete(k: string): boolean {
222      return super.delete(k.toLowerCase());
223    }

226  export function parseOptions(
227    uri: string,
228    mongoClient: MongoClient | MongoClientOptions | undefined = undefined,
229    options: MongoClientOptions = {}
230  ): MongoOptions {
231    if (mongoClient != null && !(mongoClient instanceof MongoClient)) {
232      options = mongoClient;
233      mongoClient = undefined;
234    }
235  
236    // validate BSONOptions
237    if (options.useBigInt64 && typeof options.promoteLongs === 'boolean' && !options.promoteLongs) {
238      throw new MongoAPIError('Must request either bigint or Long for int64 deserialization');
239    }
240  
241    if (options.useBigInt64 && typeof options.promoteValues === 'boolean' && !options.promoteValues) {
242      throw new MongoAPIError('Must request either bigint or Long for int64 deserialization');
243    }
244  
245    const url = new ConnectionString(uri);
246    const { hosts, isSRV } = url;
247  
248    const mongoOptions = Object.create(null);
249  
250    // Feature flags
251    for (const flag of Object.getOwnPropertySymbols(options)) {
252      if (FEATURE_FLAGS.has(flag)) {
253        mongoOptions[flag] = options[flag];
254      }
255    }
256  
257    mongoOptions.hosts = isSRV ? [] : hosts.map(HostAddress.fromString);
258  
259    const urlOptions = new CaseInsensitiveMap<unknown[]>();
260  
261    if (url.pathname !== '/' && url.pathname !== '') {
262      const dbName = decodeURIComponent(
263        url.pathname[0] === '/' ? url.pathname.slice(1) : url.pathname
264      );
265      if (dbName) {
266        urlOptions.set('dbName', [dbName]);
267      }
268    }
269  
270    if (url.username !== '') {
271      const auth: Document = {
272        username: decodeURIComponent(url.username)
273      };
274  
275      if (typeof url.password === 'string') {
276        auth.password = decodeURIComponent(url.password);
277      }
278  
279      urlOptions.set('auth', [auth]);
280    }
281  
282    for (const key of url.searchParams.keys()) {
283      const values = url.searchParams.getAll(key);
284  
285      const isReadPreferenceTags = /readPreferenceTags/i.test(key);
286  
287      if (!isReadPreferenceTags && values.length > 1) {
288        throw new MongoInvalidArgumentError(
289          `URI option "${key}" cannot appear more than once in the connection string`
290        );
291      }
292  
293      if (!isReadPreferenceTags && values.includes('')) {
294        throw new MongoAPIError(`URI option "${key}" cannot be specified with no value`);
295      }
296  
297      if (!urlOptions.has(key)) {
298        urlOptions.set(key, values);
299      }
300    }
301  
302    const objectOptions = new CaseInsensitiveMap<unknown>(
303      Object.entries(options).filter(([, v]) => v != null)
304    );
305  
306    // Validate options that can only be provided by one of uri or object
307  
308    if (urlOptions.has('serverApi')) {
309      throw new MongoParseError(
310        'URI cannot contain `serverApi`, it can only be passed to the client'
311      );
312    }
313  
314    const uriMechanismProperties = urlOptions.get('authMechanismProperties');
315    if (uriMechanismProperties) {
316      for (const property of uriMechanismProperties) {
317        if (/(^|,)ALLOWED_HOSTS:/.test(property as string)) {
318          throw new MongoParseError(
319            'Auth mechanism property ALLOWED_HOSTS is not allowed in the connection string.'
320          );
321        }
322      }
323    }
324  
325    if (objectOptions.has('loadBalanced')) {
326      throw new MongoParseError('loadBalanced is only a valid option in the URI');
327    }
328  
329    // All option collection
330  
331    const allProvidedOptions = new CaseInsensitiveMap<unknown[]>();
332  
333    const allProvidedKeys = new Set<string>([...urlOptions.keys(), ...objectOptions.keys()]);
334  
335    for (const key of allProvidedKeys) {
336      const values = [];
337      const objectOptionValue = objectOptions.get(key);
338      if (objectOptionValue != null) {
339        values.push(objectOptionValue);
340      }
341  
342      const urlValues = urlOptions.get(key) ?? [];
343      values.push(...urlValues);
344      allProvidedOptions.set(key, values);
345    }
346  
347    if (allProvidedOptions.has('tls') || allProvidedOptions.has('ssl')) {
348      const tlsAndSslOpts = (allProvidedOptions.get('tls') || [])
349        .concat(allProvidedOptions.get('ssl') || [])
350        .map(getBoolean.bind(null, 'tls/ssl'));
351      if (new Set(tlsAndSslOpts).size !== 1) {
352        throw new MongoParseError('All values of tls/ssl must be the same.');
353      }
354    }
355  
356    checkTLSOptions(allProvidedOptions);
357  
358    const unsupportedOptions = setDifference(
359      allProvidedKeys,
360      Array.from(Object.keys(OPTIONS)).map(s => s.toLowerCase())
361    );
362    if (unsupportedOptions.size !== 0) {
363      const optionWord = unsupportedOptions.size > 1 ? 'options' : 'option';
364      const isOrAre = unsupportedOptions.size > 1 ? 'are' : 'is';
365      throw new MongoParseError(
366        `${optionWord} ${Array.from(unsupportedOptions).join(', ')} ${isOrAre} not supported`
367      );
368    }
369  
370    // Option parsing and setting
371  
372    for (const [key, descriptor] of Object.entries(OPTIONS)) {
373      const values = allProvidedOptions.get(key);
374      if (!values || values.length === 0) {
375        if (DEFAULT_OPTIONS.has(key)) {
376          setOption(mongoOptions, key, descriptor, [DEFAULT_OPTIONS.get(key)]);
377        }
378      } else {
379        const { deprecated } = descriptor;
380        if (deprecated) {
381          const deprecatedMsg = typeof deprecated === 'string' ? `: ${deprecated}` : '';
382          emitWarning(`${key} is a deprecated option${deprecatedMsg}`);
383        }
384  
385        setOption(mongoOptions, key, descriptor, values);
386      }
387    }
388  
389    if (mongoOptions.credentials) {
390      const isGssapi = mongoOptions.credentials.mechanism === AuthMechanism.MONGODB_GSSAPI;
391      const isX509 = mongoOptions.credentials.mechanism === AuthMechanism.MONGODB_X509;
392      const isAws = mongoOptions.credentials.mechanism === AuthMechanism.MONGODB_AWS;
393      const isOidc = mongoOptions.credentials.mechanism === AuthMechanism.MONGODB_OIDC;
394      if (
395        (isGssapi || isX509) &&
396        allProvidedOptions.has('authSource') &&
397        mongoOptions.credentials.source !== '$external'
398      ) {
399        // If authSource was explicitly given and its incorrect, we error
400        throw new MongoParseError(
401          `authMechanism ${mongoOptions.credentials.mechanism} requires an authSource of '$external'`
402        );
403      }
404  
405      if (
406        !(isGssapi || isX509 || isAws || isOidc) &&
407        mongoOptions.dbName &&
408        !allProvidedOptions.has('authSource')
409      ) {
410        // inherit the dbName unless GSSAPI or X509, then silently ignore dbName
411        // and there was no specific authSource given
412        mongoOptions.credentials = MongoCredentials.merge(mongoOptions.credentials, {
413          source: mongoOptions.dbName
414        });
415      }
416  
417      if (isAws && mongoOptions.credentials.username && !mongoOptions.credentials.password) {
418        throw new MongoMissingCredentialsError(
419          `When using ${mongoOptions.credentials.mechanism} password must be set when a username is specified`
420        );
421      }
422  
423      mongoOptions.credentials.validate();
424  
425      // Check if the only auth related option provided was authSource, if so we can remove credentials
426      if (
427        mongoOptions.credentials.password === '' &&
428        mongoOptions.credentials.username === '' &&
429        mongoOptions.credentials.mechanism === AuthMechanism.MONGODB_DEFAULT &&
430        Object.keys(mongoOptions.credentials.mechanismProperties).length === 0
431      ) {
432        delete mongoOptions.credentials;
433      }
434    }
435  
436    if (!mongoOptions.dbName) {
437      // dbName default is applied here because of the credential validation above
438      mongoOptions.dbName = 'test';
439    }
440  
441    validateLoadBalancedOptions(hosts, mongoOptions, isSRV);
442  
443    if (mongoClient && mongoOptions.autoEncryption) {
444      Encrypter.checkForMongoCrypt();
445      mongoOptions.encrypter = new Encrypter(mongoClient, uri, options);
446      mongoOptions.autoEncrypter = mongoOptions.encrypter.autoEncrypter;
447    }
448  
449    // Potential SRV Overrides and SRV connection string validations
450  
451    mongoOptions.userSpecifiedAuthSource =
452      objectOptions.has('authSource') || urlOptions.has('authSource');
453    mongoOptions.userSpecifiedReplicaSet =
454      objectOptions.has('replicaSet') || urlOptions.has('replicaSet');
455  
456    if (isSRV) {
457      // SRV Record is resolved upon connecting
458      mongoOptions.srvHost = hosts[0];
459  
460      if (mongoOptions.directConnection) {
461        throw new MongoAPIError('SRV URI does not support directConnection');
462      }
463  
464      if (mongoOptions.srvMaxHosts > 0 && typeof mongoOptions.replicaSet === 'string') {
465        throw new MongoParseError('Cannot use srvMaxHosts option with replicaSet');
466      }
467  
468      // SRV turns on TLS by default, but users can override and turn it off
469      const noUserSpecifiedTLS = !objectOptions.has('tls') && !urlOptions.has('tls');
470      const noUserSpecifiedSSL = !objectOptions.has('ssl') && !urlOptions.has('ssl');
471      if (noUserSpecifiedTLS && noUserSpecifiedSSL) {
472        mongoOptions.tls = true;
473      }
474    } else {
475      const userSpecifiedSrvOptions =
476        urlOptions.has('srvMaxHosts') ||
477        objectOptions.has('srvMaxHosts') ||
478        urlOptions.has('srvServiceName') ||
479        objectOptions.has('srvServiceName');
480  
481      if (userSpecifiedSrvOptions) {
482        throw new MongoParseError(
483          'Cannot use srvMaxHosts or srvServiceName with a non-srv connection string'
484        );
485      }
486    }
487  
488    if (mongoOptions.directConnection && mongoOptions.hosts.length !== 1) {
489      throw new MongoParseError('directConnection option requires exactly one host');
490    }
491  
492    if (
493      !mongoOptions.proxyHost &&
494      (mongoOptions.proxyPort || mongoOptions.proxyUsername || mongoOptions.proxyPassword)
495    ) {
496      throw new MongoParseError('Must specify proxyHost if other proxy options are passed');
497    }
498  
499    if (
500      (mongoOptions.proxyUsername && !mongoOptions.proxyPassword) ||
501      (!mongoOptions.proxyUsername && mongoOptions.proxyPassword)
502    ) {
503      throw new MongoParseError('Can only specify both of proxy username/password or neither');
504    }
505  
506    const proxyOptions = ['proxyHost', 'proxyPort', 'proxyUsername', 'proxyPassword'].map(
507      key => urlOptions.get(key) ?? []
508    );
509  
510    if (proxyOptions.some(options => options.length > 1)) {
511      throw new MongoParseError(
512        'Proxy options cannot be specified multiple times in the connection string'
513      );
514    }
515  
516    const loggerFeatureFlag = Symbol.for('@@mdb.enableMongoLogger');
517    mongoOptions[loggerFeatureFlag] = mongoOptions[loggerFeatureFlag] ?? false;
518  
519    let loggerEnvOptions: MongoLoggerEnvOptions = {};
520    let loggerClientOptions: MongoLoggerMongoClientOptions = {};
521    if (mongoOptions[loggerFeatureFlag]) {
522      loggerEnvOptions = {
523        MONGODB_LOG_COMMAND: process.env.MONGODB_LOG_COMMAND,
524        MONGODB_LOG_TOPOLOGY: process.env.MONGODB_LOG_TOPOLOGY,
525        MONGODB_LOG_SERVER_SELECTION: process.env.MONGODB_LOG_SERVER_SELECTION,
526        MONGODB_LOG_CONNECTION: process.env.MONGODB_LOG_CONNECTION,
527        MONGODB_LOG_CLIENT: process.env.MONGODB_LOG_CLIENT,
528        MONGODB_LOG_ALL: process.env.MONGODB_LOG_ALL,
529        MONGODB_LOG_MAX_DOCUMENT_LENGTH: process.env.MONGODB_LOG_MAX_DOCUMENT_LENGTH,
530        MONGODB_LOG_PATH: process.env.MONGODB_LOG_PATH,
531        ...mongoOptions[Symbol.for('@@mdb.internalLoggerConfig')]
532      };
533      loggerClientOptions = {
534        mongodbLogPath: mongoOptions.mongodbLogPath,
535        mongodbLogComponentSeverities: mongoOptions.mongodbLogComponentSeverities,
536        mongodbLogMaxDocumentLength: mongoOptions.mongodbLogMaxDocumentLength
537      };
538    }
539    mongoOptions.mongoLoggerOptions = MongoLogger.resolveOptions(
540      loggerEnvOptions,
541      loggerClientOptions
542    );
543  
544    mongoOptions.metadata = makeClientMetadata(mongoOptions);
545  
546    return mongoOptions;
547  }

549  /**
550   * #### Throws if LB mode is true:
551   * - hosts contains more than one host
552   * - there is a replicaSet name set
553   * - directConnection is set
554   * - if srvMaxHosts is used when an srv connection string is passed in
555   *
556   * @throws MongoParseError
557   */
558  function validateLoadBalancedOptions(
559    hosts: HostAddress[] | string[],
560    mongoOptions: MongoOptions,
561    isSrv: boolean
562  ): void {
563    if (mongoOptions.loadBalanced) {
564      if (hosts.length > 1) {
565        throw new MongoParseError(LB_SINGLE_HOST_ERROR);
566      }
567      if (mongoOptions.replicaSet) {
568        throw new MongoParseError(LB_REPLICA_SET_ERROR);
569      }
570      if (mongoOptions.directConnection) {
571        throw new MongoParseError(LB_DIRECT_CONNECTION_ERROR);
572      }
573  
574      if (isSrv && mongoOptions.srvMaxHosts > 0) {
575        throw new MongoParseError('Cannot limit srv hosts with loadBalanced enabled');
576      }
577    }
578    return;
579  }

581  function setOption(
582    mongoOptions: any,
583    key: string,
584    descriptor: OptionDescriptor,
585    values: unknown[]
586  ) {
587    const { target, type, transform } = descriptor;
588    const name = target ?? key;
589  
590    switch (type) {
591      case 'boolean':
592        mongoOptions[name] = getBoolean(name, values[0]);
593        break;
594      case 'int':
595        mongoOptions[name] = getIntFromOptions(name, values[0]);
596        break;
597      case 'uint':
598        mongoOptions[name] = getUIntFromOptions(name, values[0]);
599        break;
600      case 'string':
601        if (values[0] == null) {
602          break;
603        }
604        mongoOptions[name] = String(values[0]);
605        break;
606      case 'record':
607        if (!isRecord(values[0])) {
608          throw new MongoParseError(`${name} must be an object`);
609        }
610        mongoOptions[name] = values[0];
611        break;
612      case 'any':
613        mongoOptions[name] = values[0];
614        break;
615      default: {
616        if (!transform) {
617          throw new MongoParseError('Descriptors missing a type must define a transform');
618        }
619        const transformValue = transform({ name, options: mongoOptions, values });
620        mongoOptions[name] = transformValue;
621        break;
622      }
623    }
624  }

626  interface OptionDescriptor {
627    target?: string;
628    type?: 'boolean' | 'int' | 'uint' | 'record' | 'string' | 'any';
629    default?: any;
630  
631    deprecated?: boolean | string;
632    /**
633     * @param name - the original option name
634     * @param options - the options so far for resolution
635     * @param values - the possible values in precedence order
636     */
637    transform?: (args: { name: string; options: MongoOptions; values: unknown[] }) => unknown;
638  }

640  export const OPTIONS = {
641    appName: {
642      type: 'string'
643    },
644    auth: {
645      target: 'credentials',
646      transform({ name, options, values: [value] }): MongoCredentials {
647        if (!isRecord(value, ['username', 'password'] as const)) {
648          throw new MongoParseError(
649            `${name} must be an object with 'username' and 'password' properties`
650          );
651        }
652        return MongoCredentials.merge(options.credentials, {
653          username: value.username,
654          password: value.password
655        });
656      }
657    },
658    authMechanism: {
659      target: 'credentials',
660      transform({ options, values: [value] }): MongoCredentials {
661        const mechanisms = Object.values(AuthMechanism);

672          source = '$external';
673        }
674  
675        let password = options.credentials?.password;
676        if (mechanism === AuthMechanism.MONGODB_X509 && password === '') {
677          password = undefined;
678        }
