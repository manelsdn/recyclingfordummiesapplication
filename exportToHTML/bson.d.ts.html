652  /**
653   * A class representing a 64-bit integer
654   * @public
655   * @category BSONType
656   * @remarks
657   * The internal representation of a long is the two given signed, 32-bit values.
658   * We use 32-bit pieces because these are the size of integers on which
659   * Javascript performs bit-operations.  For operations like addition and
660   * multiplication, we split each number into 16 bit pieces, which can easily be
661   * multiplied within Javascript's floating-point representation without overflow
662   * or change in sign.
663   * In the algorithms below, we frequently reduce the negative case to the
664   * positive case by negating the input(s) and then post-processing the result.
665   * Note that we must ALWAYS check specially whether those values are MIN_VALUE
666   * (-2^63) because -MIN_VALUE == MIN_VALUE (since 2^63 cannot be represented as
667   * a positive number, it overflows back into a negative).  Not handling this
668   * case would often result in infinite recursion.
669   * Common constant values ZERO, ONE, NEG_ONE, etc. are found as static properties on this class.
670   */
671  export declare class Long extends BSONValue {
672      get _bsontype(): 'Long';
673      /** An indicator used to reliably determine if an object is a Long or not. */
674      get __isLong__(): boolean;
675      /**
676       * The high 32 bits as a signed value.
677       */
678      high: number;
679      /**
680       * The low 32 bits as a signed value.
681       */
682      low: number;
683      /**
684       * Whether unsigned or not.
685       */
686      unsigned: boolean;
687      /**
688       * Constructs a 64 bit two's-complement integer, given its low and high 32 bit values as *signed* integers.
689       *  See the from* functions below for more convenient ways of constructing Longs.
690       *
691       * Acceptable signatures are:
692       * - Long(low, high, unsigned?)
693       * - Long(bigint, unsigned?)
694       * - Long(string, unsigned?)
695       *
696       * @param low - The low (signed) 32 bits of the long
697       * @param high - The high (signed) 32 bits of the long
698       * @param unsigned - Whether unsigned or not, defaults to signed
699       */
700      constructor(low?: number | bigint | string, high?: number | boolean, unsigned?: boolean);
701      static TWO_PWR_24: Long;
702      /** Maximum unsigned value. */
703      static MAX_UNSIGNED_VALUE: Long;
704      /** Signed zero */
705      static ZERO: Long;
706      /** Unsigned zero. */
707      static UZERO: Long;
708      /** Signed one. */
709      static ONE: Long;
710      /** Unsigned one. */
711      static UONE: Long;
712      /** Signed negative one. */
713      static NEG_ONE: Long;
714      /** Maximum signed value. */
715      static MAX_VALUE: Long;
716      /** Minimum signed value. */
717      static MIN_VALUE: Long;
718      /**
719       * Returns a Long representing the 64 bit integer that comes by concatenating the given low and high bits.
720       * Each is assumed to use 32 bits.
721       * @param lowBits - The low 32 bits
722       * @param highBits - The high 32 bits
723       * @param unsigned - Whether unsigned or not, defaults to signed
724       * @returns The corresponding Long value
725       */
726      static fromBits(lowBits: number, highBits: number, unsigned?: boolean): Long;
727      /**
728       * Returns a Long representing the given 32 bit integer value.
729       * @param value - The 32 bit integer in question
730       * @param unsigned - Whether unsigned or not, defaults to signed
731       * @returns The corresponding Long value
732       */
733      static fromInt(value: number, unsigned?: boolean): Long;
734      /**
735       * Returns a Long representing the given value, provided that it is a finite number. Otherwise, zero is returned.
736       * @param value - The number in question
737       * @param unsigned - Whether unsigned or not, defaults to signed
738       * @returns The corresponding Long value
739       */
740      static fromNumber(value: number, unsigned?: boolean): Long;
741      /**
742       * Returns a Long representing the given value, provided that it is a finite number. Otherwise, zero is returned.
743       * @param value - The number in question
744       * @param unsigned - Whether unsigned or not, defaults to signed
745       * @returns The corresponding Long value
746       */
747      static fromBigInt(value: bigint, unsigned?: boolean): Long;
748      /**
749       * Returns a Long representation of the given string, written using the specified radix.
