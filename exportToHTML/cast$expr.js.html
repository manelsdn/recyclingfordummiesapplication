70   module.exports = function cast$expr(val, schema, strictQuery) {
71     if (typeof val !== 'object' || val === null) {
72       throw new Error('`$expr` must be an object');
73     }
74   
75     return _castExpression(val, schema, strictQuery);
76   };

78   function _castExpression(val, schema, strictQuery) {
79     // Preserve the value if it represents a path or if it's null
80     if (isPath(val) || val === null) {
81       return val;
82     }
83   
84     if (val.$cond != null) {
85       if (Array.isArray(val.$cond)) {
86         val.$cond = val.$cond.map(expr => _castExpression(expr, schema, strictQuery));
87       } else {
88         val.$cond.if = _castExpression(val.$cond.if, schema, strictQuery);
89         val.$cond.then = _castExpression(val.$cond.then, schema, strictQuery);
90         val.$cond.else = _castExpression(val.$cond.else, schema, strictQuery);
91       }
92     } else if (val.$ifNull != null) {
93       val.$ifNull.map(v => _castExpression(v, schema, strictQuery));
94     } else if (val.$switch != null) {
95       val.branches.map(v => _castExpression(v, schema, strictQuery));
96       val.default = _castExpression(val.default, schema, strictQuery);
97     }
98   
99     const keys = Object.keys(val);
100    for (const key of keys) {
101      if (booleanComparison.has(key)) {
102        val[key] = val[key].map(v => _castExpression(v, schema, strictQuery));
103      } else if (comparisonOperator.has(key)) {
104        val[key] = castComparison(val[key], schema, strictQuery);
105      } else if (arithmeticOperatorArray.has(key)) {
106        val[key] = castArithmetic(val[key], schema, strictQuery);
107      } else if (arithmeticOperatorNumber.has(key)) {
108        val[key] = castNumberOperator(val[key], schema, strictQuery);
109      } else if (expressionOperator.has(key)) {
110        val[key] = _castExpression(val[key], schema, strictQuery);
111      }
112    }
113  
114    if (val.$in) {
115      val.$in = castIn(val.$in, schema, strictQuery);
116    }
117    if (val.$size) {
118      val.$size = castNumberOperator(val.$size, schema, strictQuery);
119    }
120    if (val.$round) {
121      const $round = val.$round;
122      if (!Array.isArray($round) || $round.length < 1 || $round.length > 2) {
123        throw new CastError('Array', $round, '$round');
124      }
125      val.$round = $round.map(v => castNumberOperator(v, schema, strictQuery));
126    }
127  
128    _omitUndefined(val);
129  
130    return val;
131  }

133  function _omitUndefined(val) {
134    const keys = Object.keys(val);
135    for (let i = 0, len = keys.length; i < len; ++i) {
136      (val[keys[i]] === void 0) && delete val[keys[i]];
137    }
138  }

140  // { $op: <number> }
141  function castNumberOperator(val) {
142    if (!isLiteral(val)) {
143      return val;
144    }
145  
146    try {
147      return castNumber(val);
148    } catch (err) {
149      throw new CastError('Number', val);
150    }
151  }

153  function castIn(val, schema, strictQuery) {
154    const path = val[1];
155    if (!isPath(path)) {
156      return val;
157    }
158    const search = val[0];
159  
160    const schematype = schema.path(path.slice(1));
161    if (schematype === null) {
162      if (strictQuery === false) {
163        return val;
164      } else if (strictQuery === 'throw') {
165        throw new StrictModeError('$in');
166      }
167  
168      return void 0;
169    }
170  
171    if (!schematype.$isMongooseArray) {
172      throw new Error('Path must be an array for $in');
173    }
174  
175    return [
176      schematype.$isMongooseDocumentArray ? schematype.$embeddedSchemaType.cast(search) : schematype.caster.cast(search),
177      path
178    ];
179  }

181  // { $op: [<number>, <number>] }
182  function castArithmetic(val) {
183    if (!Array.isArray(val)) {
184      if (!isLiteral(val)) {
185        return val;
186      }
187      try {
188        return castNumber(val);
189      } catch (err) {
190        throw new CastError('Number', val);
191      }
192    }
193  
194    return val.map(v => {
195      if (!isLiteral(v)) {
196        return v;
197      }
198      try {
199        return castNumber(v);
200      } catch (err) {
201        throw new CastError('Number', v);
202      }
203    });
204  }

206  // { $op: [expression, expression] }
207  function castComparison(val, schema, strictQuery) {
208    if (!Array.isArray(val) || val.length !== 2) {
209      throw new Error('Comparison operator must be an array of length 2');
210    }
211  
212    val[0] = _castExpression(val[0], schema, strictQuery);
213    const lhs = val[0];
214  
215    if (isLiteral(val[1])) {
216      let path = null;
217      let schematype = null;
218      let caster = null;
219      if (isPath(lhs)) {
220        path = lhs.slice(1);
221        schematype = schema.path(path);
222      } else if (typeof lhs === 'object' && lhs != null) {
223        for (const key of Object.keys(lhs)) {
224          if (dateOperators.has(key) && isPath(lhs[key])) {
225            path = lhs[key].slice(1) + '.' + key;
226            caster = castNumber;
227          } else if (arrayElementOperators.has(key) && isPath(lhs[key])) {
228            path = lhs[key].slice(1) + '.' + key;
229            schematype = schema.path(lhs[key].slice(1));
230            if (schematype != null) {
231              if (schematype.$isMongooseDocumentArray) {
232                schematype = schematype.$embeddedSchemaType;
233              } else if (schematype.$isMongooseArray) {
234                schematype = schematype.caster;
235              }
236            }
237          }
238        }
239      }
240  
241      const is$literal = typeof val[1] === 'object' && val[1] != null && val[1].$literal != null;
242      if (schematype != null) {
243        if (is$literal) {
244          val[1] = { $literal: schematype.cast(val[1].$literal) };
245        } else {
246          val[1] = schematype.cast(val[1]);
247        }
248      } else if (caster != null) {
249        if (is$literal) {
250          try {
251            val[1] = { $literal: caster(val[1].$literal) };
252          } catch (err) {
253            throw new CastError(caster.name.replace(/^cast/, ''), val[1], path + '.$literal');
254          }
255        } else {
256          try {
257            val[1] = caster(val[1]);
258          } catch (err) {
259            throw new CastError(caster.name.replace(/^cast/, ''), val[1], path);
260          }
261        }
262      } else if (path != null && strictQuery === true) {
263        return void 0;
264      } else if (path != null && strictQuery === 'throw') {
265        throw new StrictModeError(path);
266      }
267    } else {
268      val[1] = _castExpression(val[1]);
269    }
270  
271    return val;
272  }
273  
274  function isPath(val) {
275    return typeof val === 'string' && val[0] === '$';
276  }
277  
278  function isLiteral(val) {
279    if (typeof val === 'string' && val[0] === '$') {
280      return false;
281    }
282    if (typeof val === 'object' && val !== null && Object.keys(val).find(key => key[0] === '$')) {
283      // The `$literal` expression can make an object a literal
284      // https://www.mongodb.com/docs/manual/reference/operator/aggregation/literal/#mongodb-expression-exp.-literal
285      return val.$literal != null;
286    }
287    return true;
288  }
