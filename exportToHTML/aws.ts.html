1    import { getAwsCredentialProvider } from '../../deps';
2    import { type KMSProviders } from '.';
3    
4    /**
5     * @internal
6     */
7    export async function loadAWSCredentials(kmsProviders: KMSProviders): Promise<KMSProviders> {
8      const credentialProvider = getAwsCredentialProvider();
9    
10     if ('kModuleError' in credentialProvider) {
11       return kmsProviders;
12     }
13   
14     const { fromNodeProviderChain } = credentialProvider;
15     const provider = fromNodeProviderChain();
16     // The state machine is the only place calling this so it will
17     // catch if there is a rejection here.
18     const aws = await provider();
19     return { ...kmsProviders, aws };
20   }
