40   const kSession = Symbol('session');
41   
42   /**
43    * This class acts as a parent class for any operation and is responsible for setting this.options,
44    * as well as setting and getting a session.
45    * Additionally, this class implements `hasAspect`, which determines whether an operation has
46    * a specific aspect.
47    * @internal
48    */
49   export abstract class AbstractOperation<TResult = any> {
50     ns!: MongoDBNamespace;
51     readPreference: ReadPreference;
52     server!: Server;
53     bypassPinningCheck: boolean;
54     trySecondaryWrite: boolean;
55   
56     // BSON serialization options
57     bsonOptions?: BSONSerializeOptions;
58   
59     options: OperationOptions;
60   
61     [kSession]: ClientSession | undefined;
62   
63     constructor(options: OperationOptions = {}) {
64       this.readPreference = this.hasAspect(Aspect.WRITE_OPERATION)
65         ? ReadPreference.primary
66         : ReadPreference.fromOptions(options) ?? ReadPreference.primary;
67   
68       // Pull the BSON serialize options from the already-resolved options
69       this.bsonOptions = resolveBSONOptions(options);
70   
71       this[kSession] = options.session != null ? options.session : undefined;
72   
73       this.options = options;
74       this.bypassPinningCheck = !!options.bypassPinningCheck;
75       this.trySecondaryWrite = false;
76     }
77   
78     abstract execute(server: Server, session: ClientSession | undefined): Promise<TResult>;
79   
80     hasAspect(aspect: symbol): boolean {
81       const ctor = this.constructor as OperationConstructor;
82       if (ctor.aspects == null) {
83         return false;
84       }
85   
86       return ctor.aspects.has(aspect);
87     }
88   
89     get session(): ClientSession | undefined {
90       return this[kSession];
91     }
92   
93     clearSession() {
94       this[kSession] = undefined;
95     }
96   
97     get canRetryRead(): boolean {
98       return true;
99     }
100  
101    get canRetryWrite(): boolean {
102      return true;
103    }
104  }

106  export function defineAspects(
107    operation: OperationConstructor,
108    aspects: symbol | symbol[] | Set<symbol>
109  ): Set<symbol> {
110    if (!Array.isArray(aspects) && !(aspects instanceof Set)) {
111      aspects = [aspects];
112    }
113  
114    aspects = new Set(aspects);
115    Object.defineProperty(operation, 'aspects', {
116      value: aspects,
117      writable: false
118    });
119  
120    return aspects;
121  }
