1    import type { Document } from '../bson';
2    import { ServerType } from '../sdam/common';
3    import { parseServerType } from '../sdam/server_description';
4    import type { CompressorName } from './wire_protocol/compression';
5    
6    const RESPONSE_FIELDS = [
7      'minWireVersion',
8      'maxWireVersion',
9      'maxBsonObjectSize',
10     'maxMessageSizeBytes',
11     'maxWriteBatchSize',
12     'logicalSessionTimeoutMinutes'
13   ] as const;
14   
15   /** @public */
16   export interface StreamDescriptionOptions {
17     compressors?: CompressorName[];
18     logicalSessionTimeoutMinutes?: number;
19     loadBalanced: boolean;
20   }
21   
22   /** @public */
23   export class StreamDescription {
24     address: string;
25     type: string;
26     minWireVersion?: number;
27     maxWireVersion?: number;
28     maxBsonObjectSize: number;
29     maxMessageSizeBytes: number;
30     maxWriteBatchSize: number;
31     compressors: CompressorName[];
32     compressor?: CompressorName;
33     logicalSessionTimeoutMinutes?: number;
34     loadBalanced: boolean;
35   
36     __nodejs_mock_server__?: boolean;
37   
38     zlibCompressionLevel?: number;
39   
40     constructor(address: string, options?: StreamDescriptionOptions) {
41       this.address = address;
42       this.type = ServerType.Unknown;
43       this.minWireVersion = undefined;
44       this.maxWireVersion = undefined;
45       this.maxBsonObjectSize = 16777216;
46       this.maxMessageSizeBytes = 48000000;
47       this.maxWriteBatchSize = 100000;
48       this.logicalSessionTimeoutMinutes = options?.logicalSessionTimeoutMinutes;
49       this.loadBalanced = !!options?.loadBalanced;
50       this.compressors =
51         options && options.compressors && Array.isArray(options.compressors)
52           ? options.compressors
53           : [];
54     }
55   
56     receiveResponse(response: Document | null): void {
57       if (response == null) {
58         return;
59       }
60       this.type = parseServerType(response);
61       for (const field of RESPONSE_FIELDS) {
62         if (response[field] != null) {
63           this[field] = response[field];
64         }
65   
66         // testing case
67         if ('__nodejs_mock_server__' in response) {
68           this.__nodejs_mock_server__ = response['__nodejs_mock_server__'];
69         }
70       }
71   
72       if (response.compression) {
73         this.compressor = this.compressors.filter(c => response.compression?.includes(c))[0];
74       }
75     }
76   }
