54   /**
55    * The **ReadPreference** class is a class that represents a MongoDB ReadPreference and is
56    * used to construct connections.
57    * @public
58    *
59    * @see https://www.mongodb.com/docs/manual/core/read-preference/
60    */
61   export class ReadPreference {
62     mode: ReadPreferenceMode;
63     tags?: TagSet[];
64     hedge?: HedgeOptions;
65     maxStalenessSeconds?: number;
66     minWireVersion?: number;
67   
68     public static PRIMARY = ReadPreferenceMode.primary;
69     public static PRIMARY_PREFERRED = ReadPreferenceMode.primaryPreferred;
70     public static SECONDARY = ReadPreferenceMode.secondary;
71     public static SECONDARY_PREFERRED = ReadPreferenceMode.secondaryPreferred;
72     public static NEAREST = ReadPreferenceMode.nearest;
73   
74     public static primary = new ReadPreference(ReadPreferenceMode.primary);
75     public static primaryPreferred = new ReadPreference(ReadPreferenceMode.primaryPreferred);
76     public static secondary = new ReadPreference(ReadPreferenceMode.secondary);
77     public static secondaryPreferred = new ReadPreference(ReadPreferenceMode.secondaryPreferred);
78     public static nearest = new ReadPreference(ReadPreferenceMode.nearest);
79   
80     /**
81      * @param mode - A string describing the read preference mode (primary|primaryPreferred|secondary|secondaryPreferred|nearest)
82      * @param tags - A tag set used to target reads to members with the specified tag(s). tagSet is not available if using read preference mode primary.
83      * @param options - Additional read preference options
84      */
85     constructor(mode: ReadPreferenceMode, tags?: TagSet[], options?: ReadPreferenceOptions) {
86       if (!ReadPreference.isValid(mode)) {
87         throw new MongoInvalidArgumentError(`Invalid read preference mode ${JSON.stringify(mode)}`);
88       }
89       if (options == null && typeof tags === 'object' && !Array.isArray(tags)) {
90         options = tags;
91         tags = undefined;
92       } else if (tags && !Array.isArray(tags)) {
93         throw new MongoInvalidArgumentError('ReadPreference tags must be an array');
94       }
95   
96       this.mode = mode;
97       this.tags = tags;
98       this.hedge = options?.hedge;
99       this.maxStalenessSeconds = undefined;
100      this.minWireVersion = undefined;
101  
102      options = options ?? {};
103      if (options.maxStalenessSeconds != null) {
104        if (options.maxStalenessSeconds <= 0) {
105          throw new MongoInvalidArgumentError('maxStalenessSeconds must be a positive integer');
106        }
107  
108        this.maxStalenessSeconds = options.maxStalenessSeconds;
109  
110        // NOTE: The minimum required wire version is 5 for this read preference. If the existing
111        //       topology has a lower value then a MongoError will be thrown during server selection.
112        this.minWireVersion = 5;
113      }
114  
115      if (this.mode === ReadPreference.PRIMARY) {
116        if (this.tags && Array.isArray(this.tags) && this.tags.length > 0) {
117          throw new MongoInvalidArgumentError('Primary read preference cannot be combined with tags');
118        }
119  
120        if (this.maxStalenessSeconds) {
121          throw new MongoInvalidArgumentError(
122            'Primary read preference cannot be combined with maxStalenessSeconds'
123          );
124        }
125  
126        if (this.hedge) {
127          throw new MongoInvalidArgumentError(
128            'Primary read preference cannot be combined with hedge'
129          );
130        }
131      }
132    }
133  
134    // Support the deprecated `preference` property introduced in the porcelain layer
135    get preference(): ReadPreferenceMode {
136      return this.mode;
137    }
138  
139    static fromString(mode: string): ReadPreference {
140      return new ReadPreference(mode as ReadPreferenceMode);
141    }
142  
143    /**
144     * Construct a ReadPreference given an options object.
145     *
146     * @param options - The options object from which to extract the read preference.
147     */
148    static fromOptions(options?: ReadPreferenceFromOptions): ReadPreference | undefined {
149      if (!options) return;
150      const readPreference =
151        options.readPreference ?? options.session?.transaction.options.readPreference;
152      const readPreferenceTags = options.readPreferenceTags;
153  
154      if (readPreference == null) {
155        return;
156      }
157  
158      if (typeof readPreference === 'string') {
159        return new ReadPreference(readPreference, readPreferenceTags, {
160          maxStalenessSeconds: options.maxStalenessSeconds,
161          hedge: options.hedge
162        });
163      } else if (!(readPreference instanceof ReadPreference) && typeof readPreference === 'object') {
164        const mode = readPreference.mode || readPreference.preference;
165        if (mode && typeof mode === 'string') {
166          return new ReadPreference(mode, readPreference.tags ?? readPreferenceTags, {
167            maxStalenessSeconds: readPreference.maxStalenessSeconds,
168            hedge: options.hedge
169          });
170        }
171      }
172  
173      if (readPreferenceTags) {
174        readPreference.tags = readPreferenceTags;
175      }
176  
177      return readPreference as ReadPreference;
178    }
179  
180    /**
181     * Replaces options.readPreference with a ReadPreference instance
182     */
183    static translate(options: ReadPreferenceLikeOptions): ReadPreferenceLikeOptions {
184      if (options.readPreference == null) return options;
185      const r = options.readPreference;
186  
187      if (typeof r === 'string') {
188        options.readPreference = new ReadPreference(r);
189      } else if (r && !(r instanceof ReadPreference) && typeof r === 'object') {
190        const mode = r.mode || r.preference;
191        if (mode && typeof mode === 'string') {
192          options.readPreference = new ReadPreference(mode, r.tags, {
193            maxStalenessSeconds: r.maxStalenessSeconds
194          });
195        }
196      } else if (!(r instanceof ReadPreference)) {
197        throw new MongoInvalidArgumentError(`Invalid read preference: ${r}`);
198      }
199  
200      return options;
201    }

203    /**
204     * Validate if a mode is legal
205     *
206     * @param mode - The string representing the read preference mode.
207     */
208    static isValid(mode: string): boolean {
209      const VALID_MODES = new Set([
210        ReadPreference.PRIMARY,
211        ReadPreference.PRIMARY_PREFERRED,
212        ReadPreference.SECONDARY,
213        ReadPreference.SECONDARY_PREFERRED,
214        ReadPreference.NEAREST,
215        null
216      ]);
217  
218      return VALID_MODES.has(mode as ReadPreferenceMode);
219    }
220  
221    /**
222     * Validate if a mode is legal
223     *
224     * @param mode - The string representing the read preference mode.
225     */
226    isValid(mode?: string): boolean {
227      return ReadPreference.isValid(typeof mode === 'string' ? mode : this.mode);
228    }
229  
230    /**
231     * Indicates that this readPreference needs the "SecondaryOk" bit when sent over the wire
232     * @see https://www.mongodb.com/docs/manual/reference/mongodb-wire-protocol/#op-query
233     */
234    secondaryOk(): boolean {
235      const NEEDS_SECONDARYOK = new Set<string>([
236        ReadPreference.PRIMARY_PREFERRED,
237        ReadPreference.SECONDARY,
238        ReadPreference.SECONDARY_PREFERRED,
239        ReadPreference.NEAREST
240      ]);
241  
242      return NEEDS_SECONDARYOK.has(this.mode);
243    }
244  
245    /**
246     * Check if the two ReadPreferences are equivalent
247     *
248     * @param readPreference - The read preference with which to check equality
249     */
250    equals(readPreference: ReadPreference): boolean {
251      return readPreference.mode === this.mode;
252    }
253  
254    /** Return JSON representation */
255    toJSON(): Document {
256      const readPreference = { mode: this.mode } as Document;
257      if (Array.isArray(this.tags)) readPreference.tags = this.tags;
258      if (this.maxStalenessSeconds) readPreference.maxStalenessSeconds = this.maxStalenessSeconds;
259      if (this.hedge) readPreference.hedge = this.hedge;
260      return readPreference;
261    }
