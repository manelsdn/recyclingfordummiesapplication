77   /**
78    * A readable stream that enables you to read buffers from GridFS.
79    *
80    * Do not instantiate this class directly. Use `openDownloadStream()` instead.
81    * @public
82    */
83   export class GridFSBucketReadStream extends Readable {
84     /** @internal */
85     s: GridFSBucketReadStreamPrivate;
86   
87     /**
88      * Fires when the stream loaded the file document corresponding to the provided id.
89      * @event
90      */
91     static readonly FILE = 'file' as const;
92   
93     /**
94      * @param chunks - Handle for chunks collection
95      * @param files - Handle for files collection
96      * @param readPreference - The read preference to use
97      * @param filter - The filter to use to find the file document
98      * @internal
99      */
100    constructor(
101      chunks: Collection<GridFSChunk>,
102      files: Collection<GridFSFile>,
103      readPreference: ReadPreference | undefined,
104      filter: Document,
105      options?: GridFSBucketReadStreamOptions
106    ) {
107      super({ emitClose: true });
108      this.s = {
109        bytesToTrim: 0,
110        bytesToSkip: 0,
111        bytesRead: 0,
112        chunks,
113        expected: 0,
114        files,
115        filter,
116        init: false,
117        expectedEnd: 0,
118        options: {
119          start: 0,
120          end: 0,
121          ...options
122        },
123        readPreference
124      };
125    }
126  
127    /**
128     * Reads from the cursor and pushes to the stream.
129     * Private Impl, do not call directly
130     * @internal
131     */
132    override _read(): void {
133      if (this.destroyed) return;
134      waitForFile(this, () => doRead(this));
135    }
136  
137    /**
138     * Sets the 0-based offset in bytes to start streaming from. Throws
139     * an error if this stream has entered flowing mode
140     * (e.g. if you've already called `on('data')`)
141     *
142     * @param start - 0-based offset in bytes to start streaming from
143     */
144    start(start = 0): this {
145      throwIfInitialized(this);
146      this.s.options.start = start;
147      return this;
148    }
149  
150    /**
151     * Sets the 0-based offset in bytes to start streaming from. Throws
152     * an error if this stream has entered flowing mode
153     * (e.g. if you've already called `on('data')`)
154     *
155     * @param end - Offset in bytes to stop reading at
156     */
157    end(end = 0): this {
158      throwIfInitialized(this);
159      this.s.options.end = end;
160      return this;
161    }
162  
163    /**
164     * Marks this stream as aborted (will never push another `data` event)
165     * and kills the underlying cursor. Will emit the 'end' event, and then
166     * the 'close' event once the cursor is successfully killed.
167     */
168    async abort(): Promise<void> {
169      this.push(null);
170      this.destroy();
171      await this.s.cursor?.close();
172    }

175  function throwIfInitialized(stream: GridFSBucketReadStream): void {
176    if (stream.s.init) {
177      throw new MongoGridFSStreamError('Options cannot be changed after the stream is initialized');
178    }
179  }

181  function doRead(stream: GridFSBucketReadStream): void {
182    if (stream.destroyed) return;
183    if (!stream.s.cursor) return;
184    if (!stream.s.file) return;
185  
186    const handleReadResult = ({
187      error,
188      doc
189    }: { error: Error; doc: null } | { error: null; doc: any }) => {
190      if (stream.destroyed) {
191        return;
192      }
193      if (error) {
194        stream.destroy(error);
195        return;
196      }
197      if (!doc) {
198        stream.push(null);
199  
200        stream.s.cursor?.close().then(
201          () => null,
202          error => stream.destroy(error)
203        );
204        return;
205      }
206  
207      if (!stream.s.file) return;
208  
209      const bytesRemaining = stream.s.file.length - stream.s.bytesRead;
210      const expectedN = stream.s.expected++;
211      const expectedLength = Math.min(stream.s.file.chunkSize, bytesRemaining);
212      if (doc.n > expectedN) {
213        return stream.destroy(
214          new MongoGridFSChunkError(
215            `ChunkIsMissing: Got unexpected n: ${doc.n}, expected: ${expectedN}`
216          )
217        );
218      }
219  
220      if (doc.n < expectedN) {
221        return stream.destroy(
222          new MongoGridFSChunkError(`ExtraChunk: Got unexpected n: ${doc.n}, expected: ${expectedN}`)
223        );
224      }
225  
226      let buf = Buffer.isBuffer(doc.data) ? doc.data : doc.data.buffer;
227  
228      if (buf.byteLength !== expectedLength) {
229        if (bytesRemaining <= 0) {
230          return stream.destroy(
231            new MongoGridFSChunkError(
232              `ExtraChunk: Got unexpected n: ${doc.n}, expected file length ${stream.s.file.length} bytes but already read ${stream.s.bytesRead} bytes`
233            )
234          );
235        }
236  
237        return stream.destroy(
238          new MongoGridFSChunkError(
239            `ChunkIsWrongSize: Got unexpected length: ${buf.byteLength}, expected: ${expectedLength}`
240          )
241        );
242      }
243  
244      stream.s.bytesRead += buf.byteLength;
245  
246      if (buf.byteLength === 0) {
247        return stream.push(null);
248      }
249  
250      let sliceStart = null;
251      let sliceEnd = null;
252  
253      if (stream.s.bytesToSkip != null) {
254        sliceStart = stream.s.bytesToSkip;
255        stream.s.bytesToSkip = 0;
256      }
257  
258      const atEndOfStream = expectedN === stream.s.expectedEnd - 1;
259      const bytesLeftToRead = stream.s.options.end - stream.s.bytesToSkip;
260      if (atEndOfStream && stream.s.bytesToTrim != null) {
261        sliceEnd = stream.s.file.chunkSize - stream.s.bytesToTrim;
262      } else if (stream.s.options.end && bytesLeftToRead < doc.data.byteLength) {
263        sliceEnd = bytesLeftToRead;
264      }
265  
266      if (sliceStart != null || sliceEnd != null) {
267        buf = buf.slice(sliceStart || 0, sliceEnd || buf.byteLength);
268      }
269  
270      stream.push(buf);
271      return;
272    };
273  
274    stream.s.cursor.next().then(
275      doc => handleReadResult({ error: null, doc }),
276      error => handleReadResult({ error, doc: null })
277    );
278  }

280  function init(stream: GridFSBucketReadStream): void {
281    const findOneOptions: FindOptions = {};
282    if (stream.s.readPreference) {
283      findOneOptions.readPreference = stream.s.readPreference;
284    }
285    if (stream.s.options && stream.s.options.sort) {
286      findOneOptions.sort = stream.s.options.sort;
287    }
288    if (stream.s.options && stream.s.options.skip) {
289      findOneOptions.skip = stream.s.options.skip;
290    }
291  
292    const handleReadResult = ({
293      error,
294      doc
295    }: { error: Error; doc: null } | { error: null; doc: any }) => {
296      if (error) {
297        return stream.destroy(error);
298      }
299  
300      if (!doc) {
301        const identifier = stream.s.filter._id
302          ? stream.s.filter._id.toString()
303          : stream.s.filter.filename;
304        const errmsg = `FileNotFound: file ${identifier} was not found`;
305        // TODO(NODE-3483)
306        const err = new MongoRuntimeError(errmsg);
307        err.code = 'ENOENT'; // TODO: NODE-3338 set property as part of constructor
308        return stream.destroy(err);
309      }
310  
311      // If document is empty, kill the stream immediately and don't
312      // execute any reads
313      if (doc.length <= 0) {
314        stream.push(null);
315        return;
316      }
317  
318      if (stream.destroyed) {
319        // If user destroys the stream before we have a cursor, wait
320        // until the query is done to say we're 'closed' because we can't
321        // cancel a query.
322        stream.destroy();
323        return;
324      }
325  
326      try {
327        stream.s.bytesToSkip = handleStartOption(stream, doc, stream.s.options);
328      } catch (error) {
329        return stream.destroy(error);
330      }
331  
332      const filter: Document = { files_id: doc._id };
333  
334      // Currently (MongoDB 3.4.4) skip function does not support the index,
335      // it needs to retrieve all the documents first and then skip them. (CS-25811)
336      // As work around we use $gte on the "n" field.
337      if (stream.s.options && stream.s.options.start != null) {
338        const skip = Math.floor(stream.s.options.start / doc.chunkSize);
339        if (skip > 0) {
340          filter['n'] = { $gte: skip };
341        }
342      }
343      stream.s.cursor = stream.s.chunks.find(filter).sort({ n: 1 });
344  
345      if (stream.s.readPreference) {
346        stream.s.cursor.withReadPreference(stream.s.readPreference);
347      }
348  
349      stream.s.expectedEnd = Math.ceil(doc.length / doc.chunkSize);
350      stream.s.file = doc as GridFSFile;
351  
352      try {
353        stream.s.bytesToTrim = handleEndOption(stream, doc, stream.s.cursor, stream.s.options);
354      } catch (error) {
355        return stream.destroy(error);
356      }
357  
358      stream.emit(GridFSBucketReadStream.FILE, doc);
359      return;
360    };
361  
362    stream.s.files.findOne(stream.s.filter, findOneOptions).then(
363      doc => handleReadResult({ error: null, doc }),
364      error => handleReadResult({ error, doc: null })
365    );
366  }

368  function waitForFile(stream: GridFSBucketReadStream, callback: Callback): void {
369    if (stream.s.file) {
370      return callback();
371    }
372  
373    if (!stream.s.init) {
374      init(stream);
375      stream.s.init = true;
376    }
377  
378    stream.once('file', () => {
379      callback();
380    });
381  }

383  function handleStartOption(
384    stream: GridFSBucketReadStream,
385    doc: Document,
386    options: GridFSBucketReadStreamOptions
387  ): number {
388    if (options && options.start != null) {
389      if (options.start > doc.length) {
390        throw new MongoInvalidArgumentError(
391          `Stream start (${options.start}) must not be more than the length of the file (${doc.length})`
392        );
393      }
394      if (options.start < 0) {
395        throw new MongoInvalidArgumentError(`Stream start (${options.start}) must not be negative`);
396      }
397      if (options.end != null && options.end < options.start) {
398        throw new MongoInvalidArgumentError(
399          `Stream start (${options.start}) must not be greater than stream end (${options.end})`
400        );
401      }
402  
403      stream.s.bytesRead = Math.floor(options.start / doc.chunkSize) * doc.chunkSize;
404      stream.s.expected = Math.floor(options.start / doc.chunkSize);
405  
406      return options.start - stream.s.bytesRead;
407    }
408    throw new MongoInvalidArgumentError('Start option must be defined');
409  }
410  
411  function handleEndOption(
412    stream: GridFSBucketReadStream,
413    doc: Document,
414    cursor: FindCursor<GridFSChunk>,
415    options: GridFSBucketReadStreamOptions
416  ) {
417    if (options && options.end != null) {
418      if (options.end > doc.length) {
419        throw new MongoInvalidArgumentError(
420          `Stream end (${options.end}) must not be more than the length of the file (${doc.length})`
421        );
422      }
423      if (options.start == null || options.start < 0) {
424        throw new MongoInvalidArgumentError(`Stream end (${options.end}) must not be negative`);
425      }
426  
427      const start = options.start != null ? Math.floor(options.start / doc.chunkSize) : 0;
428  
429      cursor.limit(Math.ceil(options.end / doc.chunkSize) - start);
430  
431      stream.s.expectedEnd = Math.ceil(options.end / doc.chunkSize);
432  
433      return Math.ceil(options.end / doc.chunkSize) * doc.chunkSize - options.end;
434    }
435    throw new MongoInvalidArgumentError('End option must be defined');
436  }
