100  /**
101   * The **Db** class is a class that represents a MongoDB Database.
102   * @public
103   *
104   * @example
105   * ```ts
106   * import { MongoClient } from 'mongodb';
107   *
108   * interface Pet {
109   *   name: string;
110   *   kind: 'dog' | 'cat' | 'fish';
111   * }
112   *
113   * const client = new MongoClient('mongodb://localhost:27017');
114   * const db = client.db();
115   *
116   * // Create a collection that validates our union
117   * await db.createCollection<Pet>('pets', {
118   *   validator: { $expr: { $in: ['$kind', ['dog', 'cat', 'fish']] } }
119   * })
120   * ```
121   */
122  export class Db {

136    /**
137     * Creates a new Db instance.
138     *
139     * Db name cannot contain a dot, the server may apply more restrictions when an operation is run.
140     *
141     * @param client - The MongoClient for the database.
142     * @param databaseName - The name of the database this instance represents.
143     * @param options - Optional settings for Db construction.
144     */
145    constructor(client: MongoClient, databaseName: string, options?: DbOptions) {
146      options = options ?? {};
147  
148      // Filter the options
149      options = filterOptions(options, DB_OPTIONS_ALLOW_LIST);
150  
151      // Ensure there are no dots in database name
152      if (typeof databaseName === 'string' && databaseName.includes('.')) {
153        throw new MongoInvalidArgumentError(`Database names cannot contain the character '.'`);
154      }
155  
156      // Internal state of the db object
157      this.s = {
158        // Options
159        options,
160        // Unpack read preference
161        readPreference: ReadPreference.fromOptions(options),
162        // Merge bson options
163        bsonOptions: resolveBSONOptions(options, client),
164        // Set up the primary key factory or fallback to ObjectId
165        pkFactory: options?.pkFactory ?? DEFAULT_PK_FACTORY,
166        // ReadConcern
167        readConcern: ReadConcern.fromOptions(options),
168        writeConcern: WriteConcern.fromOptions(options),
169        // Namespace
170        namespace: new MongoDBNamespace(databaseName)
171      };
172  
173      this.client = client;
174    }
175  
176    get databaseName(): string {
177      return this.s.namespace.db;
178    }
179  
180    // Options
181    get options(): DbOptions | undefined {
182      return this.s.options;
183    }

185    /**
186     * Check if a secondary can be used (because the read preference is *not* set to primary)
187     */
188    get secondaryOk(): boolean {
189      return this.s.readPreference?.preference !== 'primary' || false;
190    }

240    /**
241     * Execute a command
242     *
243     * @remarks
244     * This command does not inherit options from the MongoClient.
245     *
246     * The driver will ensure the following fields are attached to the command sent to the server:
247     * - `lsid` - sourced from an implicit session or options.session
248     * - `$readPreference` - defaults to primary or can be configured by options.readPreference
249     * - `$db` - sourced from the name of this database
250     *
251     * If the client has a serverApi setting:
252     * - `apiVersion`
253     * - `apiStrict`
254     * - `apiDeprecationErrors`
255     *
256     * When in a transaction:
257     * - `readConcern` - sourced from readConcern set on the TransactionOptions
258     * - `writeConcern` - sourced from writeConcern set on the TransactionOptions
259     *
260     * Attaching any of the above fields to the command will have no effect as the driver will overwrite the value.
261     *
262     * @param command - The command to run
263     * @param options - Optional settings for the command
264     */
265    async command(command: Document, options?: RunCommandOptions): Promise<Document> {
266      // Intentionally, we do not inherit options from parent for this operation.
267      return executeOperation(
268        this.client,
269        new RunCommandOperation(this, command, {
270          ...resolveBSONOptions(options),
271          session: options?.session,
272          readPreference: options?.readPreference
273        })
274      );
275    }

489    /**
490     * Create a new Change Stream, watching for new changes (insertions, updates,
491     * replacements, deletions, and invalidations) in this database. Will ignore all
492     * changes to system collections.
493     *
494     * @remarks
495     * watch() accepts two generic arguments for distinct use cases:
496     * - The first is to provide the schema that may be defined for all the collections within this database
497     * - The second is to override the shape of the change stream document entirely, if it is not provided the type will default to ChangeStreamDocument of the first argument
498     *
499     * @param pipeline - An array of {@link https://www.mongodb.com/docs/manual/reference/operator/aggregation-pipeline/|aggregation pipeline stages} through which to pass change stream documents. This allows for filtering (using $match) and manipulating the change stream documents.
500     * @param options - Optional settings for the command
501     * @typeParam TSchema - Type of the data being detected by the change stream
502     * @typeParam TChange - Type of the whole change stream document emitted
503     */
504    watch<
505      TSchema extends Document = Document,
506      TChange extends Document = ChangeStreamDocument<TSchema>
507    >(pipeline: Document[] = [], options: ChangeStreamOptions = {}): ChangeStream<TSchema, TChange> {
508      // Allow optionally not specifying a pipeline
509      if (!Array.isArray(pipeline)) {
510        options = pipeline;
511        pipeline = [];
512      }
513  
514      return new ChangeStream<TSchema, TChange>(this, pipeline, resolveOptions(this, options));
515    }
