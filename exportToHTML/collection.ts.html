128  /**
129   * The **Collection** class is an internal class that embodies a MongoDB collection
130   * allowing for insert/find/update/delete and other command operation on that MongoDB collection.
131   *
132   * **COLLECTION Cannot directly be instantiated**
133   * @public
134   *
135   * @example
136   * ```ts
137   * import { MongoClient } from 'mongodb';
138   *
139   * interface Pet {
140   *   name: string;
141   *   kind: 'dog' | 'cat' | 'fish';
142   * }
143   *
144   * const client = new MongoClient('mongodb://localhost:27017');
145   * const pets = client.db().collection<Pet>('pets');
146   *
147   * const petCursor = pets.find();
148   *
149   * for await (const pet of petCursor) {
150   *   console.log(`${pet.name} is a ${pet.kind}!`);
151   * }
152   * ```
153   */

720     * **Note**: When migrating from {@link Collection#count| count} to {@link Collection#countDocuments| countDocuments}
721     * the following query operators must be replaced:

723     * | Operator | Replacement |

946    /**
947     * Create a new Change Stream, watching for new changes (insertions, updates, replacements, deletions, and invalidations) in this collection.
948     *
949     * @remarks
950     * watch() accepts two generic arguments for distinct use cases:
951     * - The first is to override the schema that may be defined for this specific collection
952     * - The second is to override the shape of the change stream document entirely, if it is not provided the type will default to ChangeStreamDocument of the first argument
953     * @example
954     * By just providing the first argument I can type the change to be `ChangeStreamDocument<{ _id: number }>`
955     * ```ts
956     * collection.watch<{ _id: number }>()
957     *   .on('change', change => console.log(change._id.toFixed(4)));
958     * ```
959     *
960     * @example
961     * Passing a second argument provides a way to reflect the type changes caused by an advanced pipeline.
962     * Here, we are using a pipeline to have MongoDB filter for insert changes only and add a comment.
963     * No need start from scratch on the ChangeStreamInsertDocument type!
964     * By using an intersection we can save time and ensure defaults remain the same type!
965     * ```ts
966     * collection
967     *   .watch<Schema, ChangeStreamInsertDocument<Schema> & { comment: string }>([
968     *     { $addFields: { comment: 'big changes' } },
969     *     { $match: { operationType: 'insert' } }
970     *   ])
971     *   .on('change', change => {
972     *     change.comment.startsWith('big');
973     *     change.operationType === 'insert';
974     *     // No need to narrow in code because the generics did that for us!
975     *     expectType<Schema>(change.fullDocument);
976     *   });
977     * ```
978     *
979     * @param pipeline - An array of {@link https://www.mongodb.com/docs/manual/reference/operator/aggregation-pipeline/|aggregation pipeline stages} through which to pass change stream documents. This allows for filtering (using $match) and manipulating the change stream documents.
980     * @param options - Optional settings for the command
981     * @typeParam TLocal - Type of the data being detected by the change stream
982     * @typeParam TChange - Type of the whole change stream document emitted
983     */
984    watch<TLocal extends Document = TSchema, TChange extends Document = ChangeStreamDocument<TLocal>>(
985      pipeline: Document[] = [],
986      options: ChangeStreamOptions = {}
987    ): ChangeStream<TLocal, TChange> {
988      // Allow optionally not specifying a pipeline
989      if (!Array.isArray(pipeline)) {
990        options = pipeline;
991        pipeline = [];
992      }
993  
994      return new ChangeStream<TLocal, TChange>(this, pipeline, resolveOptions(this, options));
995    }
