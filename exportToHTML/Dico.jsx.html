Dico.jsx 
1    import React, { useState } from "react";
2    import { useNavigate } from "react-router-dom";
3    import { Link } from "react-router-dom";
4    import explicationsData from "./explicationsData";
5    import poubelleImage from "./recycler.jpg";
6    import { IoIosArrowBack, IoIosArrowForward } from "react-icons/io";
7    
8    import CardCr from "../Carte/Cardcr";
9    import CardEr from "../Carte/Carder";
10   import Card1 from "../Carte/Cardr1";
11   import Card2 from "../Carte/Cardr2";
12   import Card3 from "../Carte/Cardr3";
13   import Cards from "../Carte/Cards";
14   import AppAsset from "../Carte/App";
15   import "../Carte/index.css";
16   
17   const Dico = ({ setExplication }) => {
18       // State pour gérer les catégories de mots
19       const [categories, setCategories] = useState({});
20       // State pour gérer le terme de recherche
21       const [searchTerm, setSearchTerm] = useState("");
22       // State pour suivre l'index de la carte actuelle
23       const [currentCardIndex, setCurrentCardIndex] = useState(0);
24       // State pour contrôler l'animation de retournement de carte
25       const [isFlipping, setIsFlipping] = useState(false);
26       // State pour suivre si la souris est sur une carte
27       const [isHovering, setIsHovering] = useState(false);
28   
29       // Utilisation du hook de navigation de react-router-dom
30       const navigate = useNavigate();
31   
32       // Fonction pour gérer le changement de catégorie
33       const handleCategoryChange = (category) => {
34           setCategories({
35               ...categories,
36               [category]: !categories[category],
37           });
38       };
39   
40       // Fonction pour gérer le changement de terme de recherche
41       const handleInputChange = (e) => {
42           setSearchTerm(e.target.value);
43       };
44   
45       // Fonction pour effectuer la recherche et naviguer vers la page d'explications
46       const handleSearch = () => {
47           // Sélection des mots selon les catégories sélectionnées
48           const motsSelectionnesParCategories = Object.keys(categories)
49               .filter(category => categories[category])
50               .flatMap(category => explicationsData.filter(data => data.categorie === category).map(data => data.mot));
51           // Filtrage des mots selon le terme de recherche
52           const motsRecherche = explicationsData
53               .filter(data => data.mot.toLowerCase().includes(searchTerm.toLowerCase()))
54               .map(data => data.mot);
55           // Fusion des mots sélectionnés et des mots de recherche
56           const motsTotal = [...new Set([...motsSelectionnesParCategories, ...motsRecherche])];
57           // Mise à jour de l'état d'explication et navigation vers la page d'explications
58           setExplication(motsTotal);
59           navigate("/explications");
60       };
61   
62       // Nombre total de cartes
63       const totalCards = 7;
64   
65       // Fonction pour gérer le clic sur une carte
66       const handleClickCard = (direction) => {
67           if (!isFlipping) {
68               setIsFlipping(true);
69               setTimeout(() => {
70                   if (direction === "next" && currentCardIndex < totalCards - 1) {
71                       setCurrentCardIndex((prevIndex) => prevIndex + 1);
72                   } else if (direction === "prev" && currentCardIndex > 0) {
73                       setCurrentCardIndex((prevIndex) => prevIndex - 1);
74                   }
75                   setIsFlipping(false);
76               }, 500);
77           }
78       };
79   
80       return (
81           <div className="container">
82               <h2>Dictionnaire du recyclage</h2>
83               <div className="flex-container">
84                   <div className="image-container">
85                       <img src={poubelleImage} alt="Poubelle" style={{maxWidth: "200px", maxHeight: "200px"}}/>
86                   </div>
87                   <div className="categories">
88                       {/* Champ de recherche */}
89                       <input
90                           type="text"
91                           placeholder="Rechercher un mot..."
92                           style={{width: "calc(190% - 200px)"}}
93                           onChange={handleInputChange}
94                       />
95                       <br/><br/>
96                       {/* Cases à cocher pour les catégories */}
97                       <div className="checkboxes">
98                           {Object.keys(categories).map((category, index) => (
99                               <div key={index}>
100                                  <input
101                                      type="checkbox"
102                                      id={category}
103                                      onChange={() => handleCategoryChange(category)}
104                                      checked={categories[category]}
105                                      aria-label={category}
106                                  />
107                                  <label htmlFor={category}>{category}</label>
108                              </div>
109                          ))}
110                      </div>
111                      {/* Bouton de recherche */}
112                      <button className="button-size" onClick={handleSearch}>Je recycle</button>
113                      <br/>
114                      <br/>
115                      {/* Bouton pour accéder au quizz */}
116                      <Link to="/Quiz">
117                          <button className="button-size">Quizz</button>
118                      </Link>
119                      {/* Boutons de navigation des cartes */}
120                      <div className="card-buttons">
121                          <button className="arrow-button" onClick={() => handleClickCard("prev")}><IoIosArrowBack/>
122                          </button>
123                          <button className="arrow-button" onClick={() => handleClickCard("next")}><IoIosArrowForward/>
124                          </button>
125                      </div>
126                  </div>
127              </div>
128  
129              {/* Affichage des cartes */}
130              <div className={`card-display ${isFlipping ? "flipping" : ""}`}>
131                  {/* Carte 1 */}
132                  <div
133                      onMouseEnter={() => setIsHovering(true)}
134                      onMouseLeave={() => setIsHovering(false)}
135                      onClick={() => handleClickCard("next")}
136                      style={{backgroundColor: isHovering && currentCardIndex === 0 ? "#afe4b2" : ""}}
137                  >
138                      {currentCardIndex === 0 && <CardCr/>}
139                  </div>
140                  {/* Carte 2 */}
141                  <div
142                      onMouseEnter={() => setIsHovering(true)}
143                      onMouseLeave={() => setIsHovering(false)}
144                      onClick={() => handleClickCard("next")}
145                      style={{backgroundColor: isHovering && currentCardIndex === 1 ? "#afe4b2" : ""}}
146                  >
147                      {currentCardIndex === 1 && <CardEr/>}
148                  </div>
149                  {/* Carte 3 */}
150                  <div
151                      onMouseEnter={() => setIsHovering(true)}
152                      onMouseLeave={() => setIsHovering(false)}
153                      onClick={() => handleClickCard("next")}
154                      style={{backgroundColor: isHovering && currentCardIndex === 2 ? "#afe4b2" : ""}}
155                  >
156                      {currentCardIndex === 2 && <Card1/>}
157                  </div>
158                  {/* Carte 4 */}
159                  <div
160                      onMouseEnter={() => setIsHovering(true)}
161                      onMouseLeave={() => setIsHovering(false)}
162                      onClick={() => handleClickCard("next")}
163                      style={{backgroundColor: isHovering && currentCardIndex === 3 ? "#afe4b2" : ""}}
164                  >
165                      {currentCardIndex === 3 && <Card2/>}
166                  </div>
167                  {/* Carte 5 */}
168                  <div
169                      onMouseEnter={() => setIsHovering(true)}
170                      onMouseLeave={() => setIsHovering(false)}
171                      onClick={() => handleClickCard("next")}
172                      style={{backgroundColor: isHovering && currentCardIndex === 4 ? "#afe4b2" : ""}}
173                  >
174                      {currentCardIndex === 4 && <Card3/>}
175                  </div>
176                  {/* Carte 6 */}
177                  <div
178                      onMouseEnter={() => setIsHovering(true)}
179                      onMouseLeave={() => setIsHovering(false)}
180                      onClick={() => handleClickCard("next")}
181                      style={{backgroundColor: isHovering && currentCardIndex === 5 ? "#afe4b2" : ""}}
182                  >
183                      {currentCardIndex === 5 && <Cards/>}
184                  </div>
185                  {/* Carte 7 */}
186                  <div
187                      onMouseEnter={() => setIsHovering(true)}
188                      onMouseLeave={() => setIsHovering(false)}
189                      onClick={() => handleClickCard("next")}
190                      style={{backgroundColor: isHovering && currentCardIndex === 6 ? "#afe4b2" : ""}}
191                  >
192                      {currentCardIndex === 6 && <AppAsset/>}
193                  </div>
194              </div>
195          </div>
196      );
197  };
198  
199  export default Dico;
