73     /**
74      * Create a ServerDescription
75      * @internal
76      *
77      * @param address - The address of the server
78      * @param hello - An optional hello response for this server
79      */
80     constructor(
81       address: HostAddress | string,
82       hello?: Document,
83       options: ServerDescriptionOptions = {}
84     ) {
85       if (address == null || address === '') {
86         throw new MongoRuntimeError('ServerDescription must be provided with a non-empty address');
87       }
88   
89       this.address =
90         typeof address === 'string'
91           ? HostAddress.fromString(address).toString() // Use HostAddress to normalize
92           : address.toString();
93       this.type = parseServerType(hello, options);
94       this.hosts = hello?.hosts?.map((host: string) => host.toLowerCase()) ?? [];
95       this.passives = hello?.passives?.map((host: string) => host.toLowerCase()) ?? [];
96       this.arbiters = hello?.arbiters?.map((host: string) => host.toLowerCase()) ?? [];
97       this.tags = hello?.tags ?? {};
98       this.minWireVersion = hello?.minWireVersion ?? 0;
99       this.maxWireVersion = hello?.maxWireVersion ?? 0;
100      this.roundTripTime = options?.roundTripTime ?? -1;
101      this.lastUpdateTime = now();
102      this.lastWriteDate = hello?.lastWrite?.lastWriteDate ?? 0;
103      this.error = options.error ?? null;
104      // TODO(NODE-2674): Preserve int64 sent from MongoDB
105      this.topologyVersion = this.error?.topologyVersion ?? hello?.topologyVersion ?? null;
106      this.setName = hello?.setName ?? null;
107      this.setVersion = hello?.setVersion ?? null;
108      this.electionId = hello?.electionId ?? null;
109      this.logicalSessionTimeoutMinutes = hello?.logicalSessionTimeoutMinutes ?? null;
110      this.primary = hello?.primary ?? null;
111      this.me = hello?.me?.toLowerCase() ?? null;
112      this.$clusterTime = hello?.$clusterTime ?? null;
113    }

116      return HostAddress.fromString(this.address);

118

120      return this.hosts.concat(this.arbiters).concat(this.passives);

133    /** Is this server available for writes */
134    get isWritable(): boolean {
135      return WRITABLE_SERVER_TYPES.has(this.type);

138    get host(): string {
139      const chopLength = `:${this.port}`.length;
140      return this.address.slice(0, -chopLength);
141    }
142  
143    get port(): number {
144      const port = this.address.split(':').pop();
145      return port ? Number.parseInt(port, 10) : 27017;

148    /**
149     * Determines if another `ServerDescription` is equal to this one per the rules defined
150     * in the {@link https://github.com/mongodb/specifications/blob/master/source/server-discovery-and-monitoring/server-discovery-and-monitoring.rst#serverdescription|SDAM spec}
151     */
152    equals(other?: ServerDescription | null): boolean {
153      // Despite using the comparator that would determine a nullish topologyVersion as greater than
154      // for equality we should only always perform direct equality comparison
155      const topologyVersionsEqual =
156        this.topologyVersion === other?.topologyVersion ||
157        compareTopologyVersion(this.topologyVersion, other?.topologyVersion) === 0;
158  
159      const electionIdsEqual =
160        this.electionId != null && other?.electionId != null
161          ? compareObjectId(this.electionId, other.electionId) === 0
162          : this.electionId === other?.electionId;
163  
164      return (
165        other != null &&
166        errorStrictEqual(this.error, other.error) &&
167        this.type === other.type &&
168        this.minWireVersion === other.minWireVersion &&
169        arrayStrictEqual(this.hosts, other.hosts) &&
170        tagsStrictEqual(this.tags, other.tags) &&
171        this.setName === other.setName &&
172        this.setVersion === other.setVersion &&
173        electionIdsEqual &&
174        this.primary === other.primary &&
175        this.logicalSessionTimeoutMinutes === other.logicalSessionTimeoutMinutes &&
176        topologyVersionsEqual
177      );
178    }
179  }

181  // Parses a `hello` message and determines the server type
182  export function parseServerType(hello?: Document, options?: ServerDescriptionOptions): ServerType {
183    if (options?.loadBalanced) {
184      return ServerType.LoadBalancer;
185    }
186  
187    if (!hello || !hello.ok) {
188      return ServerType.Unknown;
189    }
190  
191    if (hello.isreplicaset) {
192      return ServerType.RSGhost;
193    }
194  
195    if (hello.msg && hello.msg === 'isdbgrid') {
196      return ServerType.Mongos;
197    }
198  
199    if (hello.setName) {
200      if (hello.hidden) {
201        return ServerType.RSOther;
202      } else if (hello.isWritablePrimary) {
203        return ServerType.RSPrimary;
204      } else if (hello.secondary) {
205        return ServerType.RSSecondary;
206      } else if (hello.arbiterOnly) {
207        return ServerType.RSArbiter;
208      } else {
209        return ServerType.RSOther;
210      }
211    }
212  
213    return ServerType.Standalone;
214  }

216  function tagsStrictEqual(tags: TagSet, tags2: TagSet): boolean {
217    const tagsKeys = Object.keys(tags);
218    const tags2Keys = Object.keys(tags2);
219  
220    return (
221      tagsKeys.length === tags2Keys.length &&
222      tagsKeys.every((key: string) => tags2[key] === tags[key])
223    );
224  }

226  /**
227   * Compares two topology versions.
228   *
229   * 1. If the response topologyVersion is unset or the ServerDescription's
230   *    topologyVersion is null, the client MUST assume the response is more recent.
231   * 1. If the response's topologyVersion.processId is not equal to the
232   *    ServerDescription's, the client MUST assume the response is more recent.
233   * 1. If the response's topologyVersion.processId is equal to the
234   *    ServerDescription's, the client MUST use the counter field to determine
235   *    which topologyVersion is more recent.
236   *
237   * ```ts
238   * currentTv <   newTv === -1
239   * currentTv === newTv === 0
240   * currentTv >   newTv === 1
241   * ```
242   */
243  export function compareTopologyVersion(
244    currentTv?: TopologyVersion | null,
245    newTv?: TopologyVersion | null
246  ): 0 | -1 | 1 {
247    if (currentTv == null || newTv == null) {
248      return -1;
249    }
250  
251    if (!currentTv.processId.equals(newTv.processId)) {
252      return -1;
253    }
254  
255    // TODO(NODE-2674): Preserve int64 sent from MongoDB
256    const currentCounter = Long.isLong(currentTv.counter)
257      ? currentTv.counter
258      : Long.fromNumber(currentTv.counter);
259    const newCounter = Long.isLong(newTv.counter) ? newTv.counter : Long.fromNumber(newTv.counter);
260  
261    return currentCounter.compare(newCounter);
262  }
