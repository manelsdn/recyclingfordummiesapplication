63     type QuerySelector<T> = {
64       // Comparison
65       $eq?: T;
66       $gt?: T;
67       $gte?: T;
68       $in?: [T] extends AnyArray<any> ? Unpacked<T>[] : T[];
69       $lt?: T;
70       $lte?: T;
71       $ne?: T;
72       $nin?: [T] extends AnyArray<any> ? Unpacked<T>[] : T[];
73       // Logical
74       $not?: T extends string ? QuerySelector<T> | RegExp : QuerySelector<T>;
75       // Element
76       /**
77        * When `true`, `$exists` matches the documents that contain the field,
78        * including documents where the field value is null.
79        */
80       $exists?: boolean;
81       $type?: string | number;
82       // Evaluation
83       $expr?: any;
84       $jsonSchema?: any;
85       $mod?: T extends number ? [number, number] : never;
86       $regex?: T extends string ? RegExp | string : never;
87       $options?: T extends string ? string : never;
88       // Geospatial
89       // TODO: define better types for geo queries
90       $geoIntersects?: { $geometry: object };
91       $geoWithin?: object;
92       $near?: object;
93       $nearSphere?: object;
94       $maxDistance?: number;
95       // Array
96       // TODO: define better types for $all and $elemMatch
97       $all?: T extends AnyArray<any> ? any[] : never;
98       $elemMatch?: T extends AnyArray<any> ? object : never;
99       $size?: T extends AnyArray<any> ? number : never;
100      // Bitwise
101      $bitsAllClear?: number | mongodb.Binary | number[];
102      $bitsAllSet?: number | mongodb.Binary | number[];
103      $bitsAnyClear?: number | mongodb.Binary | number[];
104      $bitsAnySet?: number | mongodb.Binary | number[];
105    };
