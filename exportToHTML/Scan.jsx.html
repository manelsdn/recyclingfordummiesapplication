1    import React, { useState } from "react";
2    import * as tf from "@tensorflow/tfjs"; // Importation de TensorFlow.js
3    import * as mobilenet from "@tensorflow-models/mobilenet"; // Importation du modèle MobileNet
4    
5    function MobileNetComponent() {
6      // State pour l'URL de l'image, les prédictions et l'état de chargement
7      const [imageUrl, setImageUrl] = useState("");
8      const [predictions, setPredictions] = useState([]);
9      const [loading, setLoading] = useState(false);
10   
11     // Fonction pour charger le modèle MobileNet
12     async function loadModel() {
13       setLoading(true);
14       const model = await mobilenet.load(); // Chargement du modèle MobileNet
15       setLoading(false);
16       return model;
17     }
18   
19     // Fonction pour classifier une image avec le modèle chargé
20     async function classifyImage(model, imageElement) {
21       if (model) {
22         setLoading(true);
23         const predictions = await model.classify(imageElement); // Classification de l'image
24         setLoading(false);
25         setPredictions(predictions); // Mise à jour des prédictions
26       }
27     }
28   
29     // Fonction pour gérer le téléchargement d'une image
30     async function handleImageUpload(event) {
31       const file = event.target.files[0];
32       const reader = new FileReader();
33   
34       reader.onload = async (e) => {
35         setImageUrl(e.target.result); // Mise à jour de l'URL de l'image
36         const imageElement = document.createElement("img");
37         imageElement.src = e.target.result;
38         const model = await loadModel(); // Chargement du modèle
39         await classifyImage(model, imageElement); // Classification de l'image
40       };
41   
42       reader.readAsDataURL(file);
43     }
44   
45     return (
46         <div className="container">
47           {/* Titre de la section */}
48           <h2>Détection d'objets </h2>
49           {/* Champ d'entrée pour télécharger une image */}
50           <input type="file" accept="image/*" onChange={handleImageUpload} />
51           {/* Affichage de l'état de chargement */}
52           {loading && <p>téléchargement du modèle et détection d'objets...</p>}
53           {/* Affichage de l'image téléchargée et des prédictions */}
54           {imageUrl && (
55               <div>
56                 <img src={imageUrl} alt="Uploaded" style={{ maxWidth: "300px" }} />
57                 {/* Titre des prédictions */}
58                 <h2>Prédictions:</h2>
59                 {/* Liste des prédictions */}
60                 <ul>
61                   {predictions.map((prediction, index) => (
62                       <li key={index}>
63                         {/* Affichage de la classe de l'objet et de sa probabilité */}
64                         {prediction.className} - Probability:{" "}
65                         {Math.round(prediction.probability * 100)}%
66                       </li>
67                   ))}
68                 </ul>
69               </div>
70           )}
71         </div>
72     );
73   }
74   
75   export default MobileNetComponent;
