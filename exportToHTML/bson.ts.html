26   /**
27    * BSON Serialization options.
28    * @public
29    */
30   export interface BSONSerializeOptions
31     extends Omit<SerializeOptions, 'index'>,
32       Omit<
33         DeserializeOptions,
34         | 'evalFunctions'
35         | 'cacheFunctions'
36         | 'cacheFunctionsCrc32'
37         | 'allowObjectSmallerThanBufferSize'
38         | 'index'
39         | 'validation'
40       > {
41     /**
42      * Enabling the raw option will return a [Node.js Buffer](https://nodejs.org/api/buffer.html)
43      * which is allocated using [allocUnsafe API](https://nodejs.org/api/buffer.html#static-method-bufferallocunsafesize).
44      * See this section from the [Node.js Docs here](https://nodejs.org/api/buffer.html#what-makes-bufferallocunsafe-and-bufferallocunsafeslow-unsafe)
45      * for more detail about what "unsafe" refers to in this context.
46      * If you need to maintain your own editable clone of the bytes returned for an extended life time of the process, it is recommended you allocate
47      * your own buffer and clone the contents:
48      *
49      * @example
50      * ```ts
51      * const raw = await collection.findOne({}, { raw: true });
52      * const myBuffer = Buffer.alloc(raw.byteLength);
53      * myBuffer.set(raw, 0);
54      * // Only save and use `myBuffer` beyond this point
55      * ```
56      *
57      * @remarks
58      * Please note there is a known limitation where this option cannot be used at the MongoClient level (see [NODE-3946](https://jira.mongodb.org/browse/NODE-3946)).
59      * It does correctly work at `Db`, `Collection`, and per operation the same as other BSON options work.
60      */
61     raw?: boolean;
62   
63     /** Enable utf8 validation when deserializing BSON documents.  Defaults to true. */
64     enableUtf8Validation?: boolean;
65   }

94   /**
95    * Merge the given BSONSerializeOptions, preferring options over the parent's options, and
96    * substituting defaults for values not set.
97    *
98    * @internal
99    */
100  export function resolveBSONOptions(
101    options?: BSONSerializeOptions,
102    parent?: { bsonOptions?: BSONSerializeOptions }
103  ): BSONSerializeOptions {
104    const parentOptions = parent?.bsonOptions;
105    return {
106      raw: options?.raw ?? parentOptions?.raw ?? false,
107      useBigInt64: options?.useBigInt64 ?? parentOptions?.useBigInt64 ?? false,
108      promoteLongs: options?.promoteLongs ?? parentOptions?.promoteLongs ?? true,
109      promoteValues: options?.promoteValues ?? parentOptions?.promoteValues ?? true,
110      promoteBuffers: options?.promoteBuffers ?? parentOptions?.promoteBuffers ?? false,
111      ignoreUndefined: options?.ignoreUndefined ?? parentOptions?.ignoreUndefined ?? false,
112      bsonRegExp: options?.bsonRegExp ?? parentOptions?.bsonRegExp ?? false,
113      serializeFunctions: options?.serializeFunctions ?? parentOptions?.serializeFunctions ?? false,
114      fieldsAsRaw: options?.fieldsAsRaw ?? parentOptions?.fieldsAsRaw ?? {},
115      enableUtf8Validation:
116        options?.enableUtf8Validation ?? parentOptions?.enableUtf8Validation ?? true
117    };
118  }
