89     constructor(inputId?: string | number | ObjectId | ObjectIdLike | Uint8Array) {
90       super();
91       // workingId is set based on type of input and whether valid id exists for the input
92       let workingId;
93       if (typeof inputId === 'object' && inputId && 'id' in inputId) {
94         if (typeof inputId.id !== 'string' && !ArrayBuffer.isView(inputId.id)) {
95           throw new BSONError('Argument passed in must have an id that is of type string or Buffer');
96         }
97         if ('toHexString' in inputId && typeof inputId.toHexString === 'function') {
98           workingId = ByteUtils.fromHex(inputId.toHexString());
99         } else {
100          workingId = inputId.id;
101        }
102      } else {
103        workingId = inputId;
104      }
105  
106      // The following cases use workingId to construct an ObjectId
107      if (workingId == null || typeof workingId === 'number') {
108        // The most common use case (blank id, new objectId instance)
109        // Generate a new id
110        this.buffer = ObjectId.generate(typeof workingId === 'number' ? workingId : undefined);
111      } else if (ArrayBuffer.isView(workingId) && workingId.byteLength === 12) {
112        // If intstanceof matches we can escape calling ensure buffer in Node.js environments
113        this.buffer = ByteUtils.toLocalBufferType(workingId);
114      } else if (typeof workingId === 'string') {
115        if (workingId.length === 24 && checkForHexRegExp.test(workingId)) {
116          this.buffer = ByteUtils.fromHex(workingId);
117        } else {
118          throw new BSONError(
119            'input must be a 24 character hex string, 12 byte Uint8Array, or an integer'
120          );
121        }
122      } else {
123        throw new BSONError('Argument passed in does not match the accepted types');
124      }
125      // If we are caching the hex string
126      if (ObjectId.cacheHexString) {
127        this.__id = ByteUtils.toHex(this.id);
128      }
129    }

146    /** Returns the ObjectId id as a 24 lowercase character hex string representation */
147    toHexString(): string {
148      if (ObjectId.cacheHexString && this.__id) {
149        return this.__id;
150      }

154      if (ObjectId.cacheHexString && !this.__id) {
155        this.__id = hexString;

169    /**
170     * Generate a 12 byte id buffer used in ObjectId's
171     *
172     * @param time - pass in a second based timestamp.
173     */
174    static generate(time?: number): Uint8Array {
175      if ('number' !== typeof time) {
176        time = Math.floor(Date.now() / 1000);
177      }
178  
179      const inc = ObjectId.getInc();
180      const buffer = ByteUtils.allocateUnsafe(12);
181  
182      // 4-byte timestamp
183      NumberUtils.setInt32BE(buffer, 0, time);
184  
185      // set PROCESS_UNIQUE if yet not initialized
186      if (PROCESS_UNIQUE === null) {
187        PROCESS_UNIQUE = ByteUtils.randomBytes(5);
188      }
189  
190      // 5-byte process unique
191      buffer[4] = PROCESS_UNIQUE[0];
192      buffer[5] = PROCESS_UNIQUE[1];
193      buffer[6] = PROCESS_UNIQUE[2];
194      buffer[7] = PROCESS_UNIQUE[3];
195      buffer[8] = PROCESS_UNIQUE[4];
196  
197      // 3-byte counter
198      buffer[11] = inc & 0xff;
199      buffer[10] = (inc >> 8) & 0xff;
200      buffer[9] = (inc >> 16) & 0xff;
201  
202      return buffer;
203    }

231    /**
232     * Compares the equality of this ObjectId with `otherID`.
233     *
234     * @param otherId - ObjectId instance to compare against.
235     */
236    equals(otherId: string | ObjectId | ObjectIdLike | undefined | null): boolean {
237      if (otherId === undefined || otherId === null) {
238        return false;
239      }
240  
241      if (ObjectId.is(otherId)) {
242        return (
243          this.buffer[11] === otherId.buffer[11] && ByteUtils.equals(this.buffer, otherId.buffer)
244        );
245      }
246  
247      if (typeof otherId === 'string') {
248        return otherId.toLowerCase() === this.toHexString();
249      }
250  
251      if (typeof otherId === 'object' && typeof otherId.toHexString === 'function') {
252        const otherIdString = otherId.toHexString();
253        const thisIdString = this.toHexString();
254        return typeof otherIdString === 'string' && otherIdString.toLowerCase() === thisIdString;
255      }
256  
257      return false;
258    }

290    /**
291     * Creates an ObjectId from a second based number, with the rest of the ObjectId zeroed out. Used for comparisons or sorting the ObjectId.
292     *
293     * @param time - an integer number representing a number of seconds.
294     */
295    static createFromTime(time: number): ObjectId {
296      const buffer = ByteUtils.allocate(12);
297      for (let i = 11; i >= 4; i--) buffer[i] = 0;
298      // Encode time into first 4 bytes
299      NumberUtils.setInt32BE(buffer, 0, time);
300      // Return the new objectId
301      return new ObjectId(buffer);
302    }
