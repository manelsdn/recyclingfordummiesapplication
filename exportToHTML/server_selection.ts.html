14   /** @internal */
15   export type ServerSelector = (
16     topologyDescription: TopologyDescription,
17     servers: ServerDescription[]
18   ) => ServerDescription[];

20   /**
21    * Returns a server selector that selects for writable servers
22    */
23   export function writableServerSelector(): ServerSelector {
24     return (
25       topologyDescription: TopologyDescription,
26       servers: ServerDescription[]
27     ): ServerDescription[] =>
28       latencyWindowReducer(
29         topologyDescription,
30         servers.filter((s: ServerDescription) => s.isWritable)
31       );
32   }

34   /**
35    * The purpose of this selector is to select the same server, only
36    * if it is in a state that it can have commands sent to it.
37    */
38   export function sameServerSelector(description?: ServerDescription): ServerSelector {
39     return (
40       topologyDescription: TopologyDescription,
41       servers: ServerDescription[]
42     ): ServerDescription[] => {
43       if (!description) return [];
44       // Filter the servers to match the provided description only if
45       // the type is not unknown.
46       return servers.filter(sd => {
47         return sd.address === description.address && sd.type !== ServerType.Unknown;
48       });
49     };
50   }

52   /**
53    * Returns a server selector that uses a read preference to select a
54    * server potentially for a write on a secondary.
55    */
56   export function secondaryWritableServerSelector(
57     wireVersion?: number,
58     readPreference?: ReadPreference
59   ): ServerSelector {
60     // If server version < 5.0, read preference always primary.
61     // If server version >= 5.0...
62     // - If read preference is supplied, use that.
63     // - If no read preference is supplied, use primary.
64     if (
65       !readPreference ||
66       !wireVersion ||
67       (wireVersion && wireVersion < MIN_SECONDARY_WRITE_WIRE_VERSION)
68     ) {
69       return readPreferenceServerSelector(ReadPreference.primary);
70     }
71     return readPreferenceServerSelector(readPreference);
72   }

74   /**
75    * Reduces the passed in array of servers by the rules of the "Max Staleness" specification
76    * found here: https://github.com/mongodb/specifications/blob/master/source/max-staleness/max-staleness.rst
77    *
78    * @param readPreference - The read preference providing max staleness guidance
79    * @param topologyDescription - The topology description
80    * @param servers - The list of server descriptions to be reduced
81    * @returns The list of servers that satisfy the requirements of max staleness
82    */
83   function maxStalenessReducer(
84     readPreference: ReadPreference,
85     topologyDescription: TopologyDescription,
86     servers: ServerDescription[]
87   ): ServerDescription[] {
88     if (readPreference.maxStalenessSeconds == null || readPreference.maxStalenessSeconds < 0) {
89       return servers;
90     }
91   
92     const maxStaleness = readPreference.maxStalenessSeconds;
93     const maxStalenessVariance =
94       (topologyDescription.heartbeatFrequencyMS + IDLE_WRITE_PERIOD) / 1000;
95     if (maxStaleness < maxStalenessVariance) {
96       throw new MongoInvalidArgumentError(
97         `Option "maxStalenessSeconds" must be at least ${maxStalenessVariance} seconds`
98       );
99     }
100  
101    if (maxStaleness < SMALLEST_MAX_STALENESS_SECONDS) {
102      throw new MongoInvalidArgumentError(
103        `Option "maxStalenessSeconds" must be at least ${SMALLEST_MAX_STALENESS_SECONDS} seconds`
104      );
105    }
106  
107    if (topologyDescription.type === TopologyType.ReplicaSetWithPrimary) {
108      const primary: ServerDescription = Array.from(topologyDescription.servers.values()).filter(
109        primaryFilter
110      )[0];
111  
112      return servers.reduce((result: ServerDescription[], server: ServerDescription) => {
113        const stalenessMS =
114          server.lastUpdateTime -
115          server.lastWriteDate -
116          (primary.lastUpdateTime - primary.lastWriteDate) +
117          topologyDescription.heartbeatFrequencyMS;
118  
119        const staleness = stalenessMS / 1000;
120        const maxStalenessSeconds = readPreference.maxStalenessSeconds ?? 0;
121        if (staleness <= maxStalenessSeconds) {
122          result.push(server);
123        }
124  
125        return result;
126      }, []);
127    }
128  
129    if (topologyDescription.type === TopologyType.ReplicaSetNoPrimary) {
130      if (servers.length === 0) {
131        return servers;
132      }
133  
134      const sMax = servers.reduce((max: ServerDescription, s: ServerDescription) =>
135        s.lastWriteDate > max.lastWriteDate ? s : max
136      );
137  
138      return servers.reduce((result: ServerDescription[], server: ServerDescription) => {
139        const stalenessMS =
140          sMax.lastWriteDate - server.lastWriteDate + topologyDescription.heartbeatFrequencyMS;
141  
142        const staleness = stalenessMS / 1000;
143        const maxStalenessSeconds = readPreference.maxStalenessSeconds ?? 0;
144        if (staleness <= maxStalenessSeconds) {
145          result.push(server);
146        }
147  
148        return result;
149      }, []);
150    }
151  
152    return servers;
153  }

155  /**
156   * Determines whether a server's tags match a given set of tags
157   *
158   * @param tagSet - The requested tag set to match
159   * @param serverTags - The server's tags
160   */
161  function tagSetMatch(tagSet: TagSet, serverTags: TagSet) {
162    const keys = Object.keys(tagSet);
163    const serverTagKeys = Object.keys(serverTags);
164    for (let i = 0; i < keys.length; ++i) {
165      const key = keys[i];
166      if (serverTagKeys.indexOf(key) === -1 || serverTags[key] !== tagSet[key]) {
167        return false;
168      }
169    }
170  
171    return true;
172  }

174  /**
175   * Reduces a set of server descriptions based on tags requested by the read preference
176   *
177   * @param readPreference - The read preference providing the requested tags
178   * @param servers - The list of server descriptions to reduce
179   * @returns The list of servers matching the requested tags
180   */
181  function tagSetReducer(
182    readPreference: ReadPreference,
183    servers: ServerDescription[]
184  ): ServerDescription[] {
185    if (
186      readPreference.tags == null ||
187      (Array.isArray(readPreference.tags) && readPreference.tags.length === 0)
188    ) {
189      return servers;
190    }
191  
192    for (let i = 0; i < readPreference.tags.length; ++i) {
193      const tagSet = readPreference.tags[i];
194      const serversMatchingTagset = servers.reduce(
195        (matched: ServerDescription[], server: ServerDescription) => {
196          if (tagSetMatch(tagSet, server.tags)) matched.push(server);
197          return matched;
198        },
199        []
200      );
201  
202      if (serversMatchingTagset.length) {
203        return serversMatchingTagset;
204      }
205    }
206  
207    return [];
208  }

210  /**
211   * Reduces a list of servers to ensure they fall within an acceptable latency window. This is
212   * further specified in the "Server Selection" specification, found here:
213   * https://github.com/mongodb/specifications/blob/master/source/server-selection/server-selection.rst
214   *
215   * @param topologyDescription - The topology description
216   * @param servers - The list of servers to reduce
217   * @returns The servers which fall within an acceptable latency window
218   */
219  function latencyWindowReducer(
220    topologyDescription: TopologyDescription,
221    servers: ServerDescription[]
222  ): ServerDescription[] {
223    const low = servers.reduce(
224      (min: number, server: ServerDescription) =>
225        min === -1 ? server.roundTripTime : Math.min(server.roundTripTime, min),
226      -1
227    );
228  
229    const high = low + topologyDescription.localThresholdMS;
230    return servers.reduce((result: ServerDescription[], server: ServerDescription) => {
231      if (server.roundTripTime <= high && server.roundTripTime >= low) result.push(server);
232      return result;
233    }, []);
234  }

236  // filters
237  function primaryFilter(server: ServerDescription): boolean {
238    return server.type === ServerType.RSPrimary;
239  }
240  
241  function secondaryFilter(server: ServerDescription): boolean {
242    return server.type === ServerType.RSSecondary;
243  }
244  
245  function nearestFilter(server: ServerDescription): boolean {
246    return server.type === ServerType.RSSecondary || server.type === ServerType.RSPrimary;
247  }
248  
249  function knownFilter(server: ServerDescription): boolean {
250    return server.type !== ServerType.Unknown;
251  }
252  
253  function loadBalancerFilter(server: ServerDescription): boolean {
254    return server.type === ServerType.LoadBalancer;
255  }

257  /**
258   * Returns a function which selects servers based on a provided read preference
259   *
260   * @param readPreference - The read preference to select with
261   */
262  export function readPreferenceServerSelector(readPreference: ReadPreference): ServerSelector {
263    if (!readPreference.isValid()) {
264      throw new MongoInvalidArgumentError('Invalid read preference specified');
265    }
266  
267    return (
268      topologyDescription: TopologyDescription,
269      servers: ServerDescription[]
270    ): ServerDescription[] => {
271      const commonWireVersion = topologyDescription.commonWireVersion;
272      if (
273        commonWireVersion &&
274        readPreference.minWireVersion &&
275        readPreference.minWireVersion > commonWireVersion
276      ) {
277        throw new MongoCompatibilityError(
278          `Minimum wire version '${readPreference.minWireVersion}' required, but found '${commonWireVersion}'`
279        );
280      }
281  
282      if (topologyDescription.type === TopologyType.LoadBalanced) {
283        return servers.filter(loadBalancerFilter);
284      }
285  
286      if (topologyDescription.type === TopologyType.Unknown) {
287        return [];
288      }
289  
290      if (
291        topologyDescription.type === TopologyType.Single ||
292        topologyDescription.type === TopologyType.Sharded
293      ) {
294        return latencyWindowReducer(topologyDescription, servers.filter(knownFilter));
295      }
296  
297      const mode = readPreference.mode;
298      if (mode === ReadPreference.PRIMARY) {
299        return servers.filter(primaryFilter);
300      }
301  
302      if (mode === ReadPreference.PRIMARY_PREFERRED) {
303        const result = servers.filter(primaryFilter);
304        if (result.length) {
305          return result;
306        }
307      }
308  
309      const filter = mode === ReadPreference.NEAREST ? nearestFilter : secondaryFilter;
310      const selectedServers = latencyWindowReducer(
311        topologyDescription,
312        tagSetReducer(
313          readPreference,
314          maxStalenessReducer(readPreference, topologyDescription, servers.filter(filter))
315        )
316      );
317  
318      if (mode === ReadPreference.SECONDARY_PREFERRED && selectedServers.length === 0) {
319        return servers.filter(primaryFilter);
320      }
321  
322      return selectedServers;
323    };
