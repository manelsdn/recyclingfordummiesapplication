14   // https://github.com/mongodb/specifications/blob/master/source/auth/auth.rst
15   function getDefaultAuthMechanism(hello?: Document): AuthMechanism {
16     if (hello) {
17       // If hello contains saslSupportedMechs, use scram-sha-256
18       // if it is available, else scram-sha-1
19       if (Array.isArray(hello.saslSupportedMechs)) {
20         return hello.saslSupportedMechs.includes(AuthMechanism.MONGODB_SCRAM_SHA256)
21           ? AuthMechanism.MONGODB_SCRAM_SHA256
22           : AuthMechanism.MONGODB_SCRAM_SHA1;
23       }
24   
25       // Fallback to legacy selection method. If wire version >= 3, use scram-sha-1
26       if (hello.maxWireVersion >= 3) {
27         return AuthMechanism.MONGODB_SCRAM_SHA1;
28       }
29     }
30   
31     // Default for wireprotocol < 3
32     return AuthMechanism.MONGODB_CR;
33   }

81   /**
82    * A representation of the credentials used by MongoDB
83    * @public
84    */
85   export class MongoCredentials {
86     /** The username used for authentication */
87     readonly username: string;
88     /** The password used for authentication */
89     readonly password: string;
90     /** The database that the user should authenticate against */
91     readonly source: string;
92     /** The method used to authenticate */
93     readonly mechanism: AuthMechanism;
94     /** Special properties used by some types of auth mechanisms */
95     readonly mechanismProperties: AuthMechanismProperties;
96   
97     constructor(options: MongoCredentialsOptions) {
98       this.username = options.username ?? '';
99       this.password = options.password;
100      this.source = options.source;
101      if (!this.source && options.db) {
102        this.source = options.db;
103      }
104      this.mechanism = options.mechanism || AuthMechanism.MONGODB_DEFAULT;
105      this.mechanismProperties = options.mechanismProperties || {};
106  
107      if (this.mechanism.match(/MONGODB-AWS/i)) {
108        if (!this.username && process.env.AWS_ACCESS_KEY_ID) {
109          this.username = process.env.AWS_ACCESS_KEY_ID;
110        }
111  
112        if (!this.password && process.env.AWS_SECRET_ACCESS_KEY) {
113          this.password = process.env.AWS_SECRET_ACCESS_KEY;
114        }
115  
116        if (
117          this.mechanismProperties.AWS_SESSION_TOKEN == null &&
118          process.env.AWS_SESSION_TOKEN != null
119        ) {
120          this.mechanismProperties = {
121            ...this.mechanismProperties,
122            AWS_SESSION_TOKEN: process.env.AWS_SESSION_TOKEN
123          };
124        }
125      }
126  
127      if (this.mechanism === AuthMechanism.MONGODB_OIDC && !this.mechanismProperties.ALLOWED_HOSTS) {
128        this.mechanismProperties = {
129          ...this.mechanismProperties,
130          ALLOWED_HOSTS: DEFAULT_ALLOWED_HOSTS
131        };
132      }
133  
134      Object.freeze(this.mechanismProperties);
135      Object.freeze(this);
136    }
137  
138    /** Determines if two MongoCredentials objects are equivalent */
139    equals(other: MongoCredentials): boolean {
140      return (
141        this.mechanism === other.mechanism &&
142        this.username === other.username &&
143        this.password === other.password &&
144        this.source === other.source
145      );
146    }

148    /**
149     * If the authentication mechanism is set to "default", resolves the authMechanism
150     * based on the server version and server supported sasl mechanisms.
151     *
152     * @param hello - A hello response from the server
153     */
154    resolveAuthMechanism(hello?: Document): MongoCredentials {
155      // If the mechanism is not "default", then it does not need to be resolved
156      if (this.mechanism.match(/DEFAULT/i)) {
157        return new MongoCredentials({
158          username: this.username,
159          password: this.password,
160          source: this.source,
161          mechanism: getDefaultAuthMechanism(hello),
162          mechanismProperties: this.mechanismProperties
163        });
164      }
165  
166      return this;
167    }

169    validate(): void {
170      if (
171        (this.mechanism === AuthMechanism.MONGODB_GSSAPI ||
172          this.mechanism === AuthMechanism.MONGODB_CR ||
173          this.mechanism === AuthMechanism.MONGODB_PLAIN ||
174          this.mechanism === AuthMechanism.MONGODB_SCRAM_SHA1 ||
175          this.mechanism === AuthMechanism.MONGODB_SCRAM_SHA256) &&
176        !this.username
177      ) {
178        throw new MongoMissingCredentialsError(`Username required for mechanism '${this.mechanism}'`);
179      }
180  
181      if (this.mechanism === AuthMechanism.MONGODB_OIDC) {
182        if (this.username && this.mechanismProperties.PROVIDER_NAME) {
183          throw new MongoInvalidArgumentError(
184            `username and PROVIDER_NAME may not be used together for mechanism '${this.mechanism}'.`
185          );
186        }
187  
188        if (
189          this.mechanismProperties.PROVIDER_NAME === 'azure' &&
190          !this.mechanismProperties.TOKEN_AUDIENCE
191        ) {
192          throw new MongoAzureError(TOKEN_AUDIENCE_MISSING_ERROR);
193        }
194  
195        if (
196          this.mechanismProperties.PROVIDER_NAME &&
197          !ALLOWED_PROVIDER_NAMES.includes(this.mechanismProperties.PROVIDER_NAME)
198        ) {
199          throw new MongoInvalidArgumentError(
200            `Currently only a PROVIDER_NAME in ${ALLOWED_PROVIDER_NAMES.join(
201              ','
202            )} is supported for mechanism '${this.mechanism}'.`
203          );
204        }
205  
206        if (
207          this.mechanismProperties.REFRESH_TOKEN_CALLBACK &&
208          !this.mechanismProperties.REQUEST_TOKEN_CALLBACK
209        ) {
210          throw new MongoInvalidArgumentError(
211            `A REQUEST_TOKEN_CALLBACK must be provided when using a REFRESH_TOKEN_CALLBACK for mechanism '${this.mechanism}'`
212          );
213        }
214  
215        if (
216          !this.mechanismProperties.PROVIDER_NAME &&
217          !this.mechanismProperties.REQUEST_TOKEN_CALLBACK
218        ) {
219          throw new MongoInvalidArgumentError(
220            `Either a PROVIDER_NAME or a REQUEST_TOKEN_CALLBACK must be specified for mechanism '${this.mechanism}'.`
221          );
222        }
223  
224        if (this.mechanismProperties.ALLOWED_HOSTS) {
225          const hosts = this.mechanismProperties.ALLOWED_HOSTS;
226          if (!Array.isArray(hosts)) {
227            throw new MongoInvalidArgumentError(ALLOWED_HOSTS_ERROR);
228          }
229          for (const host of hosts) {
230            if (typeof host !== 'string') {
231              throw new MongoInvalidArgumentError(ALLOWED_HOSTS_ERROR);
232            }
233          }
234        }
235      }
236  
237      if (AUTH_MECHS_AUTH_SRC_EXTERNAL.has(this.mechanism)) {
238        if (this.source != null && this.source !== '$external') {
239          // TODO(NODE-3485): Replace this with a MongoAuthValidationError
240          throw new MongoAPIError(
241            `Invalid source '${this.source}' for mechanism '${this.mechanism}' specified.`
242          );
243        }
244      }
245  
246      if (this.mechanism === AuthMechanism.MONGODB_PLAIN && this.source == null) {
247        // TODO(NODE-3485): Replace this with a MongoAuthValidationError
248        throw new MongoAPIError('PLAIN Authentication Mechanism needs an auth source');
249      }
250  
251      if (this.mechanism === AuthMechanism.MONGODB_X509 && this.password != null) {
252        if (this.password === '') {
253          Reflect.set(this, 'password', undefined);
254          return;
255        }
256        // TODO(NODE-3485): Replace this with a MongoAuthValidationError
257        throw new MongoAPIError(`Password not allowed for mechanism MONGODB-X509`);
258      }
259  
260      const canonicalization = this.mechanismProperties.CANONICALIZE_HOST_NAME ?? false;
261      if (!Object.values(GSSAPICanonicalizationValue).includes(canonicalization)) {
262        throw new MongoAPIError(`Invalid CANONICALIZE_HOST_NAME value: ${canonicalization}`);
263      }
264    }

266    static merge(
267      creds: MongoCredentials | undefined,
268      options: Partial<MongoCredentialsOptions>
269    ): MongoCredentials {
270      return new MongoCredentials({
271        username: options.username ?? creds?.username ?? '',
272        password: options.password ?? creds?.password ?? '',
273        mechanism: options.mechanism ?? creds?.mechanism ?? AuthMechanism.MONGODB_DEFAULT,
274        mechanismProperties: options.mechanismProperties ?? creds?.mechanismProperties ?? {},
275        source: options.source ?? options.db ?? creds?.source ?? 'admin'
276      });
277    }
