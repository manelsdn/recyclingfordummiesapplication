19   class ScramSHA extends AuthProvider {
20     cryptoMethod: CryptoMethod;
21     randomBytesAsync: (size: number) => Promise<Buffer>;
22     constructor(cryptoMethod: CryptoMethod) {
23       super();
24       this.cryptoMethod = cryptoMethod || 'sha1';
25       this.randomBytesAsync = promisify(crypto.randomBytes);
26     }
27   
28     override async prepare(
29       handshakeDoc: HandshakeDocument,
30       authContext: AuthContext
31     ): Promise<HandshakeDocument> {
32       const cryptoMethod = this.cryptoMethod;
33       const credentials = authContext.credentials;
34       if (!credentials) {
35         throw new MongoMissingCredentialsError('AuthContext must provide credentials.');
36       }
37   
38       const nonce = await this.randomBytesAsync(24);
39       // store the nonce for later use
40       authContext.nonce = nonce;
41   
42       const request = {
43         ...handshakeDoc,
44         speculativeAuthenticate: {
45           ...makeFirstMessage(cryptoMethod, credentials, nonce),
46           db: credentials.source
47         }
48       };
49   
50       return request;
51     }
52   
53     override async auth(authContext: AuthContext) {
54       const { reauthenticating, response } = authContext;
55       if (response?.speculativeAuthenticate && !reauthenticating) {
56         return continueScramConversation(
57           this.cryptoMethod,
58           response.speculativeAuthenticate,
59           authContext
60         );
61       }
62       return executeScram(this.cryptoMethod, authContext);
63     }
64   }

66   function cleanUsername(username: string) {
67     return username.replace('=', '=3D').replace(',', '=2C');
68   }
69   
70   function clientFirstMessageBare(username: string, nonce: Buffer) {
71     // NOTE: This is done b/c Javascript uses UTF-16, but the server is hashing in UTF-8.
72     // Since the username is not sasl-prep-d, we need to do this here.
73     return Buffer.concat([
74       Buffer.from('n=', 'utf8'),
75       Buffer.from(username, 'utf8'),
76       Buffer.from(',r=', 'utf8'),
77       Buffer.from(nonce.toString('base64'), 'utf8')
78     ]);
79   }

81   function makeFirstMessage(
82     cryptoMethod: CryptoMethod,
83     credentials: MongoCredentials,
84     nonce: Buffer
85   ) {
86     const username = cleanUsername(credentials.username);
87     const mechanism =
88       cryptoMethod === 'sha1' ? AuthMechanism.MONGODB_SCRAM_SHA1 : AuthMechanism.MONGODB_SCRAM_SHA256;
89   
90     // NOTE: This is done b/c Javascript uses UTF-16, but the server is hashing in UTF-8.
91     // Since the username is not sasl-prep-d, we need to do this here.
92     return {
93       saslStart: 1,
94       mechanism,
95       payload: new Binary(
96         Buffer.concat([Buffer.from('n,,', 'utf8'), clientFirstMessageBare(username, nonce)])
97       ),
98       autoAuthorize: 1,
99       options: { skipEmptyExchange: true }
100    };
101  }

103  async function executeScram(cryptoMethod: CryptoMethod, authContext: AuthContext): Promise<void> {
104    const { connection, credentials } = authContext;
105    if (!credentials) {
106      throw new MongoMissingCredentialsError('AuthContext must provide credentials.');
107    }
108    if (!authContext.nonce) {
109      throw new MongoInvalidArgumentError('AuthContext must contain a valid nonce property');
110    }
111    const nonce = authContext.nonce;
112    const db = credentials.source;
113  
114    const saslStartCmd = makeFirstMessage(cryptoMethod, credentials, nonce);
115    const response = await connection.commandAsync(ns(`${db}.$cmd`), saslStartCmd, undefined);
116    await continueScramConversation(cryptoMethod, response, authContext);
117  }

119  async function continueScramConversation(
120    cryptoMethod: CryptoMethod,
121    response: Document,
122    authContext: AuthContext
123  ): Promise<void> {
124    const connection = authContext.connection;
125    const credentials = authContext.credentials;
126    if (!credentials) {
127      throw new MongoMissingCredentialsError('AuthContext must provide credentials.');
128    }
129    if (!authContext.nonce) {
130      throw new MongoInvalidArgumentError('Unable to continue SCRAM without valid nonce');
131    }
132    const nonce = authContext.nonce;
133  
134    const db = credentials.source;
135    const username = cleanUsername(credentials.username);
136    const password = credentials.password;
137  
138    const processedPassword =
139      cryptoMethod === 'sha256' ? saslprep(password) : passwordDigest(username, password);
140  
141    const payload: Binary = Buffer.isBuffer(response.payload)
142      ? new Binary(response.payload)
143      : response.payload;
144  
145    const dict = parsePayload(payload);
146  
147    const iterations = parseInt(dict.i, 10);
148    if (iterations && iterations < 4096) {
149      // TODO(NODE-3483)
150      throw new MongoRuntimeError(`Server returned an invalid iteration count ${iterations}`);
151    }
152  
153    const salt = dict.s;
154    const rnonce = dict.r;
155    if (rnonce.startsWith('nonce')) {
156      // TODO(NODE-3483)
157      throw new MongoRuntimeError(`Server returned an invalid nonce: ${rnonce}`);
158    }
159  
160    // Set up start of proof
161    const withoutProof = `c=biws,r=${rnonce}`;
162    const saltedPassword = HI(
163      processedPassword,
164      Buffer.from(salt, 'base64'),
165      iterations,
166      cryptoMethod
167    );
168  
169    const clientKey = HMAC(cryptoMethod, saltedPassword, 'Client Key');
170    const serverKey = HMAC(cryptoMethod, saltedPassword, 'Server Key');
171    const storedKey = H(cryptoMethod, clientKey);
172    const authMessage = [
173      clientFirstMessageBare(username, nonce),
174      payload.toString('utf8'),
175      withoutProof
176    ].join(',');
177  
178    const clientSignature = HMAC(cryptoMethod, storedKey, authMessage);
179    const clientProof = `p=${xor(clientKey, clientSignature)}`;
180    const clientFinal = [withoutProof, clientProof].join(',');
181  
182    const serverSignature = HMAC(cryptoMethod, serverKey, authMessage);
183    const saslContinueCmd = {
184      saslContinue: 1,
185      conversationId: response.conversationId,
186      payload: new Binary(Buffer.from(clientFinal))
187    };
188  
189    const r = await connection.commandAsync(ns(`${db}.$cmd`), saslContinueCmd, undefined);
190    const parsedResponse = parsePayload(r.payload);
191  
192    if (!compareDigest(Buffer.from(parsedResponse.v, 'base64'), serverSignature)) {
193      throw new MongoRuntimeError('Server returned an invalid signature');
194    }
195  
196    if (r.done !== false) {
197      // If the server sends r.done === true we can save one RTT
198      return;
199    }
200  
201    const retrySaslContinueCmd = {
202      saslContinue: 1,
203      conversationId: r.conversationId,
204      payload: Buffer.alloc(0)
205    };
206  
207    await connection.commandAsync(ns(`${db}.$cmd`), retrySaslContinueCmd, undefined);
208  }

210  function parsePayload(payload: Binary) {
211    const payloadStr = payload.toString('utf8');
212    const dict: Document = {};
213    const parts = payloadStr.split(',');
214    for (let i = 0; i < parts.length; i++) {
215      const valueParts = parts[i].split('=');
216      dict[valueParts[0]] = valueParts[1];
217    }
218    return dict;
219  }

221  function passwordDigest(username: string, password: string) {
222    if (typeof username !== 'string') {
223      throw new MongoInvalidArgumentError('Username must be a string');
224    }
225  
226    if (typeof password !== 'string') {
227      throw new MongoInvalidArgumentError('Password must be a string');
228    }
229  
230    if (password.length === 0) {
231      throw new MongoInvalidArgumentError('Password cannot be empty');
232    }
233  
234    let md5: crypto.Hash;
235    try {
236      md5 = crypto.createHash('md5');
237    } catch (err) {
238      if (crypto.getFips()) {
239        // This error is (slightly) more helpful than what comes from OpenSSL directly, e.g.
240        // 'Error: error:060800C8:digital envelope routines:EVP_DigestInit_ex:disabled for FIPS'
241        throw new Error('Auth mechanism SCRAM-SHA-1 is not supported in FIPS mode');
242      }
243      throw err;
244    }
245    md5.update(`${username}:mongo:${password}`, 'utf8');
246    return md5.digest('hex');
247  }

249  // XOR two buffers
250  function xor(a: Buffer, b: Buffer) {
251    if (!Buffer.isBuffer(a)) {
252      a = Buffer.from(a);
253    }
254  
255    if (!Buffer.isBuffer(b)) {
256      b = Buffer.from(b);
257    }
258  
259    const length = Math.max(a.length, b.length);
260    const res = [];
261  
262    for (let i = 0; i < length; i += 1) {
263      res.push(a[i] ^ b[i]);
264    }
265  
266    return Buffer.from(res).toString('base64');
267  }

269  function H(method: CryptoMethod, text: Buffer) {
270    return crypto.createHash(method).update(text).digest();
271  }
272  
273  function HMAC(method: CryptoMethod, key: Buffer, text: Buffer | string) {
274    return crypto.createHmac(method, key).update(text).digest();
275  }

277  interface HICache {
278    [key: string]: Buffer;
279  }

281  let _hiCache: HICache = {};
282  let _hiCacheCount = 0;
283  function _hiCachePurge() {
284    _hiCache = {};
285    _hiCacheCount = 0;
286  }
287  
288  const hiLengthMap = {
289    sha256: 32,
290    sha1: 20
291  };
292  
293  function HI(data: string, salt: Buffer, iterations: number, cryptoMethod: CryptoMethod) {
294    // omit the work if already generated
295    const key = [data, salt.toString('base64'), iterations].join('_');
296    if (_hiCache[key] != null) {
297      return _hiCache[key];
298    }
299  
300    // generate the salt
301    const saltedData = crypto.pbkdf2Sync(
302      data,
303      salt,
304      iterations,
305      hiLengthMap[cryptoMethod],
306      cryptoMethod
307    );
308  
309    // cache a copy to speed up the next lookup, but prevent unbounded cache growth
310    if (_hiCacheCount >= 200) {
311      _hiCachePurge();
312    }
313  
314    _hiCache[key] = saltedData;
315    _hiCacheCount += 1;
316    return saltedData;
317  }

319  function compareDigest(lhs: Buffer, rhs: Uint8Array) {
320    if (lhs.length !== rhs.length) {
321      return false;
322    }
323  
324    if (typeof crypto.timingSafeEqual === 'function') {
325      return crypto.timingSafeEqual(lhs, rhs);
326    }
327  
328    let result = 0;
329    for (let i = 0; i < lhs.length; i++) {
330      result |= lhs[i] ^ rhs[i];
331    }
332  
333    return result === 0;
334  }
