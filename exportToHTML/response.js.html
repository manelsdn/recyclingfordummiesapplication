99   /**
100   * Send a response.
101   *
102   * Examples:
103   *
104   *     res.send(Buffer.from('wahoo'));
105   *     res.send({ some: 'json' });
106   *     res.send('<p>some html</p>');
107   *
108   * @param {string|number|boolean|object|Buffer} body
109   * @public
110   */
111  
112  res.send = function send(body) {
113    var chunk = body;
114    var encoding;
115    var req = this.req;
116    var type;
117  
118    // settings
119    var app = this.app;
120  
121    // allow status / body
122    if (arguments.length === 2) {
123      // res.send(body, status) backwards compat
124      if (typeof arguments[0] !== 'number' && typeof arguments[1] === 'number') {
125        deprecate('res.send(body, status): Use res.status(status).send(body) instead');
126        this.statusCode = arguments[1];
127      } else {
128        deprecate('res.send(status, body): Use res.status(status).send(body) instead');
129        this.statusCode = arguments[0];
130        chunk = arguments[1];
131      }
132    }
133  
134    // disambiguate res.send(status) and res.send(status, num)
135    if (typeof chunk === 'number' && arguments.length === 1) {
136      // res.send(status) will set status message as text string
137      if (!this.get('Content-Type')) {
138        this.type('txt');
139      }
140  
141      deprecate('res.send(status): Use res.sendStatus(status) instead');
142      this.statusCode = chunk;
143      chunk = statuses.message[chunk]
144    }
145  
146    switch (typeof chunk) {
147      // string defaulting to html
148      case 'string':
149        if (!this.get('Content-Type')) {
150          this.type('html');
151        }
152        break;
153      case 'boolean':
154      case 'number':
155      case 'object':
156        if (chunk === null) {
157          chunk = '';
158        } else if (Buffer.isBuffer(chunk)) {
159          if (!this.get('Content-Type')) {
160            this.type('bin');
161          }
162        } else {
163          return this.json(chunk);
164        }
165        break;
166    }
167  
168    // write strings in utf-8
169    if (typeof chunk === 'string') {
170      encoding = 'utf8';
171      type = this.get('Content-Type');
172  
173      // reflect this in content-type
174      if (typeof type === 'string') {
175        this.set('Content-Type', setCharset(type, 'utf-8'));
176      }
177    }
178  
179    // determine if ETag should be generated
180    var etagFn = app.get('etag fn')
181    var generateETag = !this.get('ETag') && typeof etagFn === 'function'
182  
183    // populate Content-Length
184    var len
185    if (chunk !== undefined) {
186      if (Buffer.isBuffer(chunk)) {
187        // get length of Buffer
188        len = chunk.length
189      } else if (!generateETag && chunk.length < 1000) {
190        // just calculate length when no ETag + small chunk
191        len = Buffer.byteLength(chunk, encoding)
192      } else {
193        // convert chunk to Buffer and calculate
194        chunk = Buffer.from(chunk, encoding)
195        encoding = undefined;
196        len = chunk.length
197      }
198  
199      this.set('Content-Length', len);
200    }
201  
202    // populate ETag
203    var etag;
204    if (generateETag && len !== undefined) {
205      if ((etag = etagFn(chunk, encoding))) {
206        this.set('ETag', etag);
207      }
208    }
209  
210    // freshness
211    if (req.fresh) this.statusCode = 304;
212  
213    // strip irrelevant headers
214    if (204 === this.statusCode || 304 === this.statusCode) {
215      this.removeHeader('Content-Type');
216      this.removeHeader('Content-Length');
217      this.removeHeader('Transfer-Encoding');
218      chunk = '';
219    }
220  
221    // alter headers for 205
222    if (this.statusCode === 205) {
223      this.set('Content-Length', '0')
224      this.removeHeader('Transfer-Encoding')
225      chunk = ''
226    }
227  
228    if (req.method === 'HEAD') {
229      // skip body for HEAD
230      this.end();
231    } else {
232      // respond
233      this.end(chunk, encoding);
234    }
235  
236    return this;
237  };

282  /**
283   * Send JSON response with JSONP callback support.
284   *
285   * Examples:
286   *
287   *     res.jsonp(null);
288   *     res.jsonp({ user: 'tj' });
289   *
290   * @param {string|number|boolean|object} obj
291   * @public
292   */
293  
294  res.jsonp = function jsonp(obj) {
295    var val = obj;
296  
297    // allow status / body
298    if (arguments.length === 2) {
299      // res.jsonp(body, status) backwards compat
300      if (typeof arguments[1] === 'number') {
301        deprecate('res.jsonp(obj, status): Use res.status(status).jsonp(obj) instead');
302        this.statusCode = arguments[1];
303      } else {
304        deprecate('res.jsonp(status, obj): Use res.status(status).jsonp(obj) instead');
305        this.statusCode = arguments[0];
306        val = arguments[1];
307      }
308    }
309  
310    // settings
311    var app = this.app;
312    var escape = app.get('json escape')
313    var replacer = app.get('json replacer');
314    var spaces = app.get('json spaces');
315    var body = stringify(val, replacer, spaces, escape)
316    var callback = this.req.query[app.get('jsonp callback name')];
317  
318    // content-type
319    if (!this.get('Content-Type')) {
320      this.set('X-Content-Type-Options', 'nosniff');
321      this.set('Content-Type', 'application/json');
322    }
323  
324    // fixup callback
325    if (Array.isArray(callback)) {
326      callback = callback[0];
327    }
328  
329    // jsonp
330    if (typeof callback === 'string' && callback.length !== 0) {
331      this.set('X-Content-Type-Options', 'nosniff');
332      this.set('Content-Type', 'text/javascript');
333  
334      // restrict callback charset
335      callback = callback.replace(/[^\[\]\w$.]/g, '');
336  
337      if (body === undefined) {
338        // empty argument
339        body = ''
340      } else if (typeof body === 'string') {
341        // replace chars not allowed in JavaScript that are in JSON
342        body = body
343          .replace(/\u2028/g, '\\u2028')
344          .replace(/\u2029/g, '\\u2029')
345      }
346  
347      // the /**/ is a specific security mitigation for "Rosetta Flash JSONP abuse"
348      // the typeof check is just to reduce client error noise
349      body = '/**/ typeof ' + callback + ' === \'function\' && ' + callback + '(' + body + ');';
350    }
351  
352    return this.send(body);
353  };

379  /**
380   * Transfer the file at the given `path`.
381   *
382   * Automatically sets the _Content-Type_ response header field.
383   * The callback `callback(err)` is invoked when the transfer is complete
384   * or when an error occurs. Be sure to check `res.headersSent`
385   * if you wish to attempt responding, as the header and some data
386   * may have already been transferred.
387   *
388   * Options:
389   *
390   *   - `maxAge`   defaulting to 0 (can be string converted by `ms`)
391   *   - `root`     root directory for relative filenames
392   *   - `headers`  object of headers to serve with file
393   *   - `dotfiles` serve dotfiles, defaulting to false; can be `"allow"` to send them
394   *
395   * Other options are passed along to `send`.
396   *
397   * Examples:
398   *
399   *  The following example illustrates how `res.sendFile()` may
400   *  be used as an alternative for the `static()` middleware for
401   *  dynamic situations. The code backing `res.sendFile()` is actually
402   *  the same code, so HTTP cache support etc is identical.
403   *
404   *     app.get('/user/:uid/photos/:file', function(req, res){
405   *       var uid = req.params.uid
406   *         , file = req.params.file;
407   *
408   *       req.user.mayViewFilesFrom(uid, function(yes){
409   *         if (yes) {
410   *           res.sendFile('/uploads/' + uid + '/' + file);
411   *         } else {
412   *           res.send(403, 'Sorry! you cant see that.');
413   *         }
414   *       });
415   *     });
416   *
417   * @public
418   */
419  
420  res.sendFile = function sendFile(path, options, callback) {
421    var done = callback;
422    var req = this.req;
423    var res = this;
424    var next = req.next;
425    var opts = options || {};
426  
427    if (!path) {
428      throw new TypeError('path argument is required to res.sendFile');
429    }
430  
431    if (typeof path !== 'string') {
432      throw new TypeError('path must be a string to res.sendFile')
433    }
434  
435    // support function as second arg
436    if (typeof options === 'function') {
437      done = options;
438      opts = {};
439    }
440  
441    if (!opts.root && !isAbsolute(path)) {
442      throw new TypeError('path must be absolute or specify root to res.sendFile');
443    }
444  
445    // create file stream
446    var pathname = encodeURI(path);
447    var file = send(req, pathname, opts);
448  
449    // transfer
450    sendfile(res, file, opts, function (err) {
451      if (done) return done(err);
452      if (err && err.code === 'EISDIR') return next();
453  
454      // next() all but write errors
455      if (err && err.code !== 'ECONNABORTED' && err.syscall !== 'write') {
456        next(err);
457      }
458    });
459  };

533  /**
534   * Transfer the file at the given `path` as an attachment.
535   *
536   * Optionally providing an alternate attachment `filename`,
537   * and optional callback `callback(err)`. The callback is invoked
538   * when the data transfer is complete, or when an error has
539   * occurred. Be sure to check `res.headersSent` if you plan to respond.
540   *
541   * Optionally providing an `options` object to use with `res.sendFile()`.
542   * This function will set the `Content-Disposition` header, overriding
543   * any `Content-Disposition` header passed as header options in order
544   * to set the attachment and filename.
545   *
546   * This method uses `res.sendFile()`.
547   *
548   * @public
549   */
550  
551  res.download = function download (path, filename, options, callback) {
552    var done = callback;
553    var name = filename;
554    var opts = options || null
555  
556    // support function as second or third arg
557    if (typeof filename === 'function') {
558      done = filename;
559      name = null;
560      opts = null
561    } else if (typeof options === 'function') {
562      done = options
563      opts = null
564    }
565  
566    // support optional filename, where options may be in it's place
567    if (typeof filename === 'object' &&
568      (typeof options === 'function' || options === undefined)) {
569      name = null
570      opts = filename
571    }
572  
573    // set Content-Disposition when file is sent
574    var headers = {
575      'Content-Disposition': contentDisposition(name || path)
576    };
577  
578    // merge user-provided headers
579    if (opts && opts.headers) {
580      var keys = Object.keys(opts.headers)
581      for (var i = 0; i < keys.length; i++) {
582        var key = keys[i]
583        if (key.toLowerCase() !== 'content-disposition') {
584          headers[key] = opts.headers[key]
585        }
586      }
587    }
588  
589    // merge user-provided options
590    opts = Object.create(opts)
591    opts.headers = headers
592  
593    // Resolve the full path for sendFile
594    var fullPath = !opts.root
595      ? resolve(path)
596      : path
597  
598    // send file
599    return this.sendFile(fullPath, opts, done)
600  };

628  /**
629   * Respond to the Acceptable formats using an `obj`
630   * of mime-type callbacks.
631   *
632   * This method uses `req.accepted`, an array of
633   * acceptable types ordered by their quality values.
634   * When "Accept" is not present the _first_ callback
635   * is invoked, otherwise the first match is used. When
636   * no match is performed the server responds with
637   * 406 "Not Acceptable".
638   *
639   * Content-Type is set for you, however if you choose
640   * you may alter this within the callback using `res.type()`
641   * or `res.set('Content-Type', ...)`.
642   *
643   *    res.format({
644   *      'text/plain': function(){
645   *        res.send('hey');
646   *      },
647   *
648   *      'text/html': function(){
649   *        res.send('<p>hey</p>');
650   *      },
651   *
652   *      'application/json': function () {
653   *        res.send({ message: 'hey' });
654   *      }
655   *    });
656   *
657   * In addition to canonicalized MIME types you may
658   * also use extnames mapped to these types:
659   *
660   *    res.format({
661   *      text: function(){
662   *        res.send('hey');
663   *      },
664   *
665   *      html: function(){
666   *        res.send('<p>hey</p>');
667   *      },
668   *
669   *      json: function(){
670   *        res.send({ message: 'hey' });
671   *      }
672   *    });
673   *
674   * By default Express passes an `Error`
675   * with a `.status` of 406 to `next(err)`
676   * if a match is not made. If you provide
677   * a `.default` callback it will be invoked
678   * instead.
679   *
680   * @param {Object} obj
681   * @return {ServerResponse} for chaining
682   * @public
683   */
684  
685  res.format = function(obj){
686    var req = this.req;
687    var next = req.next;
688  
689    var keys = Object.keys(obj)
690      .filter(function (v) { return v !== 'default' })
691  
692    var key = keys.length > 0
693      ? req.accepts(keys)
694      : false;
695  
696    this.vary("Accept");
697  
698    if (key) {
699      this.set('Content-Type', normalizeType(key).value);
700      obj[key](req, this, next);
701    } else if (obj.default) {
702      obj.default(req, this, next)
703    } else {
704      next(createError(406, {
705        types: normalizeTypes(keys).map(function (o) { return o.value })
706      }))
707    }
708  
709    return this;
710  };

732   *
733   * Example:
734   *
735   *    res.append('Link', ['<http://localhost/>', '<http://localhost:3000/>']);
736   *    res.append('Set-Cookie', 'foo=bar; Path=/; HttpOnly');
737   *    res.append('Warning', '199 Miscellaneous warning');
738   *
739   * @param {String} field
740   * @param {String|Array} val
741   * @return {ServerResponse} for chaining
742   * @public
743   */
744  
745  res.append = function append(field, val) {
746    var prev = this.get(field);
747    var value = val;
748  
749    if (prev) {
750      // concat the new and prev vals
751      value = Array.isArray(prev) ? prev.concat(val)
752        : Array.isArray(val) ? [prev].concat(val)
753          : [prev, val]
754    }
755  
756    return this.set(field, value);
757  };

759  /**
760   * Set header `field` to `val`, or pass
761   * an object of header fields.
762   *
763   * Examples:
764   *
765   *    res.set('Foo', ['bar', 'baz']);
766   *    res.set('Accept', 'application/json');
767   *    res.set({ Accept: 'text/plain', 'X-API-Key': 'tobi' });
768   *
769   * Aliased as `res.header()`.

779    if (arguments.length === 2) {
780      var value = Array.isArray(val)
781        ? val.map(String)
782        : String(val);
783  
784      // add charset to content-type
785      if (field.toLowerCase() === 'content-type') {
786        if (Array.isArray(value)) {
787          throw new TypeError('Content-Type cannot be set to an Array');
788        }
789        if (!charsetRegExp.test(value)) {
790          var charset = mime.charsets.lookup(value.split(';')[0]);
791          if (charset) value += '; charset=' + charset.toLowerCase();
792        }
793      }
794  
795      this.setHeader(field, value);
796    } else {
797      for (var key in field) {
798        this.set(key, field[key]);
799      }
800    }
801    return this;
802  };

831  /**
832   * Set cookie `name` to `value`, with the given `options`.
833   *
834   * Options:
835   *
836   *    - `maxAge`   max-age in milliseconds, converted to `expires`
837   *    - `signed`   sign the cookie
838   *    - `path`     defaults to "/"
839   *
840   * Examples:
841   *
842   *    // "Remember Me" for 15 minutes
843   *    res.cookie('rememberme', '1', { expires: new Date(Date.now() + 900000), httpOnly: true });
844   *
845   *    // same as above
846   *    res.cookie('rememberme', '1', { maxAge: 900000, httpOnly: true })
847   *
848   * @param {String} name
849   * @param {String|Object} value
850   * @param {Object} [options]
851   * @return {ServerResponse} for chaining
852   * @public
853   */
854  
855  res.cookie = function (name, value, options) {
856    var opts = merge({}, options);
857    var secret = this.req.secret;
858    var signed = opts.signed;
859  
860    if (signed && !secret) {
861      throw new Error('cookieParser("secret") required for signed cookies');
862    }
863  
864    var val = typeof value === 'object'
865      ? 'j:' + JSON.stringify(value)
866      : String(value);
867  
868    if (signed) {
869      val = 's:' + sign(val, secret);
870    }
871  
872    if (opts.maxAge != null) {
873      var maxAge = opts.maxAge - 0
874  
875      if (!isNaN(maxAge)) {
876        opts.expires = new Date(Date.now() + maxAge)
877        opts.maxAge = Math.floor(maxAge / 1000)
878      }
879    }
880  
881    if (opts.path == null) {
882      opts.path = '/';
883    }
884  
885    this.append('Set-Cookie', cookie.serialize(name, String(val), opts));
886  
887    return this;
888  };

890  /**
891   * Set the location header to `url`.
892   *
893   * The given `url` can also be "back", which redirects
894   * to the _Referrer_ or _Referer_ headers or "/".
895   *
896   * Examples:
897   *
898   *    res.location('/foo/bar').;
899   *    res.location('http://example.com');
900   *    res.location('../login');
901   *
902   * @param {String} url
903   * @return {ServerResponse} for chaining
904   * @public
905   */
906  
907  res.location = function location(url) {
908    var loc;
909  
910    // "back" is an alias for the referrer
911    if (url === 'back') {
912      loc = this.req.get('Referrer') || '/';
913    } else {
914      loc = String(url);
915    }
916  
917    var m = schemaAndHostRegExp.exec(loc);
918    var pos = m ? m[0].length + 1 : 0;
919  
920    // Only encode after host to avoid invalid encoding which can introduce
921    // vulnerabilities (e.g. `\\` to `%5C`).
922    loc = loc.slice(0, pos) + encodeUrl(loc.slice(pos));
923  
924    return this.set('Location', loc);
925  };

927  /**
928   * Redirect to the given `url` with optional response `status`
929   * defaulting to 302.
930   *
931   * The resulting `url` is determined by `res.location()`, so
932   * it will play nicely with mounted apps, relative paths,
933   * `"back"` etc.
934   *
935   * Examples:
936   *
937   *    res.redirect('/foo/bar');
938   *    res.redirect('http://example.com');
939   *    res.redirect(301, 'http://example.com');
940   *    res.redirect('../login'); // /blog/post/1 -> /blog/login
941   *
942   * @public
943   */
944  
945  res.redirect = function redirect(url) {
946    var address = url;
947    var body;
948    var status = 302;
949  
950    // allow status / url
951    if (arguments.length === 2) {
952      if (typeof arguments[0] === 'number') {
953        status = arguments[0];
954        address = arguments[1];
955      } else {
956        deprecate('res.redirect(url, status): Use res.redirect(status, url) instead');
957        status = arguments[1];
958      }
959    }
960  
961    // Set location header
962    address = this.location(address).get('Location');
963  
964    // Support text/{plain,html} by default
965    this.format({
966      text: function(){
967        body = statuses.message[status] + '. Redirecting to ' + address
968      },
969  
970      html: function(){
971        var u = escapeHtml(address);
972        body = '<p>' + statuses.message[status] + '. Redirecting to <a href="' + u + '">' + u + '</a></p>'
973      },
974  
975      default: function(){
976        body = '';
977      }
978    });
979  
980    // Respond
981    this.statusCode = status;
982    this.set('Content-Length', Buffer.byteLength(body));
983  
984    if (this.req.method === 'HEAD') {
985      this.end();
986    } else {
987      this.end(body);
988    }
989  };

1012 /**
1013  * Render `view` with the given `options` and optional callback `fn`.
1014  * When a callback function is given a response will _not_ be made
1015  * automatically, otherwise a response of _200_ and _text/html_ is given.
1016  *
1017  * Options:
1018  *
1019  *  - `cache`     boolean hinting to the engine it should cache
1020  *  - `filename`  filename of the view being rendered
1021  *
1022  * @public
1023  */
1024 
1025 res.render = function render(view, options, callback) {
1026   var app = this.req.app;
1027   var done = callback;
1028   var opts = options || {};
1029   var req = this.req;
1030   var self = this;
1031 
1032   // support callback function as second arg
1033   if (typeof options === 'function') {
1034     done = options;
1035     opts = {};
1036   }
1037 
1038   // merge res.locals
1039   opts._locals = self.locals;
1040 
1041   // default callback to respond
1042   done = done || function (err, str) {
1043     if (err) return req.next(err);
1044     self.send(str);
1045   };
1046 
1047   // render
1048   app.render(view, opts, done);
1049 };

1051 // pipe the send file stream
1052 function sendfile(res, file, options, callback) {
1053   var done = false;
1054   var streaming;
1055 
1056   // request aborted
1057   function onaborted() {
1058     if (done) return;
1059     done = true;
1060 
1061     var err = new Error('Request aborted');
1062     err.code = 'ECONNABORTED';
1063     callback(err);
1064   }
1065 
1066   // directory
1067   function ondirectory() {
1068     if (done) return;
1069     done = true;
1070 
1071     var err = new Error('EISDIR, read');
1072     err.code = 'EISDIR';
1073     callback(err);
1074   }
1075 
1076   // errors
1077   function onerror(err) {
1078     if (done) return;
1079     done = true;
1080     callback(err);
1081   }
1082 
1083   // ended
1084   function onend() {
1085     if (done) return;
1086     done = true;
1087     callback();
1088   }
1089 
1090   // file
1091   function onfile() {
1092     streaming = false;
1093   }
1094 
1095   // finished
1096   function onfinish(err) {
1097     if (err && err.code === 'ECONNRESET') return onaborted();
1098     if (err) return onerror(err);
1099     if (done) return;
1100 
1101     setImmediate(function () {
1102       if (streaming !== false && !done) {
1103         onaborted();
1104         return;
1105       }
1106 
1107       if (done) return;
1108       done = true;
1109       callback();
1110     });
1111   }
1112 
1113   // streaming
1114   function onstream() {
1115     streaming = true;
1116   }
1117 
1118   file.on('directory', ondirectory);
1119   file.on('end', onend);
1120   file.on('error', onerror);
1121   file.on('file', onfile);
1122   file.on('stream', onstream);
1123   onFinished(res, onfinish);
1124 
1125   if (options.headers) {
1126     // set headers on successful transfer
1127     file.on('headers', function headers(res) {
1128       var obj = options.headers;
1129       var keys = Object.keys(obj);
1130 
1131       for (var i = 0; i < keys.length; i++) {
1132         var k = keys[i];
1133         res.setHeader(k, obj[k]);
1134       }
1135     });
1136   }
1137 
1138   // pipe
1139   file.pipe(res);
1140 }

1142 /**
1143  * Stringify JSON, like JSON.stringify, but v8 optimized, with the
1144  * ability to escape characters that can trigger HTML sniffing.
1145  *
1146  * @param {*} value
1147  * @param {function} replacer
1148  * @param {number} spaces
1149  * @param {boolean} escape
1150  * @returns {string}
1151  * @private
1152  */
1153 
1154 function stringify (value, replacer, spaces, escape) {
1155   // v8 checks arguments.length for optimizing simple call
1156   // https://bugs.chromium.org/p/v8/issues/detail?id=4730
1157   var json = replacer || spaces
1158     ? JSON.stringify(value, replacer, spaces)
1159     : JSON.stringify(value);
1160 
1161   if (escape && typeof json === 'string') {
1162     json = json.replace(/[<>&]/g, function (c) {
1163       switch (c.charCodeAt(0)) {
1164         case 0x3c:
1165           return '\\u003c'
1166         case 0x3e:
1167           return '\\u003e'
1168         case 0x26:
1169           return '\\u0026'
1170         /* istanbul ignore next: unreachable default */
1171         default:
1172           return c
1173       }
1174     })
1175   }
1176 
1177   return json
1178 }
