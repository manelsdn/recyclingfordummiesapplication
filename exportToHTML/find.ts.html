13    * @public
14    * @typeParam TSchema - Unused schema definition, deprecated usage, only specify `FindOptions` with no generic
15    */
16   // eslint-disable-next-line @typescript-eslint/no-unused-vars
17   export interface FindOptions<TSchema extends Document = Document>
18     extends Omit<CommandOperationOptions, 'writeConcern'> {
19     /** Sets the limit of documents returned in the query. */
20     limit?: number;
21     /** Set to sort the documents coming back from the query. Array of indexes, `[['a', 1]]` etc. */
22     sort?: Sort;
23     /** The fields to return in the query. Object of fields to either include or exclude (one of, not both), `{'a':1, 'b': 1}` **or** `{'a': 0, 'b': 0}` */
24     projection?: Document;
25     /** Set to skip N documents ahead in your query (useful for pagination). */
26     skip?: number;
27     /** Tell the query to use specific indexes in the query. Object of indexes to use, `{'_id':1}` */
28     hint?: Hint;
29     /** Specify if the cursor can timeout. */
30     timeout?: boolean;
31     /** Specify if the cursor is tailable. */
32     tailable?: boolean;
33     /** Specify if the cursor is a tailable-await cursor. Requires `tailable` to be true */
34     awaitData?: boolean;
35     /** Set the batchSize for the getMoreCommand when iterating over the query results. */
36     batchSize?: number;
37     /** If true, returns only the index keys in the resulting documents. */
38     returnKey?: boolean;
39     /** The inclusive lower bound for a specific index */
40     min?: Document;
41     /** The exclusive upper bound for a specific index */
42     max?: Document;
43     /** Number of milliseconds to wait before aborting the query. */
44     maxTimeMS?: number;
45     /** The maximum amount of time for the server to wait on new documents to satisfy a tailable cursor query. Requires `tailable` and `awaitData` to be true */
46     maxAwaitTimeMS?: number;
47     /** The server normally times out idle cursors after an inactivity period (10 minutes) to prevent excess memory use. Set this option to prevent that. */
48     noCursorTimeout?: boolean;
49     /** Specify collation (MongoDB 3.4 or higher) settings for update operation (see 3.4 documentation for available fields). */
50     collation?: CollationOptions;
51     /** Allows disk use for blocking sort operations exceeding 100MB memory. (MongoDB 3.2 or higher) */
52     allowDiskUse?: boolean;
53     /** Determines whether to close the cursor after the first batch. Defaults to false. */
54     singleBatch?: boolean;
55     /** For queries against a sharded collection, allows the command (or subsequent getMore commands) to return partial results, rather than an error, if one or more queried shards are unavailable. */
56     allowPartialResults?: boolean;
57     /** Determines whether to return the record identifier for each document. If true, adds a field $recordId to the returned documents. */
58     showRecordId?: boolean;
59     /** Map of parameter names and values that can be accessed using $$var (requires MongoDB 5.0). */
60     let?: Document;
61     /**
62      * Option to enable an optimized code path for queries looking for a particular range of `ts` values in the oplog. Requires `tailable` to be true.
63      * @deprecated Starting from MongoDB 4.4 this flag is not needed and will be ignored.
64      */
65     oplogReplay?: boolean;
66   }

68   /** @internal */
69   export class FindOperation extends CommandOperation<Document> {
70     /**
71      * @remarks WriteConcern can still be present on the options because
72      * we inherit options from the client/db/collection.  The
73      * key must be present on the options in order to delete it.
74      * This allows typescript to delete the key but will
75      * not allow a writeConcern to be assigned as a property on options.
76      */
77     override options: FindOptions & { writeConcern?: never };
78     filter: Document;
79   
80     constructor(
81       collection: Collection | undefined,
82       ns: MongoDBNamespace,
83       filter: Document = {},
84       options: FindOptions = {}
85     ) {
86       super(collection, options);
87   
88       this.options = { ...options };
89       delete this.options.writeConcern;
90       this.ns = ns;
91   
92       if (typeof filter !== 'object' || Array.isArray(filter)) {
93         throw new MongoInvalidArgumentError('Query filter must be a plain object or ObjectId');
94       }
95   
96       // special case passing in an ObjectId as a filter
97       this.filter = filter != null && filter._bsontype === 'ObjectId' ? { _id: filter } : filter;
98     }
99   
100    override async execute(server: Server, session: ClientSession | undefined): Promise<Document> {
101      this.server = server;
102  
103      const options = this.options;
104  
105      let findCommand = makeFindCommand(this.ns, this.filter, options);
106      if (this.explain) {
107        findCommand = decorateWithExplain(findCommand, this.explain);
108      }
109  
110      return server.commandAsync(this.ns, findCommand, {
111        ...this.options,
112        ...this.bsonOptions,
113        documentsReturnedIn: 'firstBatch',
114        session
115      });
116    }
117  }

119  function makeFindCommand(ns: MongoDBNamespace, filter: Document, options: FindOptions): Document {
120    const findCommand: Document = {
121      find: ns.collection,
122      filter
123    };
124  
125    if (options.sort) {
126      findCommand.sort = formatSort(options.sort);
127    }
128  
129    if (options.projection) {
130      let projection = options.projection;
131      if (projection && Array.isArray(projection)) {
132        projection = projection.length
133          ? projection.reduce((result, field) => {
134              result[field] = 1;
135              return result;
136            }, {})
137          : { _id: 1 };
138      }
139  
140      findCommand.projection = projection;
141    }
142  
143    if (options.hint) {
144      findCommand.hint = normalizeHintField(options.hint);
145    }
146  
147    if (typeof options.skip === 'number') {
148      findCommand.skip = options.skip;
149    }
150  
151    if (typeof options.limit === 'number') {
152      if (options.limit < 0) {
153        findCommand.limit = -options.limit;
154        findCommand.singleBatch = true;
155      } else {
156        findCommand.limit = options.limit;
157      }
158    }
159  
160    if (typeof options.batchSize === 'number') {
161      if (options.batchSize < 0) {
162        if (
163          options.limit &&
164          options.limit !== 0 &&
165          Math.abs(options.batchSize) < Math.abs(options.limit)
166        ) {
167          findCommand.limit = -options.batchSize;
168        }
169  
170        findCommand.singleBatch = true;
171      } else {
172        findCommand.batchSize = options.batchSize;
173      }
174    }
175  
176    if (typeof options.singleBatch === 'boolean') {
177      findCommand.singleBatch = options.singleBatch;
178    }
179  
180    // we check for undefined specifically here to allow falsy values
181    // eslint-disable-next-line no-restricted-syntax
182    if (options.comment !== undefined) {
183      findCommand.comment = options.comment;
184    }
185  
186    if (typeof options.maxTimeMS === 'number') {
187      findCommand.maxTimeMS = options.maxTimeMS;
188    }
189  
190    const readConcern = ReadConcern.fromOptions(options);
191    if (readConcern) {
192      findCommand.readConcern = readConcern.toJSON();
193    }
194  
195    if (options.max) {
196      findCommand.max = options.max;
197    }
198  
199    if (options.min) {
200      findCommand.min = options.min;
201    }
202  
203    if (typeof options.returnKey === 'boolean') {
204      findCommand.returnKey = options.returnKey;
205    }
206  
207    if (typeof options.showRecordId === 'boolean') {
208      findCommand.showRecordId = options.showRecordId;
209    }
210  
211    if (typeof options.tailable === 'boolean') {
212      findCommand.tailable = options.tailable;
213    }
214  
215    if (typeof options.oplogReplay === 'boolean') {
216      findCommand.oplogReplay = options.oplogReplay;
217    }
218  
219    if (typeof options.timeout === 'boolean') {
220      findCommand.noCursorTimeout = !options.timeout;
221    } else if (typeof options.noCursorTimeout === 'boolean') {
222      findCommand.noCursorTimeout = options.noCursorTimeout;
223    }
224  
225    if (typeof options.awaitData === 'boolean') {
226      findCommand.awaitData = options.awaitData;
227    }
228  
229    if (typeof options.allowPartialResults === 'boolean') {
230      findCommand.allowPartialResults = options.allowPartialResults;
231    }
232  
233    if (options.collation) {
234      findCommand.collation = options.collation;
235    }
236  
237    if (typeof options.allowDiskUse === 'boolean') {
238      findCommand.allowDiskUse = options.allowDiskUse;
239    }
240  
241    if (options.let) {
242      findCommand.let = options.let;
243    }
244  
245    return findCommand;
246  }
