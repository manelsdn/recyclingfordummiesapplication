69   /** @public */
70   export interface AbstractCursorOptions extends BSONSerializeOptions {
71     session?: ClientSession;
72     readPreference?: ReadPreferenceLike;
73     readConcern?: ReadConcernLike;
74     /**
75      * Specifies the number of documents to return in each response from MongoDB
76      */
77     batchSize?: number;
78     /**
79      * When applicable `maxTimeMS` controls the amount of time the initial command
80      * that constructs a cursor should take. (ex. find, aggregate, listCollections)
81      */
82     maxTimeMS?: number;
83     /**
84      * When applicable `maxAwaitTimeMS` controls the amount of time subsequent getMores
85      * that a cursor uses to fetch more data should take. (ex. cursor.next())
86      */
87     maxAwaitTimeMS?: number;
88     /**
89      * Comment to apply to the operation.
90      *
91      * In server versions pre-4.4, 'comment' must be string.  A server
92      * error will be thrown if any other type is provided.
93      *
94      * In server versions 4.4 and above, 'comment' can be any valid BSON type.
95      */
96     comment?: unknown;
97     /**
98      * By default, MongoDB will automatically close a cursor when the
99      * client has exhausted all results in the cursor. However, for [capped collections](https://www.mongodb.com/docs/manual/core/capped-collections)
100     * you may use a Tailable Cursor that remains open after the client exhausts
101     * the results in the initial cursor.
102     */
103    tailable?: boolean;
104    /**
105     * If awaitData is set to true, when the cursor reaches the end of the capped collection,
106     * MongoDB blocks the query thread for a period of time waiting for new data to arrive.
107     * When new data is inserted into the capped collection, the blocked thread is signaled
108     * to wake up and return the next batch to the client.
109     */
110    awaitData?: boolean;
111    noCursorTimeout?: boolean;
112  }

295    async *[Symbol.asyncIterator](): AsyncGenerator<TSchema, void, void> {
296      if (this.closed) {
297        return;
298      }
299  
300      try {
301        while (true) {
302          const document = await this.next();
303  
304          // Intentional strict null check, because users can map cursors to falsey values.
305          // We allow mapping to all values except for null.
306          // eslint-disable-next-line no-restricted-syntax
307          if (document === null) {
308            if (!this.closed) {
309              const message =
310                'Cursor returned a `null` document, but the cursor is not exhausted.  Mapping documents to `null` is not supported in the cursor transform.';
311  
312              await cleanupCursor(this, { needsToEmitClosed: true }).catch(() => null);
313  
314              throw new MongoAPIError(message);
315            }
316            break;
317          }
318  
319          yield document;
320  
321          if (this[kId] === Long.ZERO) {
322            // Cursor exhausted
323            break;
324          }
325        }
326      } finally {
327        // Only close the cursor if it has not already been closed. This finally clause handles
328        // the case when a user would break out of a for await of loop early.
329        if (!this.closed) {
330          await this.close().catch(() => null);
331        }
332      }
333    }

335    stream(options?: CursorStreamOptions): Readable & AsyncIterable<TSchema> {
336      if (options?.transform) {
337        const transform = options.transform;
338        const readable = new ReadableCursorStream(this);
339  
340        return readable.pipe(
341          new Transform({
342            objectMode: true,
343            highWaterMark: 1,
344            transform(chunk, _, callback) {
345              try {
346                const transformed = transform(chunk);
347                callback(undefined, transformed);
348              } catch (err) {
349                callback(err);
350              }
351            }
352          })
353        );
354      }
355  
356      return new ReadableCursorStream(this);
357    }

359    async hasNext(): Promise<boolean> {
360      if (this[kId] === Long.ZERO) {
361        return false;
362      }
363  
364      if (this[kDocuments].length !== 0) {
365        return true;
366      }
367  
368      const doc = await next<TSchema>(this, { blocking: true, transform: false });
369  
370      if (doc) {
371        this[kDocuments].unshift(doc);
372        return true;
373      }
374  
375      return false;
376    }

458    /**
459     * Map all documents using the provided function
460     * If there is a transform set on the cursor, that will be called first and the result passed to
461     * this function's transform.
462     *
463     * @remarks
464     *
465     * **Note** Cursors use `null` internally to indicate that there are no more documents in the cursor. Providing a mapping
466     * function that maps values to `null` will result in the cursor closing itself before it has finished iterating
467     * all documents.  This will **not** result in a memory leak, just surprising behavior.  For example:
468     *
469     * ```typescript
470     * const cursor = collection.find({});
471     * cursor.map(() => null);
472     *
473     * const documents = await cursor.toArray();
474     * // documents is always [], regardless of how many documents are in the collection.
475     * ```
476     *
477     * Other falsey values are allowed:
478     *
479     * ```typescript
480     * const cursor = collection.find({});
481     * cursor.map(() => '');
482     *
483     * const documents = await cursor.toArray();
484     * // documents is now an array of empty strings
485     * ```
486     *
487     * **Note for Typescript Users:** adding a transform changes the return type of the iteration of this cursor,
488     * it **does not** return a new instance of a cursor. This means when calling map,
489     * you should always assign the result to a new variable in order to get a correctly typed cursor variable.
490     * Take note of the following example:
491     *
492     * @example
493     * ```typescript
494     * const cursor: FindCursor<Document> = coll.find();
495     * const mappedCursor: FindCursor<number> = cursor.map(doc => Object.keys(doc).length);
496     * const keyCounts: number[] = await mappedCursor.toArray(); // cursor.toArray() still returns Document[]
497     * ```
498     * @param transform - The mapping transformation method.
499     */
500    map<T = any>(transform: (doc: TSchema) => T): AbstractCursor<T> {
501      assertUninitialized(this);
502      const oldTransform = this[kTransform] as (doc: TSchema) => TSchema; // TODO(NODE-3283): Improve transform typing
503      if (oldTransform) {
504        this[kTransform] = doc => {
505          return transform(oldTransform(doc));
506        };
507      } else {
508        this[kTransform] = transform;
509      }
510  
511      return this as unknown as AbstractCursor<T>;
512    }

581    /**
582     * Rewind this cursor to its uninitialized state. Any options that are present on the cursor will
583     * remain in effect. Iterating this cursor will cause new queries to be sent to the server, even
584     * if the resultant data has already been retrieved by this cursor.
585     */
586    rewind(): void {
587      if (!this[kInitialized]) {
588        return;
589      }
590  
591      this[kId] = null;
592      this[kDocuments].clear();
593      this[kClosed] = false;
594      this[kKilled] = false;
595      this[kInitialized] = false;
596  
597      const session = this[kSession];
598      if (session) {
599        // We only want to end this session if we created it, and it hasn't ended yet
600        if (session.explicit === false) {
601          if (!session.hasEnded) {
602            session.endSession().catch(() => null);
603          }
604          this[kSession] = this.client.startSession({ owner: this, explicit: false });
605        }
606      }
607    }

629    /**
630     * @internal
631     *
632     * This function is exposed for the unified test runner's createChangeStream
633     * operation.  We cannot refactor to use the abstract _initialize method without
634     * a significant refactor.
635     */
636    async [kInit](): Promise<void> {
637      try {
638        const state = await this._initialize(this[kSession]);
639        const response = state.response;
640        this[kServer] = state.server;
641        if (response.cursor) {
642          // TODO(NODE-2674): Preserve int64 sent from MongoDB
643          this[kId] =
644            typeof response.cursor.id === 'number'
645              ? Long.fromNumber(response.cursor.id)
646              : typeof response.cursor.id === 'bigint'
647              ? Long.fromBigInt(response.cursor.id)
648              : response.cursor.id;
649  
650          if (response.cursor.ns) {
651            this[kNamespace] = ns(response.cursor.ns);
652          }
653  
654          this[kDocuments].pushMany(response.cursor.firstBatch);
655        }
656  
657        // When server responses return without a cursor document, we close this cursor
658        // and return the raw server response. This is often the case for explain commands
659        // for example
660        if (this[kId] == null) {
661          this[kId] = Long.ZERO;
662          // TODO(NODE-3286): ExecutionResult needs to accept a generic parameter
663          this[kDocuments].push(state.response as TODO_NODE_3286);
664        }
665  
666        // the cursor is now initialized, even if it is dead
667        this[kInitialized] = true;
668      } catch (error) {
669        // the cursor is now initialized, even if an error occurred
670        this[kInitialized] = true;
671        await cleanupCursor(this, { error });
672        throw error;
673      }
674  
675      if (this.isDead) {
676        await cleanupCursor(this, undefined);
677      }
678  
679      return;
680    }

683  /**
684   * @param cursor - the cursor on which to call `next`
685   * @param blocking - a boolean indicating whether or not the cursor should `block` until data
686   *     is available.  Generally, this flag is set to `false` because if the getMore returns no documents,
687   *     the cursor has been exhausted.  In certain scenarios (ChangeStreams, tailable await cursors and
688   *     `tryNext`, for example) blocking is necessary because a getMore returning no documents does
689   *     not indicate the end of the cursor.
690   * @param transform - if true, the cursor's transform function is applied to the result document (if the transform exists)
691   * @returns the next document in the cursor, or `null`.  When `blocking` is `true`, a `null` document means
692   * the cursor has been exhausted.  Otherwise, it means that there is no document available in the cursor's buffer.
693   */
694  async function next<T>(
695    cursor: AbstractCursor<T>,
696    {
697      blocking,
698      transform
699    }: {
700      blocking: boolean;
701      transform: boolean;
702    }
703  ): Promise<T | null> {
704    if (cursor.closed) {
705      return null;
706    }
707  
708    do {
709      if (cursor[kId] == null) {
710        // All cursors must operate within a session, one must be made implicitly if not explicitly provided
711        await cursor[kInit]();
712      }
713  
714      if (cursor[kDocuments].length !== 0) {
715        const doc = cursor[kDocuments].shift();
716  
717        if (doc != null && transform && cursor[kTransform]) {
718          try {
719            return cursor[kTransform](doc);
720          } catch (error) {
721            // `cleanupCursorAsync` should never throw, but if it does we want to throw the original
722            // error instead.
723            await cleanupCursor(cursor, { error, needsToEmitClosed: true }).catch(() => null);
724            throw error;
725          }
726        }
727  
728        return doc;
729      }
730  
731      if (cursor.isDead) {
732        // if the cursor is dead, we clean it up
733        // cleanupCursorAsync should never throw, but if it does it indicates a bug in the driver
734        // and we should surface the error
735        await cleanupCursor(cursor, {});
736        return null;
737      }
738  
739      // otherwise need to call getMore
740      const batchSize = cursor[kOptions].batchSize || 1000;
741  
742      try {
743        const response = await cursor.getMore(batchSize);
744  
745        if (response) {
746          const cursorId =
747            typeof response.cursor.id === 'number'
748              ? Long.fromNumber(response.cursor.id)
749              : typeof response.cursor.id === 'bigint'
750              ? Long.fromBigInt(response.cursor.id)
751              : response.cursor.id;
752  
753          cursor[kDocuments].pushMany(response.cursor.nextBatch);
754          cursor[kId] = cursorId;
755        }
756      } catch (error) {
757        // `cleanupCursorAsync` should never throw, but if it does we want to throw the original
758        // error instead.
759        await cleanupCursor(cursor, { error }).catch(() => null);
760        throw error;
761      }
762  
763      if (cursor.isDead) {
764        // If we successfully received a response from a cursor BUT the cursor indicates that it is exhausted,
765        // we intentionally clean up the cursor to release its session back into the pool before the cursor
766        // is iterated.  This prevents a cursor that is exhausted on the server from holding
767        // onto a session indefinitely until the AbstractCursor is iterated.
768        //
769        // cleanupCursorAsync should never throw, but if it does it indicates a bug in the driver
770        // and we should surface the error
771        await cleanupCursor(cursor, {});
772      }
773  
774      if (cursor[kDocuments].length === 0 && blocking === false) {
775        return null;
776      }
777    } while (!cursor.isDead || cursor[kDocuments].length !== 0);
778  
779    return null;
780  }

782  async function cleanupCursor(
783    cursor: AbstractCursor,
784    options: { error?: AnyError | undefined; needsToEmitClosed?: boolean } | undefined
785  ): Promise<void> {
786    const cursorId = cursor[kId];
787    const cursorNs = cursor[kNamespace];
788    const server = cursor[kServer];
789    const session = cursor[kSession];
790    const error = options?.error;
791  
792    // Cursors only emit closed events once the client-side cursor has been exhausted fully or there
793    // was an error.  Notably, when the server returns a cursor id of 0 and a non-empty batch, we
794    // cleanup the cursor but don't emit a `close` event.
795    const needsToEmitClosed = options?.needsToEmitClosed ?? cursor[kDocuments].length === 0;
796  
797    if (error) {
798      if (cursor.loadBalanced && error instanceof MongoNetworkError) {
799        return completeCleanup();
800      }
801    }
802  
803    if (cursorId == null || server == null || cursorId.isZero() || cursorNs == null) {
804      if (needsToEmitClosed) {
805        cursor[kClosed] = true;
806        cursor[kId] = Long.ZERO;
807        cursor.emit(AbstractCursor.CLOSE);
808      }
809  
810      if (session) {
811        if (session.owner === cursor) {
812          await session.endSession({ error });
813          return;
814        }
815  
816        if (!session.inTransaction()) {
817          maybeClearPinnedConnection(session, { error });
818        }
819      }
820  
821      return;
822    }
823  
824    async function completeCleanup() {
825      if (session) {
826        if (session.owner === cursor) {
827          try {
828            await session.endSession({ error });
829          } finally {
830            cursor.emit(AbstractCursor.CLOSE);
831          }
832          return;
833        }
834  
835        if (!session.inTransaction()) {
836          maybeClearPinnedConnection(session, { error });
837        }
838      }
839  
840      cursor.emit(AbstractCursor.CLOSE);
841      return;
842    }
843  
844    cursor[kKilled] = true;
845  
846    if (session.hasEnded) {
847      return completeCleanup();
848    }
849  
850    try {
851      await executeOperation(
852        cursor[kClient],
853        new KillCursorsOperation(cursorId, cursorNs, server, { session })
854      ).catch(() => null);
855    } finally {
856      await completeCleanup();
857    }
858  }

867  class ReadableCursorStream extends Readable {
868    private _cursor: AbstractCursor;
869    private _readInProgress = false;
870  
871    constructor(cursor: AbstractCursor) {
872      super({
873        objectMode: true,
874        autoDestroy: false,
875        highWaterMark: 1
876      });
877      this._cursor = cursor;
878    }
879  
880    // eslint-disable-next-line @typescript-eslint/no-unused-vars
881    override _read(size: number): void {
882      if (!this._readInProgress) {
883        this._readInProgress = true;
884        this._readNext();
885      }
886    }
887  
888    override _destroy(error: Error | null, callback: (error?: Error | null) => void): void {
889      this._cursor.close().then(
890        () => callback(error),
891        closeError => callback(closeError)
892      );
893    }
894  
895    private _readNext() {
896      next(this._cursor, { blocking: true, transform: true }).then(
897        result => {
898          if (result == null) {
899            this.push(null);
900          } else if (this.destroyed) {
901            this._cursor.close().catch(() => null);
902          } else {
903            if (this.push(result)) {
904              return this._readNext();
905            }
906  
907            this._readInProgress = false;
908          }
909        },
910        err => {
911          // NOTE: This is questionable, but we have a test backing the behavior. It seems the
912          //       desired behavior is that a stream ends cleanly when a user explicitly closes
913          //       a client during iteration. Alternatively, we could do the "right" thing and
914          //       propagate the error message by removing this special case.
915          if (err.message.match(/server is closed/)) {
916            this._cursor.close().catch(() => null);
917            return this.push(null);
918          }
919  
920          // NOTE: This is also perhaps questionable. The rationale here is that these errors tend
921          //       to be "operation was interrupted", where a cursor has been closed but there is an
922          //       active getMore in-flight. This used to check if the cursor was killed but once
923          //       that changed to happen in cleanup legitimate errors would not destroy the
924          //       stream. There are change streams test specifically test these cases.
925          if (err.message.match(/operation was interrupted/)) {
926            return this.push(null);
927          }
928  
929          // NOTE: The two above checks on the message of the error will cause a null to be pushed
930          //       to the stream, thus closing the stream before the destroy call happens. This means
931          //       that either of those error messages on a change stream will not get a proper
932          //       'error' event to be emitted (the error passed to destroy). Change stream resumability
933          //       relies on that error event to be emitted to create its new cursor and thus was not
934          //       working on 4.4 servers because the error emitted on failover was "interrupted at
935          //       shutdown" while on 5.0+ it is "The server is in quiesce mode and will shut down".
936          //       See NODE-4475.
937          return this.destroy(err);
938        }
939      );
940    }
941  }
