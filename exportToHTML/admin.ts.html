22   /**
23    * The **Admin** class is an internal class that allows convenient access to
24    * the admin functionality and commands for MongoDB.
25    *
26    * **ADMIN Cannot directly be instantiated**
27    * @public
28    *
29    * @example
30    * ```ts
31    * import { MongoClient } from 'mongodb';
32    *
33    * const client = new MongoClient('mongodb://localhost:27017');
34    * const admin = client.db().admin();
35    * const dbInfo = await admin.listDatabases();
36    * for (const db of dbInfo.databases) {
37    *   console.log(db.name);
38    * }
39    * ```
40    */
41   export class Admin {
42     /** @internal */
43     s: AdminPrivate;
44   
45     /**
46      * Create a new Admin instance
47      * @internal
48      */
49     constructor(db: Db) {
50       this.s = { db };
51     }
52   
53     /**
54      * Execute a command
55      *
56      * The driver will ensure the following fields are attached to the command sent to the server:
57      * - `lsid` - sourced from an implicit session or options.session
58      * - `$readPreference` - defaults to primary or can be configured by options.readPreference
59      * - `$db` - sourced from the name of this database
60      *
61      * If the client has a serverApi setting:
62      * - `apiVersion`
63      * - `apiStrict`
64      * - `apiDeprecationErrors`
65      *
66      * When in a transaction:
67      * - `readConcern` - sourced from readConcern set on the TransactionOptions
68      * - `writeConcern` - sourced from writeConcern set on the TransactionOptions
69      *
70      * Attaching any of the above fields to the command will have no effect as the driver will overwrite the value.
71      *
72      * @param command - The command to execute
73      * @param options - Optional settings for the command
74      */
75     async command(command: Document, options?: RunCommandOptions): Promise<Document> {
76       return executeOperation(
77         this.s.db.client,
78         new RunAdminCommandOperation(command, {
79           ...resolveBSONOptions(options),
80           session: options?.session,
81           readPreference: options?.readPreference
82         })
83       );
84     }
85   
86     /**
87      * Retrieve the server build information
88      *
89      * @param options - Optional settings for the command
90      */
91     async buildInfo(options?: CommandOperationOptions): Promise<Document> {
92       return this.command({ buildinfo: 1 }, options);
93     }

97      *

99      */
100    async serverInfo(options?: CommandOperationOptions): Promise<Document> {
101      return this.command({ buildinfo: 1 }, options);
102    }

104    /**
105     * Retrieve this db's server status.
106     *
107     * @param options - Optional settings for the command
108     */
109    async serverStatus(options?: CommandOperationOptions): Promise<Document> {
110      return this.command({ serverStatus: 1 }, options);
111    }

113    /**
114     * Ping the MongoDB server and retrieve results
115     *
116     * @param options - Optional settings for the command
117     */
118    async ping(options?: CommandOperationOptions): Promise<Document> {
119      return this.command({ ping: 1 }, options);
120    }

122    /**
123     * Remove a user from a database
124     *
125     * @param username - The username to remove
126     * @param options - Optional settings for the command
127     */
128    async removeUser(username: string, options?: RemoveUserOptions): Promise<boolean> {
129      return executeOperation(
130        this.s.db.client,
131        new RemoveUserOperation(this.s.db, username, { dbName: 'admin', ...options })
132      );
133    }
134  
135    /**
136     * Validate an existing collection
137     *
138     * @param collectionName - The name of the collection to validate.
139     * @param options - Optional settings for the command
140     */
141    async validateCollection(
142      collectionName: string,
143      options: ValidateCollectionOptions = {}
144    ): Promise<Document> {
145      return executeOperation(
146        this.s.db.client,
147        new ValidateCollectionOperation(this, collectionName, options)
148      );
149    }
150  
151    /**
152     * List the available databases
153     *
154     * @param options - Optional settings for the command
155     */
156    async listDatabases(options?: ListDatabasesOptions): Promise<ListDatabasesResult> {
157      return executeOperation(this.s.db.client, new ListDatabasesOperation(this.s.db, options));
158    }
159  
160    /**
161     * Get ReplicaSet status
162     *
163     * @param options - Optional settings for the command
164     */
165    async replSetGetStatus(options?: CommandOperationOptions): Promise<Document> {
166      return this.command({ replSetGetStatus: 1 }, options);
167    }
