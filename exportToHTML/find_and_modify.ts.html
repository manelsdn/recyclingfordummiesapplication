110  function configureFindAndModifyCmdBaseUpdateOpts(
111    cmdBase: FindAndModifyCmdBase,
112    options: FindOneAndReplaceOptions | FindOneAndUpdateOptions
113  ): FindAndModifyCmdBase {
114    cmdBase.new = options.returnDocument === ReturnDocument.AFTER;
115    cmdBase.upsert = options.upsert === true;
116  
117    if (options.bypassDocumentValidation === true) {
118      cmdBase.bypassDocumentValidation = options.bypassDocumentValidation;
119    }
120    return cmdBase;
121  }

123  /** @internal */
124  class FindAndModifyOperation extends CommandOperation<Document> {
125    override options: FindOneAndReplaceOptions | FindOneAndUpdateOptions | FindOneAndDeleteOptions;
126    cmdBase: FindAndModifyCmdBase;
127    collection: Collection;
128    query: Document;
129    doc?: Document;
130  
131    constructor(
132      collection: Collection,
133      query: Document,
134      options: FindOneAndReplaceOptions | FindOneAndUpdateOptions | FindOneAndDeleteOptions
135    ) {
136      super(collection, options);
137      this.options = options ?? {};
138      this.cmdBase = {
139        remove: false,
140        new: false,
141        upsert: false
142      };
143  
144      options.includeResultMetadata ??= false;
145  
146      const sort = formatSort(options.sort);
147      if (sort) {
148        this.cmdBase.sort = sort;
149      }
150  
151      if (options.projection) {
152        this.cmdBase.fields = options.projection;
153      }
154  
155      if (options.maxTimeMS) {
156        this.cmdBase.maxTimeMS = options.maxTimeMS;
157      }
158  
159      // Decorate the findAndModify command with the write Concern
160      if (options.writeConcern) {
161        this.cmdBase.writeConcern = options.writeConcern;
162      }
163  
164      if (options.let) {
165        this.cmdBase.let = options.let;
166      }
167  
168      // we check for undefined specifically here to allow falsy values
169      // eslint-disable-next-line no-restricted-syntax
170      if (options.comment !== undefined) {
171        this.cmdBase.comment = options.comment;
172      }
173  
174      // force primary read preference
175      this.readPreference = ReadPreference.primary;
176  
177      this.collection = collection;
178      this.query = query;
179    }
180  
181    override async execute(server: Server, session: ClientSession | undefined): Promise<Document> {
182      const coll = this.collection;
183      const query = this.query;
184      const options = { ...this.options, ...this.bsonOptions };
185  
186      // Create findAndModify command object
187      const cmd: Document = {
188        findAndModify: coll.collectionName,
189        query: query,
190        ...this.cmdBase
191      };
192  
193      // Have we specified collation
194      try {
195        decorateWithCollation(cmd, coll, options);
196      } catch (err) {
197        return err;
198      }
199  
200      if (options.hint) {
201        // TODO: once this method becomes a CommandOperation we will have the server
202        // in place to check.
203        const unacknowledgedWrite = this.writeConcern?.w === 0;
204        if (unacknowledgedWrite || maxWireVersion(server) < 8) {
205          throw new MongoCompatibilityError(
206            'The current topology does not support a hint on findAndModify commands'
207          );
208        }
209  
210        cmd.hint = options.hint;
211      }
212  
213      // Execute the command
214      const result = await super.executeCommand(server, session, cmd);
215      return options.includeResultMetadata ? result : result.value ?? null;
216    }
217  }

228      this.cmdBase.remove = true;

232  /** @internal */
233  export class FindOneAndReplaceOperation extends FindAndModifyOperation {
234    constructor(
235      collection: Collection,
236      filter: Document,
237      replacement: Document,
238      options: FindOneAndReplaceOptions
239    ) {
240      if (filter == null || typeof filter !== 'object') {
241        throw new MongoInvalidArgumentError('Argument "filter" must be an object');
242      }
243  
244      if (replacement == null || typeof replacement !== 'object') {
245        throw new MongoInvalidArgumentError('Argument "replacement" must be an object');
246      }
247  
248      if (hasAtomicOperators(replacement)) {
249        throw new MongoInvalidArgumentError('Replacement document must not contain atomic operators');
250      }
251  
252      super(collection, filter, options);
253      this.cmdBase.update = replacement;
254      configureFindAndModifyCmdBaseUpdateOpts(this.cmdBase, options);
255    }
256  }

258  /** @internal */
259  export class FindOneAndUpdateOperation extends FindAndModifyOperation {
260    constructor(
261      collection: Collection,
262      filter: Document,
263      update: Document,
264      options: FindOneAndUpdateOptions
265    ) {
266      if (filter == null || typeof filter !== 'object') {
267        throw new MongoInvalidArgumentError('Argument "filter" must be an object');
268      }
269  
270      if (update == null || typeof update !== 'object') {
271        throw new MongoInvalidArgumentError('Argument "update" must be an object');
272      }
273  
274      if (!hasAtomicOperators(update)) {
275        throw new MongoInvalidArgumentError('Update document requires atomic operators');
276      }
277  
278      super(collection, filter, options);
279      this.cmdBase.update = update;
280      configureFindAndModifyCmdBaseUpdateOpts(this.cmdBase, options);
281  
282      if (options.arrayFilters) {
283        this.cmdBase.arrayFilters = options.arrayFilters;
284      }
285    }
