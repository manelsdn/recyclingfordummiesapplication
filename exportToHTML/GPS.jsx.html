GPS.jsx 
1    import React, { useEffect, useRef } from "react";
2    import L from "leaflet"; // Import de Leaflet pour la manipulation de la carte
3    import "leaflet/dist/leaflet.css"; // Import des styles CSS de Leaflet
4    import "leaflet-routing-machine/dist/leaflet-routing-machine.css"; // Import des styles CSS du module de routage de Leaflet
5    import "leaflet-routing-machine"; // Import du module de routage de Leaflet
6    import "leaflet-control-geocoder/dist/Control.Geocoder.js"; // Import du module de géocodage de Leaflet
7    import "leaflet-control-geocoder/dist/Control.Geocoder.css"; // Import des styles CSS du module de géocodage de Leaflet
8    import "leaflet.locatecontrol"; // Import du module de localisation de Leaflet
9    import "leaflet.locatecontrol/dist/L.Control.Locate.css"; // Import des styles CSS du module de localisation de Leaflet
10   
11   const GPS = () => {
12       const mapRef = useRef(null); // Référence à l'élément de la carte
13       const controlRef = useRef(null); // Référence au contrôle de routage
14       const startMarkerRef = useRef(null); // Référence au marqueur de départ
15       const endMarkerRef = useRef(null); // Référence au marqueur de fin
16       const routeRef = useRef(null); // Référence à la route
17   
18       useEffect(() => {
19           if (!mapRef.current) {
20               // Création de la carte
21               mapRef.current = L.map("map").setView([45.77402, 4.86515], 10);
22   
23               // Ajout de la couche de tuiles OpenStreetMap à la carte
24               L.tileLayer("https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png", {
25                   maxZoom: 19,
26                   attribution:
27                       'Map data &copy; <a href="https://www.openstreetmap.org/">OpenStreetMap</a> contributors',
28               }).addTo(mapRef.current);
29           }
30   
31           // Remplacement des marqueurs par des icônes de position de FontAwesome
32           startMarkerRef.current = L.marker([45.77402, 4.86515], {
33               icon: L.divIcon({
34                   className: "icon-position-green", // Classe CSS pour styliser l'icône
35                   html: '<i class="fa fa-crosshairs"></i>', // Icône de FontAwesome
36               }),
37           }).addTo(mapRef.current);
38   
39           endMarkerRef.current = L.marker([45.75, 4.85], {
40               icon: L.divIcon({
41                   className: "icon-position-green", // Classe CSS pour styliser l'icône
42                   html: '<i class="fa fa-crosshairs"></i>', // Icône de FontAwesome
43               }),
44           }).addTo(mapRef.current);
45   
46           // Création du contrôle de routage
47           controlRef.current = L.Routing.control({
48               waypoints: [
49                   L.latLng(45.77402, 4.86515), // Point de départ
50                   L.latLng(45.75, 4.85), // Point d'arrivée (exemple)
51               ],
52               routeWhileDragging: true,
53               geocoder: L.Control.Geocoder.nominatim(),
54               show: false, // Ne pas afficher l'itinéraire par défaut
55               lineOptions: {
56                   styles: [{color: 'green', opacity: 0.8, weight: 6}] // Options de style de la ligne de l'itinéraire
57               }
58           }).addTo(mapRef.current);
59   
60           // Ajout du contrôle de localisation à la carte
61           L.control
62               .locate({
63                   position: "topleft",
64                   drawCircle: true,
65                   follow: true,
66                   setView: true,
67                   keepCurrentZoomLevel: false,
68                   stopFollowingOnDrag: false,
69                   remainActive: false,
70                   markerClass: L.circleMarker,
71                   circleStyle: {
72                       opacity: 0.5,
73                       fillOpacity: 0.1,
74                   },
75                   markerStyle: {
76                       opacity: 0.9,
77                       draggable: false,
78                   },
79                   icon: "fa fa-crosshairs",
80                   iconLoading: "fa fa-spinner fa-spin",
81                   circlePadding: [0, 0],
82                   metric: true,
83                   onLocationError: function (err) {
84                       alert(err.message);
85                   },
86                   onLocationOutsideMapBounds: function (context) {
87                       alert(context.options.strings.outsideMapBoundsMsg);
88                   },
89                   showPopup: true,
90                   strings: {
91                       title: "Show me where I am",
92                       metersUnit: "meters",
93                       feetUnit: "feet",
94                       popup: "You are within {distance} {unit} from this point",
95                       outsideMapBoundsMsg:
96                           "You seem located outside the boundaries of the map",
97                   },
98                   locateOptions: {},
99               })
100              .addTo(mapRef.current);
101  
102          return () => {
103              // Nettoyage des marqueurs et du contrôle de routage lorsque le composant est démonté
104              if (controlRef.current) {
105                  controlRef.current.remove();
106              }
107              if (startMarkerRef.current) {
108                  mapRef.current.removeLayer(startMarkerRef.current);
109              }
110              if (endMarkerRef.current) {
111                  mapRef.current.removeLayer(endMarkerRef.current);
112              }
113          };
114      }, []);
115  
116      return (
117          <div className="container">
118              <h2>Repérer les points de recyclage</h2>
119              {/* Conteneur de la carte */}
120              <div id="map" style={{ width: "100%", height: "500px" }}></div>
121          </div>
122      );
123  };
124  
125  export default GPS;
