1    import type { Db } from '../db';
2    import { MongoUnexpectedServerResponseError } from '../error';
3    import type { Server } from '../sdam/server';
4    import type { ClientSession } from '../sessions';
5    import { CommandOperation, type CommandOperationOptions } from './command';
6    
7    /** @public */
8    export type ProfilingLevelOptions = CommandOperationOptions;
9    
10   /** @internal */
11   export class ProfilingLevelOperation extends CommandOperation<string> {
12     override options: ProfilingLevelOptions;
13   
14     constructor(db: Db, options: ProfilingLevelOptions) {
15       super(db, options);
16       this.options = options;
17     }
18   
19     override async execute(server: Server, session: ClientSession | undefined): Promise<string> {
20       const doc = await super.executeCommand(server, session, { profile: -1 });
21       if (doc.ok === 1) {
22         const was = doc.was;
23         if (was === 0) return 'off';
24         if (was === 1) return 'slow_only';
25         if (was === 2) return 'all';
26         throw new MongoUnexpectedServerResponseError(`Illegal profiling level value ${was}`);
27       } else {
28         throw new MongoUnexpectedServerResponseError('Error with profile command');
29       }
30     }
31   }
