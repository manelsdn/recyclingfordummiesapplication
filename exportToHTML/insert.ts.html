15   /** @internal */
16   export class InsertOperation extends CommandOperation<Document> {
17     override options: BulkWriteOptions;
18     documents: Document[];
19   
20     constructor(ns: MongoDBNamespace, documents: Document[], options: BulkWriteOptions) {
21       super(undefined, options);
22       this.options = { ...options, checkKeys: options.checkKeys ?? false };
23       this.ns = ns;
24       this.documents = documents;
25     }
26   
27     override async execute(server: Server, session: ClientSession | undefined): Promise<Document> {
28       const options = this.options ?? {};
29       const ordered = typeof options.ordered === 'boolean' ? options.ordered : true;
30       const command: Document = {
31         insert: this.ns.collection,
32         documents: this.documents,
33         ordered
34       };
35   
36       if (typeof options.bypassDocumentValidation === 'boolean') {
37         command.bypassDocumentValidation = options.bypassDocumentValidation;
38       }
39   
40       // we check for undefined specifically here to allow falsy values
41       // eslint-disable-next-line no-restricted-syntax
42       if (options.comment !== undefined) {
43         command.comment = options.comment;
44       }
45   
46       return super.executeCommand(server, session, command);
47     }
48   }

53     bypassDocumentValidation?: boolean;
54     /** Force server to assign _id values instead of driver. */
55     forceServerObjectId?: boolean;

58   /** @public */
59   export interface InsertOneResult<TSchema = Document> {
60     /** Indicates whether this write result was acknowledged. If not, then all other members of this result will be undefined */
61     acknowledged: boolean;
62     /** The identifier that was inserted. If the server generated the identifier, this value will be null as the driver does not have access to that data */
63     insertedId: InferIdType<TSchema>;
64   }
65   
66   export class InsertOneOperation extends InsertOperation {
67     constructor(collection: Collection, doc: Document, options: InsertOneOptions) {
68       super(collection.s.namespace, prepareDocs(collection, [doc], options), options);
69     }
70   
71     override async execute(
72       server: Server,
73       session: ClientSession | undefined
74     ): Promise<InsertOneResult> {
75       const res = await super.execute(server, session);
76       if (res.code) throw new MongoServerError(res);
77       if (res.writeErrors) {
78         // This should be a WriteError but we can't change it now because of error hierarchy
79         throw new MongoServerError(res.writeErrors[0]);
80       }
81   
82       return {
83         acknowledged: this.writeConcern?.w !== 0,
84         insertedId: this.documents[0]._id
85       };
86     }
87   }

89   /** @public */
90   export interface InsertManyResult<TSchema = Document> {
91     /** Indicates whether this write result was acknowledged. If not, then all other members of this result will be undefined */
92     acknowledged: boolean;
93     /** The number of inserted documents for this operations */
94     insertedCount: number;
95     /** Map of the index of the inserted document to the id of the inserted document */
96     insertedIds: { [key: number]: InferIdType<TSchema> };
97   }
98   
99   /** @internal */
100  export class InsertManyOperation extends AbstractOperation<InsertManyResult> {
101    override options: BulkWriteOptions;
102    collection: Collection;
103    docs: Document[];
104  
105    constructor(collection: Collection, docs: Document[], options: BulkWriteOptions) {
106      super(options);
107  
108      if (!Array.isArray(docs)) {
109        throw new MongoInvalidArgumentError('Argument "docs" must be an array of documents');
110      }
111  
112      this.options = options;
113      this.collection = collection;
114      this.docs = docs;
115    }
116  
117    override async execute(
118      server: Server,
119      session: ClientSession | undefined
120    ): Promise<InsertManyResult> {
121      const coll = this.collection;
122      const options = { ...this.options, ...this.bsonOptions, readPreference: this.readPreference };
123      const writeConcern = WriteConcern.fromOptions(options);
124      const bulkWriteOperation = new BulkWriteOperation(
125        coll,
126        prepareDocs(coll, this.docs, options).map(document => ({ insertOne: { document } })),
127        options
128      );
129  
130      try {
131        const res = await bulkWriteOperation.execute(server, session);
132        return {
133          acknowledged: writeConcern?.w !== 0,
134          insertedCount: res.insertedCount,
135          insertedIds: res.insertedIds
136        };
137      } catch (err) {
138        if (err && err.message === 'Operation must be an object with an operation key') {
139          throw new MongoInvalidArgumentError(
140            'Collection.insertMany() cannot be called with an array that has null/undefined values'
141          );
142        }
143        throw err;
144      }
145    }
146  }
