31   export function nodejsMathRandomBytes(byteLength: number) {
32     return nodeJsByteUtils.fromNumberArray(
33       Array.from({ length: byteLength }, () => Math.floor(Math.random() * 256))
34     );
35   }

37   /**
38    * @internal
39    * WARNING: REQUIRE WILL BE REWRITTEN
40    *
41    * This code is carefully used by require_rewriter.mjs any modifications must be reflected in the plugin.
42    *
43    * @remarks
44    * "crypto" is the only dependency BSON needs. This presents a problem for creating a bundle of the BSON library
45    * in an es module format that can be used both on the browser and in Node.js. In Node.js when BSON is imported as
46    * an es module, there will be no global require function defined, making the code below fallback to the much less desireable math.random bytes.
47    * In order to make our es module bundle work as expected on Node.js we need to change this `require()` to a dynamic import, and the dynamic
48    * import must be top-level awaited since es modules are async. So we rely on a custom rollup plugin to seek out the following lines of code
49    * and replace `require` with `await import` and the IIFE line (`nodejsRandomBytes = (() => { ... })()`) with `nodejsRandomBytes = await (async () => { ... })()`
50    * when generating an es module bundle.
51    */
52   const nodejsRandomBytes: (byteLength: number) => Uint8Array = (() => {
53     try {
54       return require('crypto').randomBytes;
55     } catch {
56       return nodejsMathRandomBytes;
57     }
58   })();

60   /** @internal */
61   export const nodeJsByteUtils = {
62     toLocalBufferType(potentialBuffer: Uint8Array | NodeJsBuffer | ArrayBuffer): NodeJsBuffer {
63       if (Buffer.isBuffer(potentialBuffer)) {
64         return potentialBuffer;
65       }
66   
67       if (ArrayBuffer.isView(potentialBuffer)) {
68         return Buffer.from(
69           potentialBuffer.buffer,
70           potentialBuffer.byteOffset,
71           potentialBuffer.byteLength
72         );
73       }
74   
75       const stringTag =
76         potentialBuffer?.[Symbol.toStringTag] ?? Object.prototype.toString.call(potentialBuffer);
77       if (
78         stringTag === 'ArrayBuffer' ||
79         stringTag === 'SharedArrayBuffer' ||
80         stringTag === '[object ArrayBuffer]' ||
81         stringTag === '[object SharedArrayBuffer]'
82       ) {
83         return Buffer.from(potentialBuffer);
84       }
85   
86       throw new BSONError(`Cannot create Buffer from ${String(potentialBuffer)}`);
87     },
88   
89     allocate(size: number): NodeJsBuffer {
90       return Buffer.alloc(size);
91     },
92   
93     allocateUnsafe(size: number): NodeJsBuffer {
94       return Buffer.allocUnsafe(size);
95     },
96   
97     equals(a: Uint8Array, b: Uint8Array): boolean {
98       return nodeJsByteUtils.toLocalBufferType(a).equals(b);
99     },

131    toUTF8(buffer: Uint8Array, start: number, end: number, fatal: boolean): string {
132      const basicLatin = end - start <= 20 ? tryReadBasicLatin(buffer, start, end) : null;
133      if (basicLatin != null) {
134        return basicLatin;
135      }
136  
137      const string = nodeJsByteUtils.toLocalBufferType(buffer).toString('utf8', start, end);
138      if (fatal) {
139        // TODO(NODE-4930): Insufficiently strict BSON UTF8 validation
140        for (let i = 0; i < string.length; i++) {
141          if (string.charCodeAt(i) === 0xfffd) {
142            if (!validateUtf8(buffer, start, end)) {
143              throw new BSONError('Invalid UTF-8 string in BSON document');
144            }
145            break;
146          }
147        }
148      }
149      return string;
150    },
151  
152    utf8ByteLength(input: string): number {
153      return Buffer.byteLength(input, 'utf8');
154    },
155  
156    encodeUTF8Into(buffer: Uint8Array, source: string, byteOffset: number): number {
157      const latinBytesWritten = tryWriteBasicLatin(buffer, source, byteOffset);
158      if (latinBytesWritten != null) {
159        return latinBytesWritten;
160      }
161  
162      return nodeJsByteUtils.toLocalBufferType(buffer).write(source, byteOffset, undefined, 'utf8');
163    },

165    randomBytes: nodejsRandomBytes
166  };
