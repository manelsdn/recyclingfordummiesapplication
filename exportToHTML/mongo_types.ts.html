21   /** Given an object shaped type, return the type of the _id field or default to ObjectId @public */
22   export type InferIdType<TSchema> = TSchema extends { _id: infer IdType }
23     ? // user has defined a type for _id
24       Record<any, never> extends IdType
25       ? never // explicitly forbid empty objects as the type of _id
26       : IdType
27     : TSchema extends { _id?: infer IdType }
28     ? // optional _id defined - return ObjectId | IdType
29       unknown extends IdType
30       ? ObjectId // infer the _id type as ObjectId if the type of _id is unknown
31       : IdType
32     : ObjectId; // user has not defined _id on schema

43   /**
44    * Adds an optional _id field to an object shaped type, unless the _id field is required on that type.
45    * In the case _id is required, this method continues to require_id.
46    *
47    * @public
48    *
49    * @privateRemarks
50    * `ObjectId extends TSchema['_id']` is a confusing ordering at first glance. Rather than ask
51    * `TSchema['_id'] extends ObjectId` which translated to "Is the _id property ObjectId?"
52    * we instead ask "Does ObjectId look like (have the same shape) as the _id?"

54   export type OptionalUnlessRequiredId<TSchema> = TSchema extends { _id: any }
55     ? TSchema
56     : OptionalId<TSchema>;

58   /** TypeScript Omit (Exclude to be specific) does not work for objects with an "any" indexed type, and breaks discriminated unions @public */
59   export type EnhancedOmit<TRecordOrUnion, KeyUnion> = string extends keyof TRecordOrUnion
60     ? TRecordOrUnion // TRecordOrUnion has indexed type e.g. { _id: string; [k: string]: any; } or it is "any"
61     : TRecordOrUnion extends any
62     ? Pick<TRecordOrUnion, Exclude<keyof TRecordOrUnion, KeyUnion>> // discriminated unions

425  export type Join<T extends unknown[], D extends string> = T extends []
426    ? ''
427    : T extends [string | number]
428    ? `${T[0]}`
429    : T extends [string | number, ...infer R]
430    ? `${T[0]}${D}${Join<R, D>}`
431    : string;

433  /** @public */
434  export type PropertyType<Type, Property extends string> = string extends Property
435    ? unknown
436    : Property extends keyof Type
437    ? Type[Property]
438    : Property extends `${number}`
439    ? Type extends ReadonlyArray<infer ArrayType>
440      ? ArrayType
441      : unknown
442    : Property extends `${infer Key}.${infer Rest}`
443    ? Key extends `${number}`
444      ? Type extends ReadonlyArray<infer ArrayType>
445        ? PropertyType<ArrayType, Rest>
446        : unknown
447      : Key extends keyof Type
448      ? Type[Key] extends Map<string, infer MapType>
449        ? MapType
450        : PropertyType<Type[Key], Rest>
451      : unknown
452    : unknown;

454  /**
455   * @public
456   * returns tuple of strings (keys to be joined on '.') that represent every path into a schema
457   * https://www.mongodb.com/docs/manual/tutorial/query-embedded-documents/
458   *
459   * @remarks
460   * Through testing we determined that a depth of 8 is safe for the typescript compiler
461   * and provides reasonable compilation times. This number is otherwise not special and
462   * should be changed if issues are found with this level of checking. Beyond this
463   * depth any helpers that make use of NestedPaths should devolve to not asserting any
464   * type safety on the input.
465   */
466  export type NestedPaths<Type, Depth extends number[]> = Depth['length'] extends 8
467    ? []
468    : Type extends
469        | string
470        | number
471        | bigint
472        | boolean
473        | Date
474        | RegExp
475        | Buffer
476        | Uint8Array
477        | ((...args: any[]) => any)
478        | { _bsontype: string }
479    ? []
480    : Type extends ReadonlyArray<infer ArrayType>
481    ? [] | [number, ...NestedPaths<ArrayType, [...Depth, 1]>]
482    : Type extends Map<string, any>
483    ? [string]
484    : Type extends object
485    ? {
486        [Key in Extract<keyof Type, string>]: Type[Key] extends Type // type of value extends the parent
487          ? [Key]
488          : // for a recursive union type, the child will never extend the parent type.
489          // but the parent will still extend the child
490          Type extends Type[Key]
491          ? [Key]
492          : Type[Key] extends ReadonlyArray<infer ArrayType> // handling recursive types with arrays
493          ? Type extends ArrayType // is the type of the parent the same as the type of the array?
494            ? [Key] // yes, it's a recursive array type
495            : // for unions, the child type extends the parent
496            ArrayType extends Type
497            ? [Key] // we have a recursive array union
498            : // child is an array, but it's not a recursive array
499              [Key, ...NestedPaths<Type[Key], [...Depth, 1]>]
500          : // child is not structured the same as the parent
501            [Key, ...NestedPaths<Type[Key], [...Depth, 1]>] | [Key];
502      }[Extract<keyof Type, string>]
503    : [];
504
