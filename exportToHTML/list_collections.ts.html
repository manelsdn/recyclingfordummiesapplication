9    /** @public */
10   export interface ListCollectionsOptions extends Omit<CommandOperationOptions, 'writeConcern'> {
11     /** Since 4.0: If true, will only return the collection name in the response, and will omit additional info */
12     nameOnly?: boolean;
13     /** Since 4.0: If true and nameOnly is true, allows a user without the required privilege (i.e. listCollections action on the database) to run the command when access control is enforced. */
14     authorizedCollections?: boolean;
15     /** The batchSize for the returned command cursor or if pre 2.8 the systems batch collection */
16     batchSize?: number;
17   }
18   
19   /** @internal */
20   export class ListCollectionsOperation extends CommandOperation<Document> {
21     /**
22      * @remarks WriteConcern can still be present on the options because
23      * we inherit options from the client/db/collection.  The
24      * key must be present on the options in order to delete it.
25      * This allows typescript to delete the key but will
26      * not allow a writeConcern to be assigned as a property on options.
27      */
28     override options: ListCollectionsOptions & { writeConcern?: never };
29     db: Db;
30     filter: Document;
31     nameOnly: boolean;
32     authorizedCollections: boolean;
33     batchSize?: number;
34   
35     constructor(db: Db, filter: Document, options?: ListCollectionsOptions) {
36       super(db, options);
37   
38       this.options = { ...options };
39       delete this.options.writeConcern;
40       this.db = db;
41       this.filter = filter;
42       this.nameOnly = !!this.options.nameOnly;
43       this.authorizedCollections = !!this.options.authorizedCollections;
44   
45       if (typeof this.options.batchSize === 'number') {
46         this.batchSize = this.options.batchSize;
47       }
48     }
49   
50     override async execute(server: Server, session: ClientSession | undefined): Promise<Document> {
51       return super.executeCommand(server, session, this.generateCommand(maxWireVersion(server)));
52     }
53   
54     /* This is here for the purpose of unit testing the final command that gets sent. */
55     generateCommand(wireVersion: number): Document {
56       const command: Document = {
57         listCollections: 1,
58         filter: this.filter,
59         cursor: this.batchSize ? { batchSize: this.batchSize } : {},
60         nameOnly: this.nameOnly,
61         authorizedCollections: this.authorizedCollections
62       };
63   
64       // we check for undefined specifically here to allow falsy values
65       // eslint-disable-next-line no-restricted-syntax
66       if (wireVersion >= 9 && this.options.comment !== undefined) {
67         command.comment = this.options.comment;
68       }
69   
70       return command;
71     }
72   }
