339    onMessage(message: OpMsgResponse | OpQueryResponse) {
340      const delayedTimeoutId = this[kDelayedTimeoutId];
341      if (delayedTimeoutId != null) {
342        clearTimeout(delayedTimeoutId);
343        this[kDelayedTimeoutId] = null;
344      }
345  
346      const socketTimeoutMS = this[kStream].timeout ?? 0;
347      this[kStream].setTimeout(0);
348  
349      // always emit the message, in case we are streaming
350      this.emit('message', message);
351      let operationDescription = this[kQueue].get(message.responseTo);
352  
353      if (!operationDescription && this.isMonitoringConnection) {
354        // This is how we recover when the initial hello's requestId is not
355        // the responseTo when hello responses have been skipped:
356  
357        // First check if the map is of invalid size
358        if (this[kQueue].size > 1) {
359          this.cleanup(true, new MongoRuntimeError(INVALID_QUEUE_SIZE));
360        } else {
361          // Get the first orphaned operation description.
362          const entry = this[kQueue].entries().next();
363          if (entry.value != null) {
364            const [requestId, orphaned]: [number, OperationDescription] = entry.value;
365            // If the orphaned operation description exists then set it.
366            operationDescription = orphaned;
367            // Remove the entry with the bad request id from the queue.
368            this[kQueue].delete(requestId);
369          }
370        }
371      }
372  
373      if (!operationDescription) {
374        return;
375      }
376  
377      const callback = operationDescription.cb;
378  
379      // SERVER-45775: For exhaust responses we should be able to use the same requestId to
380      // track response, however the server currently synthetically produces remote requests
381      // making the `responseTo` change on each response
382      this[kQueue].delete(message.responseTo);
383      if ('moreToCome' in message && message.moreToCome) {
384        // If the operation description check above does find an orphaned
385        // description and sets the operationDescription then this line will put one
386        // back in the queue with the correct requestId and will resolve not being able
387        // to find the next one via the responseTo of the next streaming hello.
388        this[kQueue].set(message.requestId, operationDescription);
389        this[kStream].setTimeout(socketTimeoutMS);
390      }
391  
392      try {
393        // Pass in the entire description because it has BSON parsing options
394        message.parse(operationDescription);
395      } catch (err) {
396        // If this error is generated by our own code, it will already have the correct class applied
397        // if it is not, then it is coming from a catastrophic data parse failure or the BSON library
398        // in either case, it should not be wrapped
399        callback(err);
400        return;
401      }
402  
403      if (message.documents[0]) {
404        const document: Document = message.documents[0];
405        const session = operationDescription.session;
406        if (session) {
407          updateSessionFromResponse(session, document);
408        }
409  
410        if (document.$clusterTime) {
411          this[kClusterTime] = document.$clusterTime;
412          this.emit(Connection.CLUSTER_TIME_RECEIVED, document.$clusterTime);
413        }
414  
415        if (document.writeConcernError) {
416          callback(new MongoWriteConcernError(document.writeConcernError, document), document);
417          return;
418        }
419  
420        if (document.ok === 0 || document.$err || document.errmsg || document.code) {
421          callback(new MongoServerError(document));
422          return;
423        }
424      }
425  
426      callback(undefined, message.documents[0]);
427    }

429    destroy(options: DestroyOptions, callback?: Callback): void {
430      if (this.closed) {
431        process.nextTick(() => callback?.());
432        return;
433      }
434      if (typeof callback === 'function') {
435        this.once('close', () => process.nextTick(() => callback()));
436      }
437  
438      // load balanced mode requires that these listeners remain on the connection
439      // after cleanup on timeouts, errors or close so we remove them before calling
440      // cleanup.
441      this.removeAllListeners(Connection.PINNED);
442      this.removeAllListeners(Connection.UNPINNED);
443      const message = `connection ${this.id} to ${this.address} closed`;
444      this.cleanup(options.force, new MongoNetworkError(message));
445    }

447    /**
448     * A method that cleans up the connection.  When `force` is true, this method
449     * forcibly destroys the socket.
450     *
451     * If an error is provided, any in-flight operations will be closed with the error.
452     *
453     * This method does nothing if the connection is already closed.
454     */
455    private cleanup(force: boolean, error?: Error): void {
456      if (this.closed) {
457        return;
458      }
459  
460      this.closed = true;
461  
462      const completeCleanup = () => {
463        for (const op of this[kQueue].values()) {
464          op.cb(error);
465        }
466  
467        this[kQueue].clear();
468  
469        this.emit(Connection.CLOSE);
470      };
471  
472      this[kStream].removeAllListeners();
473      this[kMessageStream].removeAllListeners();
474  
475      this[kMessageStream].destroy();
476  
477      if (force) {
478        this[kStream].destroy();
479        completeCleanup();
480        return;
481      }
482  
483      if (!this[kStream].writableEnded) {
484        this[kStream].end(() => {
485          this[kStream].destroy();
486          completeCleanup();
487        });
488      } else {
489        completeCleanup();
490      }
491    }

493    command(
494      ns: MongoDBNamespace,
495      command: Document,
496      options: CommandOptions | undefined,
497      callback: Callback
498    ): void {
499      let cmd = { ...command };
500  
501      const readPreference = getReadPreference(options);
502      const shouldUseOpMsg = supportsOpMsg(this);
503      const session = options?.session;
504  
505      let clusterTime = this.clusterTime;
506  
507      if (this.serverApi) {
508        const { version, strict, deprecationErrors } = this.serverApi;
509        cmd.apiVersion = version;
510        if (strict != null) cmd.apiStrict = strict;
511        if (deprecationErrors != null) cmd.apiDeprecationErrors = deprecationErrors;
512      }
513  
514      if (hasSessionSupport(this) && session) {
515        if (
516          session.clusterTime &&
517          clusterTime &&
518          session.clusterTime.clusterTime.greaterThan(clusterTime.clusterTime)
519        ) {
520          clusterTime = session.clusterTime;
521        }
522  
523        const err = applySession(session, cmd, options);
524        if (err) {
525          return callback(err);
526        }
527      } else if (session?.explicit) {
528        return callback(new MongoCompatibilityError('Current topology does not support sessions'));
529      }
530  
531      // if we have a known cluster time, gossip it
532      if (clusterTime) {
533        cmd.$clusterTime = clusterTime;
534      }
535  
536      if (isSharded(this) && !shouldUseOpMsg && readPreference && readPreference.mode !== 'primary') {
537        cmd = {
538          $query: cmd,
539          $readPreference: readPreference.toJSON()
540        };
541      }
542  
543      const commandOptions: Document = Object.assign(
544        {
545          numberToSkip: 0,
546          numberToReturn: -1,
547          checkKeys: false,
548          // This value is not overridable
549          secondaryOk: readPreference.secondaryOk()
550        },
551        options
552      );
553  
554      const message = shouldUseOpMsg
555        ? new OpMsgRequest(ns.db, cmd, commandOptions)
556        : new OpQueryRequest(ns.db, cmd, commandOptions);
557  
558      try {
559        write(this, message, commandOptions, callback);
560      } catch (err) {
561        callback(err);
562      }
563    }
564  }

568    /** @internal */
569    [kAutoEncrypter]?: AutoEncrypter;
570  
571    constructor(stream: Stream, options: ConnectionOptions) {
572      super(stream, options);
573      this[kAutoEncrypter] = options.autoEncrypter;
574    }
575  
576    /** @internal @override */
577    override command(
578      ns: MongoDBNamespace,
579      cmd: Document,
580      options: CommandOptions,
581      callback: Callback
582    ): void {
583      const autoEncrypter = this[kAutoEncrypter];
584      if (!autoEncrypter) {
585        return callback(new MongoMissingDependencyError('No AutoEncrypter available for encryption'));
586      }
587  
588      const serverWireVersion = maxWireVersion(this);
589      if (serverWireVersion === 0) {
590        // This means the initial handshake hasn't happened yet
591        return super.command(ns, cmd, options, callback);
592      }
593  
594      if (serverWireVersion < 8) {
595        callback(
596          new MongoCompatibilityError('Auto-encryption requires a minimum MongoDB version of 4.2')
597        );
598        return;
599      }
600  
601      // Save sort or indexKeys based on the command being run
602      // the encrypt API serializes our JS objects to BSON to pass to the native code layer
603      // and then deserializes the encrypted result, the protocol level components
604      // of the command (ex. sort) are then converted to JS objects potentially losing
605      // import key order information. These fields are never encrypted so we can save the values
606      // from before the encryption and replace them after encryption has been performed
607      const sort: Map<string, number> | null = cmd.find || cmd.findAndModify ? cmd.sort : null;
608      const indexKeys: Map<string, number>[] | null = cmd.createIndexes
609        ? cmd.indexes.map((index: { key: Map<string, number> }) => index.key)
610        : null;
611  
612      autoEncrypter.encrypt(ns.toString(), cmd, options).then(
613        encrypted => {
614          // Replace the saved values
615          if (sort != null && (cmd.find || cmd.findAndModify)) {
616            encrypted.sort = sort;
617          }
618          if (indexKeys != null && cmd.createIndexes) {
619            for (const [offset, index] of indexKeys.entries()) {
620              // @ts-expect-error `encrypted` is a generic "command", but we've narrowed for only `createIndexes` commands here
621              encrypted.indexes[offset].key = index;
622            }
623          }
624  
625          super.command(ns, encrypted, options, (err, response) => {
626            if (err || response == null) {
627              callback(err, response);
628              return;
629            }
630  
631            autoEncrypter.decrypt(response, options).then(
632              res => callback(undefined, res),
633              err => callback(err)
634            );
635          });
636        },
637        err => {
638          if (err) {
639            callback(err, null);
640          }
641        }
642      );
643    }
644  }

652  function supportsOpMsg(conn: Connection) {
653    const description = conn.description;
654    if (description == null) {
655      return false;
656    }
657  
658    return maxWireVersion(conn) >= 6 && !description.__nodejs_mock_server__;

661  function streamIdentifier(stream: Stream, options: ConnectionOptions): string {
662    if (options.proxyHost) {
663      // If proxy options are specified, the properties of `stream` itself
664      // will not accurately reflect what endpoint this is connected to.
665      return options.hostAddress.toString();
666    }
667  
668    const { remoteAddress, remotePort } = stream;
669    if (typeof remoteAddress === 'string' && typeof remotePort === 'number') {
670      return HostAddress.fromHostPort(remoteAddress, remotePort).toString();
671    }
672  
673    return uuidV4().toString('hex');
674  }

676  function write(
677    conn: Connection,
678    command: WriteProtocolMessageType,
679    options: CommandOptions,
680    callback: Callback
681  ) {
682    options = options ?? {};
683    const operationDescription: OperationDescription = {
684      requestId: command.requestId,
685      cb: callback,
686      session: options.session,
687      noResponse: typeof options.noResponse === 'boolean' ? options.noResponse : false,
688      documentsReturnedIn: options.documentsReturnedIn,
689  
690      // for BSON parsing
691      useBigInt64: typeof options.useBigInt64 === 'boolean' ? options.useBigInt64 : false,
692      promoteLongs: typeof options.promoteLongs === 'boolean' ? options.promoteLongs : true,
693      promoteValues: typeof options.promoteValues === 'boolean' ? options.promoteValues : true,
694      promoteBuffers: typeof options.promoteBuffers === 'boolean' ? options.promoteBuffers : false,
695      bsonRegExp: typeof options.bsonRegExp === 'boolean' ? options.bsonRegExp : false,
696      enableUtf8Validation:
697        typeof options.enableUtf8Validation === 'boolean' ? options.enableUtf8Validation : true,
698      raw: typeof options.raw === 'boolean' ? options.raw : false,
699      started: 0
700    };
701  
702    if (conn[kDescription] && conn[kDescription].compressor) {
703      operationDescription.agreedCompressor = conn[kDescription].compressor;
704  
705      if (conn[kDescription].zlibCompressionLevel) {
706        operationDescription.zlibCompressionLevel = conn[kDescription].zlibCompressionLevel;
707      }
708    }
709  
710    if (typeof options.socketTimeoutMS === 'number') {
711      conn[kStream].setTimeout(options.socketTimeoutMS);
712    } else if (conn.socketTimeoutMS !== 0) {
713      conn[kStream].setTimeout(conn.socketTimeoutMS);
714    }
715  
716    // if command monitoring is enabled we need to modify the callback here
717    if (conn.monitorCommands) {
718      conn.emit(Connection.COMMAND_STARTED, new CommandStartedEvent(conn, command));
719  
720      operationDescription.started = now();
721      operationDescription.cb = (err, reply) => {
722        // Command monitoring spec states that if ok is 1, then we must always emit
723        // a command succeeded event, even if there's an error. Write concern errors
724        // will have an ok: 1 in their reply.
725        if (err && reply?.ok !== 1) {
726          conn.emit(
727            Connection.COMMAND_FAILED,
728            new CommandFailedEvent(conn, command, err, operationDescription.started)
729          );
730        } else {
731          if (reply && (reply.ok === 0 || reply.$err)) {
732            conn.emit(
733              Connection.COMMAND_FAILED,
734              new CommandFailedEvent(conn, command, reply, operationDescription.started)
735            );
736          } else {
737            conn.emit(
738              Connection.COMMAND_SUCCEEDED,
739              new CommandSucceededEvent(conn, command, reply, operationDescription.started)
740            );
741          }
742        }
743  
744        if (typeof callback === 'function') {
745          // Since we're passing through the reply with the write concern error now, we
746          // need it not to be provided to the original callback in this case so
747          // retryability does not get tricked into thinking the command actually
748          // succeeded.
749          callback(err, err instanceof MongoWriteConcernError ? undefined : reply);
750        }
751      };
752    }
753  
754    if (!operationDescription.noResponse) {
755      conn[kQueue].set(operationDescription.requestId, operationDescription);
756    }
757  
758    try {
759      conn[kMessageStream].writeCommand(command, operationDescription);
760    } catch (e) {
761      if (!operationDescription.noResponse) {
762        conn[kQueue].delete(operationDescription.requestId);
763        operationDescription.cb(e);
764        return;
765      }
766    }
767  
768    if (operationDescription.noResponse) {
769      operationDescription.cb();
770    }
771  }

965    onMessage(message: OpMsgResponse | OpQueryResponse) {
966      const delayedTimeoutId = this[kDelayedTimeoutId];
967      if (delayedTimeoutId != null) {
968        clearTimeout(delayedTimeoutId);
969        this[kDelayedTimeoutId] = null;
970      }
971  
972      const socketTimeoutMS = this.socket.timeout ?? 0;
973      this.socket.setTimeout(0);
974  
975      // always emit the message, in case we are streaming
976      this.emit('message', message);
977      let operationDescription = this[kQueue].get(message.responseTo);
978  
979      if (!operationDescription && this.isMonitoringConnection) {
980        // This is how we recover when the initial hello's requestId is not
981        // the responseTo when hello responses have been skipped:
982  
983        // First check if the map is of invalid size
984        if (this[kQueue].size > 1) {
985          this.cleanup(true, new MongoRuntimeError(INVALID_QUEUE_SIZE));
986        } else {
987          // Get the first orphaned operation description.
988          const entry = this[kQueue].entries().next();
989          if (entry.value != null) {
990            const [requestId, orphaned]: [number, OperationDescription] = entry.value;
991            // If the orphaned operation description exists then set it.
992            operationDescription = orphaned;
993            // Remove the entry with the bad request id from the queue.
994            this[kQueue].delete(requestId);
995          }
996        }
997      }
998  
999      if (!operationDescription) {
1000       return;
1001     }
1002 
1003     const callback = operationDescription.cb;
1004 
1005     // SERVER-45775: For exhaust responses we should be able to use the same requestId to
1006     // track response, however the server currently synthetically produces remote requests
1007     // making the `responseTo` change on each response
1008     this[kQueue].delete(message.responseTo);
1009     if ('moreToCome' in message && message.moreToCome) {
1010       // If the operation description check above does find an orphaned
1011       // description and sets the operationDescription then this line will put one
1012       // back in the queue with the correct requestId and will resolve not being able
1013       // to find the next one via the responseTo of the next streaming hello.
1014       this[kQueue].set(message.requestId, operationDescription);
1015       this.socket.setTimeout(socketTimeoutMS);
1016     }
1017 
1018     try {
1019       // Pass in the entire description because it has BSON parsing options
1020       message.parse(operationDescription);
1021     } catch (err) {
1022       // If this error is generated by our own code, it will already have the correct class applied
1023       // if it is not, then it is coming from a catastrophic data parse failure or the BSON library
1024       // in either case, it should not be wrapped
1025       callback(err);
1026       return;
1027     }
1028 
1029     if (message.documents[0]) {
1030       const document: Document = message.documents[0];
1031       const session = operationDescription.session;
1032       if (session) {
1033         updateSessionFromResponse(session, document);
1034       }
1035 
1036       if (document.$clusterTime) {
1037         this[kClusterTime] = document.$clusterTime;
1038         this.emit(Connection.CLUSTER_TIME_RECEIVED, document.$clusterTime);
1039       }
1040 
1041       if (document.writeConcernError) {
1042         callback(new MongoWriteConcernError(document.writeConcernError, document), document);
1043         return;
1044       }
1045 
1046       if (document.ok === 0 || document.$err || document.errmsg || document.code) {
1047         callback(new MongoServerError(document));
1048         return;
1049       }
1050     }
1051 
1052     callback(undefined, message.documents[0]);
1053   }

1055   destroy(options: DestroyOptions, callback?: Callback): void {
1056     if (this.closed) {
1057       process.nextTick(() => callback?.());
1058       return;
1059     }
1060     if (typeof callback === 'function') {
1061       this.once('close', () => process.nextTick(() => callback()));
1062     }
1063 
1064     // load balanced mode requires that these listeners remain on the connection
1065     // after cleanup on timeouts, errors or close so we remove them before calling
1066     // cleanup.
1067     this.removeAllListeners(Connection.PINNED);
1068     this.removeAllListeners(Connection.UNPINNED);
1069     const message = `connection ${this.id} to ${this.address} closed`;
1070     this.cleanup(options.force, new MongoNetworkError(message));
1071   }

1073   /**
1074    * A method that cleans up the connection.  When `force` is true, this method
1075    * forcibly destroys the socket.
1076    *
1077    * If an error is provided, any in-flight operations will be closed with the error.
1078    *
1079    * This method does nothing if the connection is already closed.
1080    */
1081   private cleanup(force: boolean, error?: Error): void {
1082     if (this.closed) {
1083       return;
1084     }
1085 
1086     this.closed = true;
1087 
1088     const completeCleanup = () => {
1089       for (const op of this[kQueue].values()) {
1090         op.cb(error);
1091       }
1092 
1093       this[kQueue].clear();
1094 
1095       this.emit(Connection.CLOSE);
1096     };
1097 
1098     this.socket.removeAllListeners();
1099     this[kMessageStream].removeAllListeners();
1100 
1101     this[kMessageStream].destroy();
1102 
1103     if (force) {
1104       this.socket.destroy();
1105       completeCleanup();
1106       return;
1107     }
1108 
1109     if (!this.socket.writableEnded) {
1110       this.socket.end(() => {
1111         this.socket.destroy();
1112         completeCleanup();
1113       });
1114     } else {
1115       completeCleanup();
1116     }
1117   }

1120     ns: MongoDBNamespace,
1121     command: Document,
1122     options: CommandOptions | undefined,
1123     callback: Callback
1124   ): void {
1125     let cmd = { ...command };

1128     const session = options?.session;
1129

1131

1134       cmd.apiVersion = version;
1135       if (strict != null) cmd.apiStrict = strict;
1136       if (deprecationErrors != null) cmd.apiDeprecationErrors = deprecationErrors;
1137     }
1138

1140       if (
1141         session.clusterTime &&
1142         clusterTime &&
1143         session.clusterTime.clusterTime.greaterThan(clusterTime.clusterTime)
1144       ) {
1145         clusterTime = session.clusterTime;
1146       }
1147 
1148       const err = applySession(session, cmd, options);
1149       if (err) {
1150         return callback(err);
1151       }
1152     } else if (session?.explicit) {
1153       return callback(new MongoCompatibilityError('Current topology does not support sessions'));
1154     }
1155 
1156     // if we have a known cluster time, gossip it
1157     if (clusterTime) {
1158       cmd.$clusterTime = clusterTime;
1159     }
1160 
1161     if (
1162       // @ts-expect-error ModernConnections cannot be passed as connections
1163       isSharded(this) &&
1164       !this.supportsOpMsg &&
1165       readPreference &&
1166       readPreference.mode !== 'primary'
1167     ) {
1168       cmd = {
1169         $query: cmd,
1170         $readPreference: readPreference.toJSON()
1171       };
1172     }
1173 
1174     const commandOptions: Document = Object.assign(
1175       {
1176         numberToSkip: 0,
1177         numberToReturn: -1,
1178         checkKeys: false,
1179         // This value is not overridable
1180         secondaryOk: readPreference.secondaryOk()
1181       },
1182       options
1183     );
1184 
1185     const message = this.supportsOpMsg
1186       ? new OpMsgRequest(ns.db, cmd, commandOptions)
1187       : new OpQueryRequest(ns.db, cmd, commandOptions);
1188 
1189     try {
1190       write(this as any as Connection, message, commandOptions, callback);
1191     } catch (err) {
1192       callback(err);
1193     }
1194   }

1199 /**
1200  * @internal
1201  *
1202  * This helper reads chucks of data out of a socket and buffers them until it has received a
1203  * full wire protocol message.
1204  *
1205  * By itself, produces an infinite async generator of wire protocol messages and consumers must end
1206  * the stream by calling `return` on the generator.
1207  *
1208  * Note that `for-await` loops call `return` automatically when the loop is exited.
1209  */
1210 export async function* readWireProtocolMessages(
1211   connection: ModernConnection
1212 ): AsyncGenerator<Buffer> {
1213   const bufferPool = new BufferPool();
1214   const maxBsonMessageSize = connection.hello?.maxBsonMessageSize ?? kDefaultMaxBsonMessageSize;
1215   for await (const [chunk] of on(connection.socket, 'data')) {
1216     bufferPool.append(chunk);
1217     const sizeOfMessage = bufferPool.getInt32();
1218 
1219     if (sizeOfMessage == null) {
1220       continue;
1221     }
1222 
1223     if (sizeOfMessage < 0) {
1224       throw new MongoParseError(`Invalid message size: ${sizeOfMessage}`);
1225     }
1226 
1227     if (sizeOfMessage > maxBsonMessageSize) {
1228       throw new MongoParseError(
1229         `Invalid message size: ${sizeOfMessage}, max allowed: ${maxBsonMessageSize}`
1230       );
1231     }
1232 
1233     if (sizeOfMessage > bufferPool.length) {
1234       continue;
1235     }
1236 
1237     yield bufferPool.read(sizeOfMessage);
1238   }
1239 }

1241 /**
1242  * @internal
1243  *
1244  * Writes an OP_MSG or OP_QUERY request to the socket, optionally compressing the command. This method
1245  * waits until the socket's buffer has emptied (the Nodejs socket `drain` event has fired).
1246  */
1247 export async function writeCommand(
1248   connection: ModernConnection,
1249   command: WriteProtocolMessageType,
1250   options: Partial<Pick<OperationDescription, 'agreedCompressor' | 'zlibCompressionLevel'>>
1251 ): Promise<void> {
1252   const drained = once(connection.socket, 'drain');
1253   const finalCommand =
1254     options.agreedCompressor === 'none' || !OpCompressedRequest.canCompress(command)
1255       ? command
1256       : new OpCompressedRequest(command, {
1257           agreedCompressor: options.agreedCompressor ?? 'none',
1258           zlibCompressionLevel: options.zlibCompressionLevel ?? 0
1259         });
1260   const buffer = Buffer.concat(await finalCommand.toBin());
1261   connection.socket.push(buffer);
1262   await drained;
1263 }

1265 /**
1266  * @internal
1267  *
1268  * Returns an async generator that yields full wire protocol messages from the underlying socket.  This function
1269  * yields messages until `moreToCome` is false or not present in a response, or the caller cancels the request
1270  * by calling `return` on the generator.
1271  *
1272  * Note that `for-await` loops call `return` automatically when the loop is exited.
1273  */
1274 export async function* readMany(
1275   connection: ModernConnection
1276 ): AsyncGenerator<OpMsgResponse | OpQueryResponse> {
1277   for await (const message of readWireProtocolMessages(connection)) {
1278     const response = await decompressResponse(message);
1279     yield response;
1280 
1281     if (!('moreToCome' in response) || !response.moreToCome) {
1282       return;
1283     }
1284   }
1285 }
